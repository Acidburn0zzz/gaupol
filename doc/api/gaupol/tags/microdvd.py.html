  19 """MicroDVD tag library.
  20 
  21 Module variables:
  22 
  23     _FLAGS: Common regular expression flags
  24 """
  25 
  26 
  27 import re
  28 from ._taglib import TagLibrary
  29 
  30 
  31 _FLAGS = re.MULTILINE | re.DOTALL
  32 
  33 
  34 class MicroDVD(TagLibrary):
  35 
  36     """MicroDVD tag library."""
  37 
  38     tag = r"\{[a-z]:.*?\}", re.IGNORECASE
  39     italic_tag = r"\{y:i\}", re.IGNORECASE
  40 
  41     decode_tags = [
  42         # Style x3 (single line)
  43         (r"\{y:(b|i|u)[^\}]*?(b|i|u)[^\}]*?(b|i|u)\}(.*?)$", _FLAGS,
  44             r"<\1><\2><\3>\4</\3></\2></\1>"),
  45         # Style x2 (single line)
  46         (r"\{y:(b|i|u)[^\}]*?(b|i|u)\}(.*?)$", _FLAGS,
  47             r"<\1><\2>\3</\2></\1>", 2),
  48         # Style x1 (single line)
  49         (r"\{y:(b|i|u)\}(.*?)$", _FLAGS,
  50             r"<\1>\2</\1>", 3),
  51         # Style x3 (whole subtitle)
  52         (r"\{Y:(b|i|u)[^\}]*?(b|i|u)[^\}]*?(b|i|u)\}(.*?)\Z", _FLAGS,
  53             r"<\1><\2><\3>\4</\3></\2></\1>"),
  54         # Style x2 (whole subtitle)
  55         (r"\{Y:(b|i|u)[^\}]*?(b|i|u)\}(.*?)\Z", _FLAGS,
  56             r"<\1><\2>\3</\2></\1>", 2),
  57         # Style x1 (whole subtitle)
  58         (r"\{Y:(b|i|u)\}(.*?)\Z", _FLAGS,
  59             r"<\1>\2</\1>", 3),
  60         # Color (single line)
  61         (r"\{c:\$([a-zA-Z0-9]{6})\}(.*?)$", _FLAGS,
  62             r'<color="#\1">\2</color>'),
  63         # Color (whole subtitle)
  64         (r"\{C:\$([a-zA-Z0-9]{6})\}(.*?)\Z", _FLAGS,
  65             r'<color="#\1">\2</color>'),
  66         # Font (single line)
  67         (r"\{f:(.*?)\}(.*?)$", _FLAGS,
  68             r'<font="\1">\2</font>'),
  69         # Font (whole subtitle)
  70         (r"\{F:(.*?)\}(.*?)\Z", _FLAGS,
  71             r'<font="\1">\2</font>'),
  72         # Size (single line)
  73         (r"\{s:(\d+)\}(.*?)$", _FLAGS,
  74             r'<size="\1">\2</size>'),
  75         # Size (whole subtitle)
  76         (r"\{S:(\d+)\}(.*?)\Z", _FLAGS,
  77             r'<size="\1">\2</size>'),
  78         # Remove all other tags.
  79         (r"\{.:.*?\}", re.IGNORECASE,
  80             r"")]
  81 
  82     encode_tags = [
  83         # Remove duplicate style tags (e.g. <b>test</b><b>test</b>).
  84         (r"</(b|i|u)>(\n?)<\1>", _FLAGS,
  85             r"\2", 3),
  86         # Remove other duplicate tags.
  87         (r"<(.*?)=(.*?)>(.*?)</\1>(\n?)<\1=\2>", _FLAGS,
  88             r"<\1=\2>\3\4", 3),
  89         # Style (affecting a single line subtitle fully)
  90         (r"\A<(b|i|u)>(.*?)</\1>\Z", re.MULTILINE,
  91             r"{Y:\1}\2", 3),
  92         # Style (affecting only one line)
  93         (r"<(b|i|u)>(.*?)</\1>", re.MULTILINE,
  94             r"{y:\1}\2", 3),
  95         # Style (affecting whole subtitle)
  96         (r"<(b|i|u)>(.*?)</\1>", _FLAGS,
  97             r"{Y:\1}\2", 3),
  98         # Color (affecting a single line subtitle fully)
  99         (r'\A<color="#(.{6})">(.*?)</color>\Z', re.MULTILINE,
 100             r"{C:$\1}\2"),
 101         # Color (affecting only one line)
 102         (r'<color="#(.{6})">(.*?)</color>', re.MULTILINE,
 103             r"{c:$\1}\2"),
 104         # Color (affecting whole subtitle)
 105         (r'<color="#(.{6})">(.*?)</color>', _FLAGS,
 106             r"{C:$\1}\2"),
 107         # Font (affecting a single line subtitle fully)
 108         (r'\A<font="(.*?)">(.*?)</font>\Z', re.MULTILINE,
 109             r"{F:\1}\2"),
 110         # Font (affecting only one line)
 111         (r'<font="(.*?)">(.*?)</font>', re.MULTILINE,
 112             r"{f:\1}\2"),
 113         # Font (affecting whole subtitle)
 114         (r'<font="(.*?)">(.*?)</font>', _FLAGS,
 115             r"{F:\1}\2"),
 116         # Size (affecting a single line subtitle fully)
 117         (r'\A<size="(.*?)">(.*?)</size>\Z', re.MULTILINE,
 118             r"{S:\1}\2"),
 119         # Size (affecting only one line)
 120         (r'<size="(.*?)">(.*?)</size>', re.MULTILINE,
 121             r"{s:\1}\2"),
 122         # Size (affecting whole subtitle)
 123         (r'<size="(.*?)">(.*?)</size>', _FLAGS,
 124             r"{S:\1}\2")]
 125 
 126     @classmethod
 127     def italicize(cls, text):
 128         """Italicize text."""
 129 
 130         return u"{Y:i}%s" % text

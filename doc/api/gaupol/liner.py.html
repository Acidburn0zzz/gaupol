  19 """Splitter and merger of lines.
  20 
  21 Module variables:
  22 
  23     _RE_MULTI_SPACE: Regular expression for two or more spaces
  24 """
  25 
  26 
  27 from __future__ import division
  28 import copy
  29 import re
  30 
  31 from gaupol import scriptlib
  32 from gaupol.parser import Parser
  33 
  34 
  35 _RE_MULTI_SPACE = re.compile(r" {2,}")
  36 
  37 
  38 class Liner(Parser):
  39 
  40     """Splitter and merger of lines.
  41 
  42     Instance variables:
  43 
  44         _length_func:  A function that returns the length of its argument
  45         _space_length: Length of a space according to length_func
  46         max_length:    Maximum length of a line in units of _length_func
  47         ok_clauses:    Amount of clause lines that need not be joined
  48         ok_dialogue:   Amount of dialogue lines that need not be joined
  49         re_clause:     Regular expression for a clause separator
  50         re_dialogue:   Regular expression for a dialogue separator
  51 
  52     re_clause and re_dialogue are required to have groups 'before' and 'after'
  53     surrounding a single space that acts as the separator.
  54     """
  55 
  56     def __init__(self, re_tag=None):
  57         """Initialize Liner.
  58 
  59         re_tag should be a regular expression object.
  60         """
  61         Parser.__init__(self, re_tag)
  62 
  63         self._length_func  = len
  64         self._space_length = 1
  65         self.max_length    = 44
  66         self.ok_clauses    = 2
  67         self.ok_dialogue   = 3
  68         self.re_clause     = None
  69         self.re_dialogue   = None
  70 
  71         pattern = scriptlib.get_clause_separator("latin-english")
  72         self.re_clause = re.compile(pattern, re.UNICODE)
  73 
  74         pattern = scriptlib.get_dialogue_separator("latin")
  75         self.re_dialogue = re.compile(pattern, re.UNICODE)
  76 
  77     def _get_length(self, lengths):
  78         """Get the length of units joined by spaces."""
  79 
  80         return sum(lengths) + (len(lengths) - 1) * self._space_length
  81 
  82     def _get_split(self, lengths):
  83         """Get index of split in two for units.
  84 
  85         Indexes are brute forced within reason and the result is optimal.
  86         """
  87         min_index = None
  88         min_diff = None
  89         unit_count = len(lengths)
  90         start = self._get_start_index(lengths, 2)
  91         for i in range(start, unit_count):
  92             a = self._get_length(lengths[:i])
  93             b = self._get_length(lengths[i:])
  94             diff = abs(a - b)
  95             if min_diff is None or diff < min_diff:
  96                 min_index = i
  97                 min_diff = diff
  98             if b < a:
  99                 break
 100         return min_index
 101 
 102     def _get_splits(self, lengths, max_lines):
 103         """Get indexes of splits for units."""
 104 
 105         if max_lines == 2:
 106             return [self._get_split(lengths)]
 107         if max_lines > 5 and max_lines % 2 == 0:
 108             return self._get_splits_ugly(lengths, max_lines)
 109         return self._get_splits_brute(lengths, max_lines)
 110 
 111     def _get_splits_brute(self, lengths, max_lines):
 112         """Get brute forced indexes of splits for units.
 113 
 114         Indexes are brute forced within reason and the result is optimal.
 115         """
 116         min_indexes = None
 117         min_squares = None
 118         unit_count = len(lengths)
 119         start = self._get_start_index(lengths, max_lines)
 120         for i in range(start, unit_count - max_lines + 2):
 121             indexes = self._get_splits(lengths[i:], max_lines - 1)
 122             indexes = list(x + i for x in indexes)
 123             indexes.insert(0, i)
 124             borders = [0] + indexes + [unit_count]
 125             line_lengths = []
 126             for j in range(1, len(borders)):
 127                 a, z = borders[j - 1:j + 1]
 128                 line_lengths.append(self._get_length(lengths[a:z]))
 129             mean = sum(line_lengths) / max_lines
 130             squares = sum(list((x - mean)**2 for x in line_lengths))
 131             if min_squares is None or round(min_squares - squares, 6) > 0:
 132                 min_indexes = indexes
 133                 min_squares = squares
 134             mean = self._get_length(lengths[i:]) / (max_lines - 1)
 135             if self._get_length(lengths[:i]) > mean:
 136                 break
 137         return min_indexes
 138 
 139     def _get_splits_ugly(self, lengths, max_lines):
 140         """Get unoptimal indexes of splits for units.
 141 
 142         Units are first split to groups of two, which are then split
 143         internally. The result is not optimal, but is obtained fast.
 144         """
 145         double_lengths = []
 146         unit_count = len(lengths)
 147         for i in range(1, unit_count):
 148             double_lengths.append(sum(lengths[i - 1:i + 1]))
 149         double_indexes = self._get_splits(double_lengths, int(max_lines / 2))
 150         borders = [0] + double_indexes + [unit_count]
 151         sub_indexes = []
 152         for i in range(1, len(borders)):
 153             a, z = borders[i - 1:i + 1]
 154             sub_indexes.append(self._get_split(lengths[a:z]) + a)
 155         return sorted(double_indexes + sub_indexes)
 156 
 157     def _get_start_index(self, lengths, max_lines):
 158         """Get the index for the first split candidate for units.
 159 
 160         The start index is determined based on the line length mean with the
 161         purpose to avoid brute forcing insane indexes.
 162         """
 163         unit_count = len(lengths)
 164         if unit_count > 2:
 165             mean = sum(lengths) / max_lines
 166             for i in range(2, unit_count):
 167                 if self._get_length(lengths[:i]) > mean:
 168                     return i - 1
 169         return 1
 170 
 171     def _join_even(self, max_lines):
 172         """Join the lines, each containing a logical unit, evenly."""
 173 
 174         if max_lines == 1:
 175             self.text = self.text.replace("\n", " ")
 176             return
 177         lengths = list(self._length_func(x) for x in self.text.split("\n"))
 178         indexes = self._get_splits(lengths, max_lines)
 179         text = u""
 180         units = self.text.split("\n")
 181         for i in range(len(lengths)):
 182             prefix = ("\n" if i in indexes else " ")
 183             text = text + prefix + units[i]
 184         self.text = text.strip()
 185 
 186     def _split_on_clauses(self, max_lines):
 187         """Split the text to lines based on clauses.
 188 
 189         Return True if one or more splits made.
 190         """
 191         replacement = r"\g<before>\n\g<after>"
 192         self.text = self.re_clause.sub(replacement, self.text).strip()
 193         if not self.text.count("\n"):
 194             return False
 195 
 196         line_count = max(self.ok_clauses, max_lines)
 197         if self.text.count("\n") < line_count:
 198             return True
 199         self._join_even(line_count)
 200         return True
 201 
 202     def _split_on_dialogue(self, max_lines):
 203         """Split the text to lines based on dialogue.
 204 
 205         Return True if one or more splits made.
 206         """
 207         replacement = r"\g<before>\n\g<after>"
 208         self.text = self.re_dialogue.sub(replacement, self.text).strip()
 209         if not self.text.count("\n"):
 210             return False
 211 
 212         line_count = max(self.ok_dialogue, max_lines)
 213         if self.text.count("\n") < line_count:
 214             return True
 215         self._join_even(line_count)
 216         return True
 217 
 218     def _split_on_words(self, max_lines):
 219         """Split the text to lines based on words.
 220 
 221         Return True.
 222         """
 223         self.text = self.text.replace(" ", "\n")
 224         self.text = self.text.replace("\n-\n", "\n- ")
 225         if self.text.count("\n") < max_lines:
 226             return True
 227         self._join_even(max_lines)
 228         return True
 229 
 230     def format(self):
 231         """Format lines and return text."""
 232 
 233         self.text = self.text.replace("\n", " ")
 234         self.pattern = _RE_MULTI_SPACE
 235         self.replacement = r" "
 236         self.replace_all()
 237 
 238         text = self.text
 239         tags = copy.deepcopy(self.tags)
 240         for max_lines in range(1, 20):
 241             for method in ("dialogue", "clauses", "words"):
 242                 self.text = text
 243                 self.tags = copy.deepcopy(tags)
 244                 method = getattr(self, "_split_on_%s" % method)
 245                 if method(max_lines) and self.is_legal():
 246                     return self.get_text()
 247         return self.get_text()
 248 
 249     def is_legal(self):
 250         """Return True if the text is legal.
 251 
 252         The text is legal if it does not break self.max_length.
 253         """
 254         for line in self.text.split("\n"):
 255             if " " in line and self._length_func(line) > self.max_length:
 256                 return False
 257         return True
 258 
 259     def set_length_function(self, func):
 260         """Set the length function to use."""
 261 
 262         self._length_func = func
 263         self._space_length = func(" ")
 264 
 265     def set_text(self, text, next=True):
 266         """Set the text and parse it.
 267 
 268         next should be True to start at beginning, False for end.
 269         """
 270         Parser.set_text(self, text.strip(), next)

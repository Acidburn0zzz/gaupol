  19 """Time and frame calculator."""
  20 
  21 
  22 from __future__ import division
  23 
  24 
  25 class Calculator(object):
  26 
  27     """Time and frame calculator.
  28 
  29     Instance variables:
  30 
  31         framerate: Video framerate as float
  32 
  33     Times are handled as strings, frames as integers and seconds as floats.
  34     """
  35 
  36     def __init__(self, framerate=None):
  37         """Initialize Calculator.
  38 
  39         framerate should be a FRAMERATE constant.
  40         """
  41         self.framerate = None
  42         if framerate is not None:
  43             self.set_framerate(framerate)
  44 
  45     def add_seconds_to_time(self, time, seconds):
  46         """Add amount of seconds to time."""
  47 
  48         seconds = max(self.time_to_seconds(time) + seconds, 0)
  49         return self.seconds_to_time(seconds)
  50 
  51     def add_times(self, x, y):
  52         """Add time y to time x."""
  53 
  54         x = self.time_to_seconds(x)
  55         y = self.time_to_seconds(y)
  56         return self.seconds_to_time(max(x + y, 0))
  57 
  58     def frame_to_seconds(self, frame):
  59         """Convert frame to seconds."""
  60 
  61         return frame / self.framerate
  62 
  63     def frame_to_time(self, frame):
  64         """Convert frame to time."""
  65 
  66         seconds = self.frame_to_seconds(frame)
  67         return self.seconds_to_time(seconds)
  68 
  69     def get_frame_duration(self, x, y):
  70         """Get duration from frame x to frame y."""
  71 
  72         return max(y - x, 0)
  73 
  74     def get_middle(self, x, y):
  75         """Get time or frame halfway between start and end."""
  76 
  77         if isinstance(x, basestring):
  78             x = self.time_to_seconds(x)
  79             y = self.time_to_seconds(y)
  80             return self.seconds_to_time((x + y) / 2)
  81         if isinstance(x, int):
  82             return int(round((x + y) / 2, 0))
  83 
  84     def get_time_duration(self, x, y):
  85         """Get duration from time x to time y."""
  86 
  87         x = self.time_to_seconds(x)
  88         y = self.time_to_seconds(y)
  89         if x > y:
  90             return "00:00:00.000"
  91         return self.seconds_to_time(y - x)
  92 
  93     def round_time(self, time, decimals):
  94         """Round time to amount of decimals in seconds."""
  95 
  96         seconds = self.time_to_seconds(time)
  97         seconds = round(seconds, decimals)
  98         return self.seconds_to_time(seconds)
  99 
 100     def seconds_to_frame(self, seconds):
 101         """Convert seconds to frame."""
 102 
 103         return int(round(seconds * self.framerate, 0))
 104 
 105     def seconds_to_time(self, seconds):
 106         """Convert seconds to time."""
 107 
 108         seconds = round(seconds, 3)
 109         if seconds > 359999.999:
 110             return "99:59:59.999"
 111 
 112         return "%02.0f:%02.0f:%02.0f.%03.0f" % (
 113             seconds // 3600,
 114             (seconds % 3600) // 60,
 115             int(seconds % 60),
 116             (seconds % 1) * 1000)
 117 
 118     def set_framerate(self, framerate):
 119         """Set the framerate.
 120 
 121         framerate should be a FRAMERATE constant.
 122         """
 123         self.framerate = framerate.value
 124 
 125     def time_to_frame(self, time):
 126         """Convert time to frame."""
 127 
 128         seconds = self.time_to_seconds(time)
 129         return self.seconds_to_frame(seconds)
 130 
 131     def time_to_seconds(self, time):
 132         """Convert time to seconds."""
 133 
 134         return sum((
 135             float(time[ :2]) * 3600,
 136             float(time[3:5]) *   60,
 137             float(time[6:8])       ,
 138             float(time[9: ]) / 1000,))

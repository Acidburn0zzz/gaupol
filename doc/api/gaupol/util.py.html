  39 def memoize(function):
  40     """Decorator for functions that cache their return values."""
  41 
  42     cache = {}
  43     @functools.wraps(function)
  44     def wrapper(*args, **kwargs):
  45         key = cPickle.dumps((args, kwargs))
  46         if not key in cache:
  47             cache[key] = function(*args, **kwargs)
  48         return cache[key]
  49 
  50     return wrapper

  52 def browse_url(url, browser=None):
  53     """Open URL in browser."""
  54 
  55     desktop = get_desktop_environment()
  56     if desktop == "GNOME":
  57         return start_process("gnome-open %s" % url)
  58     if desktop == "KDE":
  59         return start_process("kfmclient exec %s" % url)
  60     if sys.platform == "darwin":
  61         return start_process("open %s" % url)
  62     if sys.platform == "win32":
  63         # pylint: disable-msg=E1101
  64         return os.startfile(url)
  65 
  66     return webbrowser.open(url)

  79 def compare_versions(x, y):
  80     """Compare version strings x and y.
  81 
  82     Used version number formats are MAJOR.MINOR, MAJOR.MINOR.PATCH and
  83     MAJOR.MINOR.PATCH.DATE, where all parts are expected to be integers.
  84     Return 1 if x newer, 0 if equal or -1 if y newer.
  85     """
  86     x = list(int(d) for d in x.split("."))
  87     y = list(int(d) for d in y.split("."))
  88     return cmp(x, y)

  90 def connect(observer, observable, signal, *args):
  91     """Connect observable's signal to observer's callback method.
  92 
  93     If observable is a string, it should be an attribute of observer. If
  94     observable is not a string it should be the same as observer.
  95     """
  96     method_name = signal.replace("-", "_").replace("::", "_")
  97     if observer is not observable:
  98         method_name = observable + "_" + method_name
  99     method_name = ("_on_" + method_name).replace("__", "_")
 100     if not hasattr(observer, method_name):
 101         method_name = method_name[1:]
 102     method = getattr(observer, method_name)
 103     if observer is not observable:
 104         observable = getattr(observer, observable)
 105     return observable.connect(signal, method, *args)

 118 def exceptional(exception, handler, *data):
 119     """Decorator for handling exceptions raised  by function.
 120 
 121     handler will receive sys.exc_info() as the first argument and data as
 122     optional arguments. data are indexes of arguments and keys of keyword
 123     arguments to the function that should be passed to the handler as well.
 124     """
 125     def get_params(data, function, args, kwargs):
 126         params = list(args[x] for x in data if isinstance(x, int))
 127         for key in (x for x in data if isinstance(x, str)):
 128             if  key in kwargs:
 129                 params.append(kwargs[key])
 130                 continue
 131             spec = inspect.getargspec(function)
 132             index = spec[0].index(key) - len(spec[0])
 133             params.append(spec[3][index])
 134         return params
 135 
 136     def outer_wrapper(function):
 137         @functools.wraps(function)
 138         def inner_wrapper(*args, **kwargs):
 139             try:
 140                 return function(*args, **kwargs)
 141             except exception:
 142                 params = get_params(data, function, args, kwargs)
 143                 handler(sys.exc_info(), *params)
 144         return inner_wrapper
 145 
 146     return outer_wrapper

 148 def gc_collected(function):
 149     """Decorator for functions to be followed by a gc.collect call."""
 150 
 151     @functools.wraps(function)
 152     def wrapper(*args, **kwargs):
 153         value = function(*args, **kwargs)
 154         gc.collect()
 155         return value
 156 
 157     return wrapper

 195 def get_ranges(lst):
 196     """Get a list of ranges in list."""
 197 
 198     if not lst:
 199         return []
 200     lst = get_sorted_unique(lst)
 201     ranges = [[lst.pop(0)]]
 202     i = 0
 203     for item in lst:
 204         if item == ranges[i][-1] + 1:
 205             ranges[i].append(item)
 206         else:
 207             ranges.append([item])
 208             i += 1
 209     return ranges
 210 
 211 def get_sorted_unique(lst):
 212     """Get sorted list with duplicates removed."""
 213 
 214     lst = sorted(lst)
 215     for i in reversed(range(1, len(lst))):
 216         if lst[i] == lst[i - 1]:
 217             lst.pop(i)
 218     return lst
 219 
 220 def get_unique(lst):
 221     """Get list with duplicates removed."""
 222 
 223     lst = lst[:]
 224     for i in reversed(range(len(lst))):
 225         for j in range(0, i):
 226             if lst[j] == lst[i]:
 227                 lst.pop(i)
 228                 break
 229     return lst

 255 def ignore_exceptions(*exceptions):
 256     """Decorator for ignoring exceptions raised  by function."""
 257 
 258     if not exceptions:
 259         exceptions = (Exception,)
 260 
 261     def outer_wrapper(function):
 262         @functools.wraps(function)
 263         def inner_wrapper(*args, **kwargs):
 264             try:
 265                 return function(*args, **kwargs)
 266             except exceptions:
 267                 return None
 268         return inner_wrapper
 269 
 270     return outer_wrapper

 312 def read(path, encoding=None):
 313     """Read file and return text.
 314 
 315     Raise IOError if reading fails.
 316     Raise UnicodeError if decoding fails.
 317     """
 318     if encoding is None:
 319         encoding = get_default_encoding()
 320     fobj = codecs.open(path, "r", encoding)
 321     try:
 322         return fobj.read().strip()
 323     finally:
 324         fobj.close()
 325 
 326 def readlines(path, encoding=None):
 327     """Read file and return lines.
 328 
 329     Raise IOError if reading fails.
 330     Raise UnicodeError if decoding fails.
 331     """
 332     text = read(path, encoding)
 333     text = text.replace("\r\n", "\n").replace("\r", "\n")
 334     return text.split("\n")

 336 def shell_quote_path(path):
 337     """Quote and escape path for shell use."""
 338 
 339     if sys.platform != "win32":
 340         path = path.replace("\\", "\\\\")
 341         path = path.replace('"', '\\"')
 342     return '"%s"' % path

 344 def start_process(command, **kwargs):
 345     """Start command as a new background subprocess.
 346 
 347     command and kwargs are passed to subprocess.Popen.__init__.
 348     Return subprocess.Popen instance.
 349     """
 350     return subprocess.Popen(
 351         command,
 352         shell=True,
 353         cwd=os.getcwd(),
 354         env=os.environ.copy(),
 355         universal_newlines=True,
 356         **kwargs)

 363 def write(path, text, encoding=None):
 364     """Write text to file.
 365 
 366     Raise IOError if writing fails.
 367     Raise UnicodeError if encoding fails.
 368     """
 369     if encoding is None:
 370         encoding = get_default_encoding()
 371     fobj = codecs.open(path, "w", encoding)
 372     try:
 373         return fobj.write(text)
 374     finally:
 375         fobj.close()

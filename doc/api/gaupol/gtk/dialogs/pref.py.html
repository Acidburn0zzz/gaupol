  35 class _EditorPage(Delegate):
  36 
  37     """Editor preferences page.
  38 
  39     Instance variables:
  40 
  41         _default_font_check: gtk.CheckButton, use default font
  42         _font_button:        gtk.FontButton, custom font
  43         _font_hbox:          gtk.HBox, custom font widgets
  44         _length_cell_radio:  gtk.RadioButton, show line lengths in cells
  45         _length_combo:       gtk.ComboBox, line length units
  46         _length_edit_radio:  gtk.RadioButton, show line lengths in text views
  47         _undo_limit_hbox:    gtk.HBox, limit undo widgets
  48         _undo_limit_spin:    gtk.SpinButton, amount of undo levels
  49         _unlimit_undo_check: gtk.CheckButton, unlimited undo
  50     """
  51 
  52     def __init__(self, master):
  53 
  54         Delegate.__init__(self, master)
  55 
  56         get_widget = self._glade_xml.get_widget
  57         self._default_font_check = get_widget("editor_default_font_check")
  58         self._font_button        = get_widget("editor_font_button")
  59         self._font_hbox          = get_widget("editor_font_hbox")
  60         self._length_cell_radio  = get_widget("editor_length_cell_radio")
  61         self._length_combo       = get_widget("editor_length_combo")
  62         self._length_edit_radio  = get_widget("editor_length_edit_radio")
  63         self._length_hbox        = get_widget("editor_length_hbox")
  64         self._undo_limit_hbox    = get_widget("editor_undo_limit_hbox")
  65         self._undo_limit_spin    = get_widget("editor_undo_limit_spin")
  66         self._unlimit_undo_check = get_widget("editor_unlimit_undo_check")
  67 
  68         self._init_length_combo()
  69         self._init_data()
  70         self._init_signal_handlers()
  71 
  72     def _get_custom_font(self):
  73         """Get custom font as string."""
  74 
  75         context = gtk.Label().get_pango_context()
  76         font_desc = context.get_font_description()
  77         custom_font_desc = pango.FontDescription(conf.editor.font)
  78         font_desc.merge(custom_font_desc, True)
  79         return font_desc.to_string()
  80 
  81     def _init_data(self):
  82         """Initialize default values for widgets."""
  83 
  84         use_default = conf.editor.use_default_font
  85         self._default_font_check.set_active(use_default)
  86         self._font_hbox.set_sensitive(not use_default)
  87         self._font_button.set_font_name(self._get_custom_font())
  88 
  89         limit = conf.editor.limit_undo
  90         self._unlimit_undo_check.set_active(not limit)
  91         self._undo_limit_hbox.set_sensitive(limit)
  92         self._undo_limit_spin.set_value(conf.editor.undo_levels)
  93 
  94         cell = conf.editor.show_lengths_cell
  95         edit = conf.editor.show_lengths_edit
  96         self._length_hbox.set_sensitive(cell or edit)
  97         self._length_cell_radio.set_active(cell)
  98         self._length_edit_radio.set_active(edit)
  99         self._length_combo.set_active(conf.editor.length_unit)
 100 
 101     def _init_length_combo(self):
 102         """Initialize the line length combo box."""
 103 
 104         store = self._length_combo.get_model()
 105         store.clear()
 106         for name in cons.LENGTH_UNIT.display_names:
 107             store.append([name])
 108 
 109     def _init_signal_handlers(self):
 110         """Initialize signal handlers."""
 111 
 112         util.connect(self, "_default_font_check", "toggled")
 113         util.connect(self, "_font_button", "font-set")
 114         util.connect(self, "_length_cell_radio", "toggled")
 115         util.connect(self, "_length_combo", "changed")
 116         util.connect(self, "_length_edit_radio", "toggled")
 117         util.connect(self, "_undo_limit_spin", "value-changed")
 118         util.connect(self, "_unlimit_undo_check", "toggled")
 119 
 120     def _on_default_font_check_toggled(self, check_button):
 121         """Save the default font usage."""
 122 
 123         use_default = check_button.get_active()
 124         conf.editor.use_default_font = use_default
 125         self._font_hbox.set_sensitive(not use_default)
 126 
 127     def _on_font_button_font_set(self, font_button):
 128         """Save the custom font."""
 129 
 130         conf.editor.font = font_button.get_font_name()
 131 
 132     def _on_length_cell_radio_toggled(self, radio_button):
 133         """Save the line length showage on cells."""
 134 
 135         conf.editor.show_lengths_cell = radio_button.get_active()
 136         cell = conf.editor.show_lengths_cell
 137         edit = conf.editor.show_lengths_edit
 138         self._length_hbox.set_sensitive(cell or edit)
 139 
 140     def _on_length_combo_changed(self, combo_box):
 141         """Save the line length unit."""
 142 
 143         index = combo_box.get_active()
 144         conf.editor.length_unit = cons.LENGTH_UNIT.members[index]
 145 
 146     def _on_length_edit_radio_toggled(self, radio_button):
 147         """Save the line length showage on text views."""
 148 
 149         conf.editor.show_lengths_edit = radio_button.get_active()
 150         cell = conf.editor.show_lengths_cell
 151         edit = conf.editor.show_lengths_edit
 152         self._length_hbox.set_sensitive(cell or edit)
 153 
 154     def _on_undo_limit_spin_value_changed(self, spin_button):
 155         """Save the amount of undo levels."""
 156 
 157         conf.editor.undo_levels = spin_button.get_value_as_int()
 158 
 159     def _on_unlimit_undo_check_toggled(self, check_button):
 160         """Save the undo limiting."""
 161 
 162         limit = not check_button.get_active()
 163         conf.editor.limit_undo = limit
 164         self._undo_limit_hbox.set_sensitive(limit)

 167 class _FilePage(Delegate):
 168 
 169     """File preferences page.
 170 
 171     Instance variables:
 172 
 173         _add_button:    gtk.Button, add encoding
 174         _auto_check:    gtk.CheckButton, try encoding auto-detection
 175         _down_button:   gtk.Button, move encoding down
 176         _tree_view:     gtk.TreeView, fallback encodings
 177         _locale_check:  gtk.CheckButton, try locale encoding
 178         _remove_button: gtk.Button, remove encoding
 179         _up_button:     gtk.Button, move encoding up
 180     """
 181 
 182     def __init__(self, master):
 183 
 184         Delegate.__init__(self, master)
 185 
 186         get_widget = self._glade_xml.get_widget
 187         self._add_button    = get_widget("file_add_button")
 188         self._auto_check    = get_widget("file_auto_check")
 189         self._down_button   = get_widget("file_down_button")
 190         self._tree_view     = get_widget("file_tree_view")
 191         self._locale_check  = get_widget("file_locale_check")
 192         self._remove_button = get_widget("file_remove_button")
 193         self._up_button     = get_widget("file_up_button")
 194 
 195         self._init_tree_view()
 196         self._init_data()
 197         self._init_signal_handlers()
 198 
 199     def _get_selected_row(self):
 200         """Get the selected row in the tree view or None."""
 201 
 202         selection = self._tree_view.get_selection()
 203         store, itr = selection.get_selected()
 204         if itr is not None:
 205             return store.get_path(itr)[0]
 206         return None
 207 
 208     def _init_data(self):
 209         """Initialize default values for widgets."""
 210 
 211         self._auto_check.set_active(conf.encoding.try_auto)
 212         self._auto_check.set_sensitive(util.chardet_available())
 213         self._locale_check.set_active(conf.encoding.try_locale)
 214         self._reload_tree_view()
 215 
 216     def _init_signal_handlers(self):
 217         """Initialize signal handlers."""
 218 
 219         util.connect(self, "_add_button", "clicked")
 220         util.connect(self, "_auto_check", "toggled")
 221         util.connect(self, "_down_button", "clicked")
 222         util.connect(self, "_locale_check", "toggled")
 223         util.connect(self, "_remove_button", "clicked")
 224         util.connect(self, "_up_button", "clicked")
 225 
 226         def update(*args):
 227             self._set_sensitivities()
 228         selection = self._tree_view.get_selection()
 229         selection.connect("changed", update)
 230 
 231     def _init_tree_view(self):
 232         """Initialize the tree view."""
 233 
 234         self._tree_view.columns_autosize()
 235         selection = self._tree_view.get_selection()
 236         selection.set_mode(gtk.SELECTION_SINGLE)
 237         store = gtk.ListStore(gobject.TYPE_STRING)
 238         self._tree_view.set_model(store)
 239         column = gtk.TreeViewColumn("", gtk.CellRendererText(), text=0)
 240         self._tree_view.append_column(column)
 241 
 242     def _on_add_button_clicked(self, *args):
 243         """Add a new fallback encoding."""
 244 
 245         dialog = EncodingDialog(self._dialog)
 246         response = self.run_dialog(dialog)
 247         encoding = dialog.get_encoding()
 248         dialog.destroy()
 249         if response != gtk.RESPONSE_OK:
 250             return
 251         if encoding is None:
 252             return
 253         if encoding not in conf.encoding.fallbacks:
 254             conf.encoding.fallbacks.append(encoding)
 255         self._reload_tree_view()
 256         self._tree_view.grab_focus()
 257         store = self._tree_view.get_model()
 258         self._tree_view.set_cursor(len(store) - 1)
 259 
 260     def _on_auto_check_toggled(self, check_button):
 261         """Save the encoding auto-detection usage."""
 262 
 263         conf.encoding.try_auto = check_button.get_active()
 264 
 265     def _on_down_button_clicked(self, *args):
 266         """Move the selected fallback encoding down."""
 267 
 268         row = self._get_selected_row()
 269         encodings = conf.encoding.fallbacks
 270         encodings.insert(row + 1, encodings.pop(row))
 271         self._reload_tree_view()
 272         self._tree_view.grab_focus()
 273         self._tree_view.set_cursor(row + 1)
 274 
 275     def _on_locale_check_toggled(self, check_button):
 276         """Save the locale encoding usage."""
 277 
 278         conf.encoding.try_locale = check_button.get_active()
 279 
 280     def _on_remove_button_clicked(self, *args):
 281         """Remove the selected encoding."""
 282 
 283         row = self._get_selected_row()
 284         conf.encoding.fallbacks.pop(row)
 285         self._reload_tree_view()
 286         self._tree_view.grab_focus()
 287         store = self._tree_view.get_model()
 288         if len(store) > 0:
 289             self._tree_view.set_cursor(max(row - 1, 0))
 290 
 291     def _on_up_button_clicked(self, *args):
 292         """Move the selected encoding up."""
 293 
 294         row = self._get_selected_row()
 295         encodings = conf.encoding.fallbacks
 296         encodings.insert(row - 1, encodings.pop(row))
 297         self._reload_tree_view()
 298         self._tree_view.grab_focus()
 299         self._tree_view.set_cursor(row - 1)
 300 
 301     def _reload_tree_view(self):
 302         """Reload the tree view."""
 303 
 304         store = self._tree_view.get_model()
 305         store.clear()
 306         for encoding in conf.encoding.fallbacks:
 307             store.append([enclib.get_long_name(encoding)])
 308         self._set_sensitivities()
 309 
 310     def _set_sensitivities(self):
 311         """Set the tree view button sensitivities."""
 312 
 313         def set_sensitive(remove, up, down):
 314             self._remove_button.set_sensitive(remove)
 315             self._up_button.set_sensitive(up)
 316             self._down_button.set_sensitive(down)
 317         store = self._tree_view.get_model()
 318         row = self._get_selected_row()
 319         if row is not None:
 320             return set_sensitive(True, row > 0, row < len(store) - 1)
 321         return set_sensitive(False, False, False)

 324 class _PreviewPage(Delegate):
 325 
 326     """Preview preferences page.
 327 
 328     Instance variables:
 329 
 330         _app_combo:     gtk.ComboBox, video players
 331         _command_entry: gtk.Entry, video player command
 332         _offset_spin:   gtk.SpinButton, offset to start playing at
 333     """
 334 
 335     def __init__(self, master):
 336 
 337         Delegate.__init__(self, master)
 338 
 339         get_widget = self._glade_xml.get_widget
 340         self._app_combo     = get_widget("preview_app_combo")
 341         self._command_entry = get_widget("preview_command_entry")
 342         self._offset_spin   = get_widget("preview_offset_spin")
 343 
 344         self._init_app_combo()
 345         self._init_data()
 346         self._init_signal_handlers()
 347 
 348     def _init_app_combo(self):
 349         """Initialize the application combo box."""
 350 
 351         store = self._app_combo.get_model()
 352         store.clear()
 353         for name in cons.VIDEO_PLAYER.display_names:
 354             store.append([name])
 355         store.append([util.COMBO_SEP])
 356         store.append([_("Custom")])
 357         self._app_combo.set_row_separator_func(util.separate_combo)
 358 
 359     def _init_data(self):
 360         """Initialize default values for widgets."""
 361 
 362         if conf.preview.use_predefined:
 363             self._app_combo.set_active(conf.preview.video_player)
 364             self._command_entry.set_text(conf.preview.video_player.command)
 365             self._command_entry.set_editable(False)
 366         else:
 367             store = self._app_combo.get_model()
 368             self._app_combo.set_active(len(store) - 1)
 369             self._command_entry.set_text(conf.preview.custom_command)
 370             self._command_entry.set_editable(True)
 371 
 372         self._offset_spin.set_value(conf.preview.offset)
 373 
 374     def _init_signal_handlers(self):
 375         """Initialize signal handlers."""
 376 
 377         util.connect(self, "_app_combo", "changed")
 378         util.connect(self, "_command_entry", "changed")
 379         util.connect(self, "_offset_spin", "value-changed")
 380 
 381     def _on_app_combo_changed(self, combo_box):
 382         """Save the video player and show it's command."""
 383 
 384         index = combo_box.get_active()
 385         if index in range(len(cons.VIDEO_PLAYER.members)):
 386             conf.preview.use_predefined = True
 387             conf.preview.video_player = cons.VIDEO_PLAYER.members[index]
 388             self._command_entry.set_text(conf.preview.video_player.command)
 389             self._command_entry.set_editable(False)
 390         else:
 391             conf.preview.use_predefined = False
 392             self._command_entry.set_text(conf.preview.custom_command)
 393             self._command_entry.set_editable(True)
 394 
 395     def _on_command_entry_changed(self, entry):
 396         """Save the custom command."""
 397 
 398         if not conf.preview.use_predefined:
 399             conf.preview.custom_command = entry.get_text()
 400 
 401     def _on_offset_spin_value_changed(self, spin_button):
 402         """Save the offset."""
 403 
 404         conf.preview.offset = spin_button.get_value()

 407 class PreferencesDialog(GladeDialog):
 408 
 409     """Dialog for editing preferences.
 410 
 411     Instance variables:
 412 
 413         _editor_page:  _EditorPage
 414         _file_page:    _FilePage
 415         _preview_page: _PreviewPage
 416     """
 417 
 418     def __init__(self):
 419 
 420         GladeDialog.__init__(self, "pref-dialog")
 421 
 422         self._editor_page  = _EditorPage(self)
 423         self._file_page    = _FilePage(self)
 424         self._preview_page = _PreviewPage(self)
 425 
 426         self.set_transient_for(None)
 427         self.set_type_hint(gtk.gdk.WINDOW_TYPE_HINT_NORMAL)
 428         self.set_default_response(gtk.RESPONSE_CLOSE)

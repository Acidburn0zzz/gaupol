 696 def get_country(locale):
 697     """Get the localized country name from locale code or None.
 698 
 699     Raise KeyError if language not found.
 700     """
 701     if len(locale) == 5:
 702         return dgettext("iso_3166", COUNTRIES[locale[3:]])
 703     return None
 704 
 705 def get_language(locale):
 706     """Get the localized language name from locale code.
 707 
 708     Raise KeyError if language not found.
 709     """
 710     return dgettext("iso_639", LANGS[locale[:2]])
 711 
 712 def get_long_name(locale):
 713     """Get the localized long name from locale code.
 714 
 715     Raise KeyError if language not found.
 716     Return 'Language (Country)'.
 717     """
 718     lang = get_language(locale)
 719     country = get_country(locale)
 720     if country is not None:
 721         fields = {'language': lang, 'country': country}
 722         return _("%(language)s (%(country)s)") % fields
 723     return lang

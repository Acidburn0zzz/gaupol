  19 """List widget to display subtitle data.
  20 
  21 Module variables:
  22 
  23     _ACTIVE_ATTR: pango.AttrList of the active column header label
  24     _NORMAL_ATTR: pango.AttrList of a normal column header label
  25 """
  26 
  27 
  28 import gobject
  29 import gtk
  30 import pango
  31 
  32 from gaupol import util
  33 from gaupol.gtk import conf, cons
  34 from gaupol.gtk.cellrend import *
  35 from gaupol.gtk.index import *
  36 
  37 
  38 _ACTIVE_ATTR = pango.AttrList()
  39 _ACTIVE_ATTR.insert(pango.AttrWeight(pango.WEIGHT_BOLD, 0, -1))
  40 _NORMAL_ATTR = pango.AttrList()
  41 _NORMAL_ATTR.insert(pango.AttrWeight(pango.WEIGHT_NORMAL, 0, -1))
  42 
  43 
  44 class View(gtk.TreeView):
  45 
  46     """List widget to display subtitle data.
  47 
  48     Instance variables:
  49 
  50         _active_col: Index of the active column
  51     """
  52 
  53     def __init__(self, edit_mode):
  54 
  55         gtk.TreeView.__init__(self)
  56 
  57         self._active_col = None
  58 
  59         self._init_props(edit_mode)
  60         self._init_signal_handlers()
  61 
  62     def _get_renderer(self, col, edit_mode):
  63         """Initialize and return a new cell renderer."""
  64 
  65         font = ("" if conf.editor.use_default_font else conf.editor.font)
  66         if col == NO:
  67             renderer = gtk.CellRendererText()
  68             renderer.props.editable = False
  69         elif col in (SHOW, HIDE, DURN):
  70             if edit_mode == cons.MODE.TIME:
  71                 renderer = TimeCellRenderer()
  72             elif edit_mode == cons.MODE.FRAME:
  73                 renderer = gtk.CellRendererSpin()
  74                 adjustment = gtk.Adjustment(0, 0, 99999999, 1, 10)
  75                 renderer.props.adjustment = adjustment
  76             renderer.props.editable = True
  77         elif col in (MTXT, TTXT):
  78             renderer = MultilineCellRenderer()
  79             renderer.props.editable = True
  80         renderer.props.font = font
  81         return renderer
  82 
  83     def _init_columns(self, edit_mode):
  84         """Initialize the tree view columns."""
  85 
  86         for i in range(6):
  87 
  88             name = cons.COLUMN.display_names[i]
  89             renderer = self._get_renderer(i, edit_mode)
  90             column = gtk.TreeViewColumn(name, renderer , text=i)
  91             self.append_column(column)
  92             column.set_clickable(True)
  93             column.set_resizable(True)
  94             column.set_visible(i in conf.editor.visible_cols)
  95             column.set_expand(i in (4, 5))
  96 
  97             # Set the header label wide enough.
  98             label = gtk.Label(column.get_title())
  99             column.set_widget(label)
 100             label.show()
 101             label.set_attributes(_ACTIVE_ATTR)
 102             width = label.size_request()[0]
 103             label.set_size_request(width, -1)
 104             label.set_attributes(_NORMAL_ATTR)
 105 
 106         # Set the data in the number column automatically.
 107         def set_number(column, renderer, store, itr):
 108             renderer.props.text = store.get_path(itr)[0] + 1
 109         column = self.get_column(NO)
 110         renderer = column.get_cell_renderers()[0]
 111         column.set_cell_data_func(renderer, set_number)
 112 
 113     def _init_props(self, edit_mode):
 114         """Initialize properties."""
 115 
 116         self.set_headers_visible(True)
 117         self.set_rules_hint(True)
 118         selection = self.get_selection()
 119         selection.set_mode(gtk.SELECTION_MULTIPLE)
 120         self.set_enable_search(True)
 121         self.set_search_column(NO)
 122 
 123         columns = [gobject.TYPE_INT]
 124         if edit_mode == cons.MODE.TIME:
 125             columns += [gobject.TYPE_STRING] * 3
 126         elif edit_mode == cons.MODE.FRAME:
 127             columns += [gobject.TYPE_INT] * 3
 128         columns += [gobject.TYPE_STRING] * 2
 129         store = gtk.ListStore(*columns)
 130         self.set_model(store)
 131         self._init_columns(edit_mode)
 132 
 133     def _init_signal_handlers(self):
 134         """Initialize signal handlers."""
 135 
 136         util.connect(self, self, "cursor-changed")
 137         util.connect(self, self, "key-press-event")
 138 
 139         conf.connect(self, "editor", "font")
 140         conf.connect(self, "editor", "length_unit")
 141         conf.connect(self, "editor", "show_lengths_cell")
 142         conf.connect(self, "editor", "use_default_font")
 143 
 144     def _on_conf_editor_notify_font(self, *args):
 145         """Apply the new font."""
 146 
 147         if not conf.editor.use_default_font:
 148             for column in self.get_columns():
 149                 renderer = column.get_cell_renderers()[0]
 150                 renderer.props.font = conf.editor.font
 151             self.columns_autosize()
 152 
 153     def _on_conf_editor_notify_length_unit(self, *args):
 154         """Repaint the cells."""
 155 
 156         if conf.editor.show_lengths_cell:
 157             self.columns_autosize()
 158 
 159     def _on_conf_editor_notify_show_lengths_cell(self, *args):
 160         """Repaint the cells."""
 161 
 162         self.columns_autosize()
 163 
 164     def _on_conf_editor_notify_use_default_font(self, *args):
 165         """Apply the new font."""
 166 
 167         font = ("" if conf.editor.use_default_font else conf.editor.font)
 168         for column in self.get_columns():
 169             renderer = column.get_cell_renderers()[0]
 170             renderer.props.font = font
 171         self.columns_autosize()
 172 
 173     def _on_cursor_changed(self, *args):
 174         """Update the column header labels."""
 175 
 176         self.update_headers()
 177 
 178     def _on_key_press_event(self, widget, event):
 179         """Enable or disable the interactive search."""
 180 
 181         self.set_enable_search(event.string.isdigit())
 182 
 183     def get_focus(self):
 184         """Get the row and column of the current focus."""
 185 
 186         row, col = self.get_cursor()
 187         if row is not None:
 188             row = row[0]
 189         if col is not None:
 190             col = self.get_columns().index(col)
 191         return row, col
 192 
 193     def get_selected_rows(self):
 194         """Get a list of the selected rows."""
 195 
 196         rows = self.get_selection().get_selected_rows()[1]
 197         return list(x[0] for x in rows)
 198 
 199     def scroll_to_row(self, row):
 200         """Scroll to row."""
 201 
 202         self.scroll_to_cell(row, None, True, 0.5, 0)
 203 
 204     def select_rows(self, rows):
 205         """Select rows, clearing previous selection."""
 206 
 207         # Select by ranges to avoid sending too many 'changed' signals.
 208         selection = self.get_selection()
 209         selection.unselect_all()
 210         for lst in util.get_ranges(rows):
 211             selection.select_range(lst[0], lst[-1])
 212 
 213     def set_focus(self, row, col=None):
 214         """Set the focus to row, col.
 215 
 216         row can be -1 for the last row.
 217         """
 218         if row == -1:
 219             row = len(self.get_model()) - 1
 220         if col is not None:
 221             col = self.get_column(col)
 222         self.set_cursor(row, col)
 223 
 224     def update_headers(self):
 225         """Update the attributes of the column header labels."""
 226 
 227         col = self.get_focus()[1]
 228         if col == self._active_col:
 229             return
 230         if self._active_col is not None:
 231             label = self.get_column(self._active_col).get_widget()
 232             label.set_attributes(_NORMAL_ATTR)
 233         if col is not None:
 234             label = self.get_column(col).get_widget()
 235             label.set_attributes(_ACTIVE_ATTR)
 236         self._active_col = col

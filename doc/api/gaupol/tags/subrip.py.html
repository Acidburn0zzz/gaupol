  19 """SubRip tag library."""
  20 
  21 
  22 import re
  23 
  24 from ._taglib import TagLibrary
  25 
  26 
  27 class SubRip(TagLibrary):
  28 
  29     """SubRip tag library.
  30 
  31     Class variables:
  32 
  33         _re_redundant: Regular expression for a redundant tag pair
  34     """
  35 
  36     tag = r"</?(b|i|u)>", re.IGNORECASE
  37     italic_tag = r"</?i>", re.IGNORECASE
  38 
  39     decode_tags = [
  40         # Uppercase bold
  41         (r"(</?)B>", 0,
  42             r"\1b>"),
  43         # Uppercase italic
  44         (r"(</?)I>", 0,
  45             r"\1i>"),
  46         # Uppercase underline
  47         (r"(</?)U>", 0,
  48             r"\1u>")]
  49 
  50     encode_tags = [
  51         # All unsupported tags
  52         (r"</?[^>]{3,}>", 0,
  53             r"")]
  54 
  55     _re_redundant = re.compile(r"</([ibu])>([^\w\n]*?)<\1>", re.UNICODE)
  56 
  57     @classmethod
  58     def italicize(cls, text):
  59         """Italicize text."""
  60 
  61         return u"<i>%s</i>" % text
  62 
  63     @classmethod
  64     def remove_redundant(cls, text):
  65         """Remove redundant tags from text."""
  66 
  67         return cls._re_redundant.sub(r"\2", text)

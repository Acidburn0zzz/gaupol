  27 def notify_frozen(function):
  28     """Decorator for methods to be run in notify frozen state."""
  29 
  30     @functools.wraps(function)
  31     def wrapper(*args, **kwargs):
  32         changed = args[0].freeze_notify()
  33         value = function(*args, **kwargs)
  34         args[0].thaw_notify(changed)
  35         return value
  36 
  37     return wrapper

  40 class Observable(object):
  41 
  42     """Base class for observable objects.
  43 
  44     Class variables:
  45 
  46         _signals: List of emittable signals added automatically
  47 
  48     Instance variables:
  49 
  50         _blocked_signals: List of blocked signals
  51         _notify_frozen:   If True notify signals will be queued, not emitted
  52         _notify_queue:    List of queued notify signals
  53         _signal_handlers: Dictionary mapping signals to observers and data
  54 
  55     In addition to the signals defined in '_signals', all public instance
  56     variables will have a 'notify::name' signal generated automatically based
  57     on the name of the variable. 'notify::name' signals will be emitted
  58     whenever the value of the instance variable changes.
  59 
  60     Notify signals will be emitted for mutable variables as well, which means
  61     care should be taken not to emit thousands of signals when appending
  62     one-by-one to a large list. 'freeze_notify' and 'thaw_notify' methods as
  63     well as the 'notify_frozen' decorator will queue notify signals and emit
  64     only one of each once thawed.
  65     """
  66 
  67     # The Observable philosophy and API is highly inspired by (Py)GObject.
  68     # http://www.pygtk.org/docs/pygobject/class-gobject.html
  69 
  70     __slots__ = [
  71         "_blocked_signals",
  72         "_notify_frozen",
  73         "_notify_queue",
  74         "_signal_handlers",]
  75 
  76     _signals = []
  77 
  78     def __init__(self):
  79 
  80         self._blocked_signals = []
  81         self._notify_frozen   = False
  82         self._notify_queue    = []
  83         self._signal_handlers = {}
  84 
  85         for signal in self._signals:
  86             self._add_signal(signal)
  87 
  88     def __setattr__(self, name, value):
  89 
  90         if name in self.__slots__ or name.startswith("_"):
  91             return object.__setattr__(self, name, value)
  92         value = self._validate(name, value)
  93         signal = "notify::%s" % name
  94         if not signal in self._signal_handlers:
  95             self._add_signal(signal)
  96             return object.__setattr__(self, name, value)
  97         return_value = object.__setattr__(self, name, value)
  98         self.emit(signal, value)
  99         return return_value
 100 
 101     def _add_signal(self, signal):
 102         """Add signal to the list of signals."""
 103 
 104         if signal in self._signal_handlers:
 105             raise ValueError
 106         self._signal_handlers[signal] = []
 107 
 108     def _validate(self, name, value):
 109         """Return value or an observable version of mutable value."""
 110 
 111         if isinstance(value, dict):
 112             return ObservableDict(value, self, name)
 113         if isinstance(value, list):
 114             return ObservableList(value, self, name)
 115         if isinstance(value, set):
 116             return ObservableSet(value, self, name)
 117         return value
 118 
 119     def block(self, signal):
 120         """Block all emissions of signal.
 121 
 122         Return False if already blocked, otherwise True.
 123         """
 124         if not signal in self._blocked_signals:
 125             self._blocked_signals.append(signal)
 126             return True
 127         return False
 128 
 129     def connect(self, signal, method, *args):
 130         """Register to receive notifications of signal."""
 131 
 132         self._signal_handlers[signal].append((method, args))
 133 
 134     def disconnect(self, signal, method):
 135         """Remove registration to receive notifications of signal."""
 136 
 137         for i in reversed(range(len(self._signal_handlers[signal]))):
 138             if self._signal_handlers[signal][i][0] == method:
 139                 self._signal_handlers[signal].pop(i)
 140 
 141     def emit(self, signal, *args):
 142         """Send notification of signal to all registered observers."""
 143 
 144         if signal.startswith("notify::") and self._notify_frozen:
 145             if signal not in self._notify_queue:
 146                 self._notify_queue.append(signal)
 147             return
 148 
 149         if not signal in self._blocked_signals:
 150             # FIX: REMOVE WHEN NO EXCESSIVE SIGNALS EMITTED.
 151             print "%s.%s" % (self.__class__.__name__, signal)
 152             if signal.startswith("notify::"):
 153                 name = signal.replace("notify::", "")
 154                 args = (getattr(self, name),)
 155             for method, data in self._signal_handlers[signal]:
 156                 method(*((self,) + args + data))
 157 
 158     def freeze_notify(self):
 159         """Queue notify signals instead of emitting them.
 160 
 161         Return False if already frozen, otherwise True.
 162         """
 163         if not self._notify_frozen:
 164             self._notify_frozen = True
 165             return True
 166         return False
 167 
 168     def notify(self, name):
 169         """Emit notification signal for variable."""
 170 
 171         return self.emit("notify::%s" % name)
 172 
 173     def thaw_notify(self, do=True):
 174         """Emit all queued notify signals and queue no more.
 175 
 176         The optional 'do' keyword argument should be the return value from
 177         'freeze_notify' to avoid problems with nested functions where
 178         notifications were frozen at a higher level. If 'do' is False, nothing
 179         will be done.
 180         Return False if already thawed, otherwise True.
 181         """
 182         if do and self._notify_frozen:
 183             self._notify_frozen = False
 184             for signal in self._notify_queue:
 185                 name = signal.replace("notify::", "")
 186                 self.emit(signal, getattr(self, name))
 187             self._notify_queue = []
 188             return True
 189         return False
 190 
 191     def unblock(self, signal, do=True):
 192         """Unblock all emissions of signal.
 193 
 194         The optional 'do' keyword argument should be the return value from
 195         'block' to avoid problems with nested functions where notifications
 196         were frozen at a higher level. If 'do' is False, nothing will be done.
 197         Return False if already unblocked, otherwise True.
 198         """
 199         if do and (signal in self._blocked_signals):
 200             self._blocked_signals.remove(signal)
 201             return True
 202         return False

  19 """TMPlayer file."""
  20 
  21 
  22 import codecs
  23 
  24 from gaupol import cons
  25 from gaupol.calculator import Calculator
  26 from ._subfile import SubtitleFile
  27 
  28 
  29 class TMPlayer(SubtitleFile):
  30 
  31     """TMPlayer file."""
  32 
  33     format = cons.FORMAT.TMPLAYER
  34     has_header = False
  35     identifier = r"^\d\d:\d\d:\d\d:.*$", 0
  36     mode = cons.MODE.TIME
  37 
  38     def read(self):
  39         """Read file.
  40 
  41         Raise IOError if reading fails.
  42         Raise UnicodeError if decoding fails.
  43         Return show times, hide times, texts.
  44         """
  45         shows = []
  46         hides = []
  47         texts = []
  48         for line in self._read_lines():
  49             if len(line.strip()) >= 9:
  50                 shows.append(line[:8] + ".000")
  51                 texts.append(line[9:-1])
  52 
  53         calc = Calculator()
  54         for i in range(1, len(shows)):
  55             hides.append(shows[i])
  56         hides.append(calc.add_seconds_to_time(shows[-1], 3.000))
  57         texts = list(x.replace("|", "\n") for x in texts)
  58         return shows, hides, texts
  59 
  60     def write(self, shows, hides, texts):
  61         """Write file.
  62 
  63         Raise IOError if writing fails.
  64         Raise UnicodeError if encoding fails.
  65         """
  66         calc = Calculator()
  67         shows = list(calc.round_time(x, 0)[:8] + ":" for x in shows)
  68         texts = list(x.replace("\n", "|") for x in texts)
  69 
  70         fobj = codecs.open(self.path, "w", self.encoding)
  71         try:
  72             for i in range(len(shows)):
  73                 fobj.write("%s%s%s" % (shows[i], texts[i], self.newline.value))
  74         finally:
  75             fobj.close()

  46     def _append_subtitles(self, current, temp):
  47         """Append subtitles in temporary page to current.
  48 
  49         Return a list of rows appended.
  50         """
  51         rows = range(len(current.project.times),
  52             len(current.project.times) + len(temp.project.times))
  53         current.project.block("action-done")
  54         current.project.insert_subtitles(rows,
  55             temp.project.times,
  56             temp.project.frames,
  57             temp.project.main_texts,
  58             temp.project.tran_texts)
  59         current.project.set_action_description(
  60             cons.REGISTER.DO, _("Appending file"))
  61         current.project.unblock("action-done")
  62         return rows

  64     def _get_encodings(self, first=None):
  65         """Get a list of encodings to try."""
  66 
  67         encodings = [first]
  68         if conf.encoding.try_locale:
  69             encodings.append(enclib.get_locale_python_name())
  70         encodings += conf.encoding.fallbacks
  71         if conf.encoding.try_auto and util.chardet_available():
  72             encodings.append("auto")
  73         while None in encodings:
  74             encodings.remove(None)
  75         if not encodings:
  76             encodings.append("utf_8")
  77         return util.get_unique(encodings)

  79     def _get_file_open(self, path):
  80         """Switch page and return True if file is already open."""
  81 
  82         paths = []
  83         indexes = []
  84         for i, page in enumerate(self.pages):
  85             files = (page.project.main_file, page.project.tran_file)
  86             for file in (x for x in files if x is not None):
  87                 paths.append(file.path)
  88                 indexes.append(i)
  89         if path in paths:
  90             index = indexes[paths.index(path)]
  91             self.notebook.set_current_page(index)
  92             basename = os.path.basename(path)
  93             self.push_message(_('File "%s" is already open') % basename)
  94             return True
  95         return False

  97     def _open_file(self, path, encodings, doc, check_open=True):
  98         """Open file and return page.
  99 
 100         Raise Default if something goes wrong.
 101         """
 102         self._pre_open_check(path, check_open)
 103         basename = os.path.basename(path)
 104 
 105         if doc == cons.DOCUMENT.MAIN:
 106             page = Page()
 107             open_method = page.project.open_main
 108         elif doc == cons.DOCUMENT.TRAN:
 109             page = self.get_current_page()
 110             open_method = page.project.open_translation
 111             open_method = functools.partial(
 112                 open_method, smart=conf.file.smart_tran)
 113 
 114         for encoding in encodings:
 115             sort_count = self._try_open_file(open_method, path, encoding)
 116             if sort_count is not None:
 117                 file = page.project.get_file(doc)
 118                 self._post_open_check(file, sort_count)
 119                 return page
 120         self._show_encoding_error_dialog(basename)
 121         raise Default
 122 
 123     def _post_open_check(self, file, sort_count):
 124         """Check file after opening.
 125 
 126         Raise Default if something goes wrong.
 127         """
 128         if file.format in (cons.FORMAT.SSA, cons.FORMAT.ASS):
 129             if conf.file.warn_ssa:
 130                 self._show_ssa_warning_dialog()
 131         if sort_count > 0:
 132             basename = os.path.basename(file.path)
 133             self._show_sort_warning_dialog(basename, sort_count)
 134 
 135     def _pre_append_shift(self, current, temp):
 136         """Shift subtitles in current page before appending ones in temp."""
 137 
 138         mode = temp.project.get_mode()
 139         if mode == cons.MODE.TIME:
 140             count = current.project.times[-1][1]
 141             count = current.project.calc.time_to_seconds(count)
 142             method = temp.project.shift_seconds
 143         elif mode == cons.MODE.FRAME:
 144             count = current.project.frames[-1][1]
 145             method = temp.project.shift_frames
 146         method([], count, register=None)
 147 
 148     def _pre_open_check(self, path, check_open):
 149         """Check file before opening.
 150 
 151         Raise Default if something goes wrong.
 152         """
 153         if not os.path.isfile(path):
 154             raise Default
 155         if check_open and self._get_file_open(path):
 156             raise Default
 157         basename = os.path.basename(path)
 158         size = os.stat(path)[6] / 1048576
 159         if size > 1:
 160             self._show_size_warning_dialog(basename, size)

 162     @util.gc_collected
 163     def _select_files(self, title, doc):
 164         """Select and return files and encoding."""
 165 
 166         paths = []
 167         encoding = None
 168         util.set_cursor_busy(self.window)
 169         dialog = OpenDialog(doc, title, self.window)
 170         util.set_cursor_normal(self.window)
 171         response = self.run_dialog(dialog)
 172         if response == gtk.RESPONSE_OK:
 173             paths = dialog.get_filenames()
 174             encoding = dialog.get_encoding()
 175         dialog.destroy()
 176         while gtk.events_pending():
 177             gtk.main_iteration()
 178         return paths, encoding

 180     def _show_encoding_error_dialog(self, basename):
 181         """Show an error dialog after failing to decode file."""
 182 
 183         title = _('Failed to decode file "%s" with all attempted codecs') \
            % basename
 184         message = _("Please try to open the file with a different character "
 185             "encoding.")
 186         dialog = ErrorDialog(self.window, title, message)
 187         dialog.add_button(gtk.STOCK_OK, gtk.RESPONSE_OK)
 188         self.flash_dialog(dialog)
 189 
 190     def _show_format_error_dialog(self, basename):
 191         """Show an error dialog after failing to recognize file format."""
 192 
 193         title = _('Failed to recognize format of file "%s"') % basename
 194         message = _("Please check that the file you are trying to open is a "
 195             "subtitle file of a format supported by Gaupol.")
 196         dialog = ErrorDialog(self.window, title, message)
 197         dialog.add_button(gtk.STOCK_OK, gtk.RESPONSE_OK)
 198         self.flash_dialog(dialog)

 200     def _show_io_error_dialog(self, basename, message):
 201         """Show an error dialog after failing to read file."""
 202 
 203         title = _('Failed to open file "%s"') % basename
 204         message = _("%s.") % message
 205         dialog = ErrorDialog(self.window, title, message)
 206         dialog.add_button(gtk.STOCK_OK, gtk.RESPONSE_OK)
 207         self.flash_dialog(dialog)

 209     def _show_size_warning_dialog(self, basename, size):
 210         """Show a warning dialog when trying to open a big file.
 211 
 212         Raise Default to abort.
 213         """
 214         title = _('Open abnormally large file "%s"?') % basename
 215         message = _("Size of the file is %.1f MB, which is abnormally large "
 216             "for a text-based subtitle file. Please, check that you are not "
 217             "trying to open a binary file.") % size
 218         dialog = WarningDialog(self.window, title, message)
 219         dialog.add_button(gtk.STOCK_CANCEL, gtk.RESPONSE_NO)
 220         dialog.add_button(gtk.STOCK_OPEN, gtk.RESPONSE_YES)
 221         dialog.set_default_response(gtk.RESPONSE_NO)
 222         response = self.flash_dialog(dialog)
 223         if response != gtk.RESPONSE_YES:
 224             raise Default

 235         dialog.add_button(gtk.STOCK_CANCEL, gtk.RESPONSE_NO)
 236         dialog.add_button(gtk.STOCK_OPEN, gtk.RESPONSE_YES)
 237         dialog.set_default_response(gtk.RESPONSE_YES)
 238         response = self.flash_dialog(dialog)
 239         if response != gtk.RESPONSE_YES:
 240             raise Default

 260     def _show_translation_warning_dialog(self, page):
 261         """Show a warning dialog if opening a new translation file.
 262 
 263         Raise Default to abort.
 264         """
 265         title = _('Save changes to translation document "%s" before opening a '
 266             'new one?') % page.get_translation_basename()
 267         message = _("If you don't save, changes will be permanently lost.")
 268         dialog = WarningDialog(self.window, title, message)
 269         dialog.add_button(_("Open _Without Saving"), gtk.RESPONSE_NO)
 270         dialog.add_button(gtk.STOCK_CANCEL, gtk.RESPONSE_CANCEL)
 271         dialog.add_button(gtk.STOCK_SAVE, gtk.RESPONSE_YES)
 272         dialog.set_default_response(gtk.RESPONSE_YES)
 273         response = self.flash_dialog(dialog)
 274         if response == gtk.RESPONSE_YES:
 275             return self.save_translation(page)
 276         if response != gtk.RESPONSE_NO:
 277             raise Default

 279     def _try_open_file(self, method, path, encoding):
 280         """Try to open file and return sort count.
 281 
 282         Raise Default if reading file fails.
 283         Return None if decoding file fails.
 284         """
 285         util.set_cursor_busy(self.window)
 286         try:
 287             if encoding == "auto":
 288                 encoding = enclib.detect(path)
 289             return method(path, encoding)
 290         except FormatError:
 291             util.set_cursor_normal(self.window)
 292             basename = os.path.basename(path)
 293             self._show_format_error_dialog(basename)
 294             raise Default
 295         except IOError, (no, message):
 296             util.set_cursor_normal(self.window)
 297             basename = os.path.basename(path)
 298             self._show_io_error_dialog(basename, message)
 299             raise Default
 300         except (UnicodeError, ValueError):
 301             return None
 302         finally:
 303             util.set_cursor_normal(self.window)

 305     def add_new_page(self, page):
 306         """Add a new page."""
 307 
 308         self.pages.append(page)
 309         page.connect("close-request", self.on_page_close_request)
 310         page.project.connect("action-done", self.on_project_action_done)
 311         page.project.connect("action-redone", self.on_project_action_redone)
 312         page.project.connect("action-undone", self.on_project_action_undone)
 313         method = self.on_page_tab_widget_button_press_event
 314         page.tab_widget.connect("button-press-event", method)
 315         self.connect_to_view_signals(page.view)
 316         page.project.clipboard.data = self.clipboard.data[:]
 317 
 318         scroller = gtk.ScrolledWindow()
 319         scroller.set_policy(gtk.POLICY_AUTOMATIC, gtk.POLICY_AUTOMATIC)
 320         scroller.add(page.view)
 321         self.notebook.append_page(scroller, page.tab_widget)
 322         self.notebook.set_tab_reorderable(scroller, True)
 323         self.notebook.show_all()
 324         self.notebook.set_current_page(self.pages.index(page))
 325         self.emit("page-added", page)

 327     def add_to_recent_files(self, path, doc):
 328         """Add path to recent files."""
 329 
 330         group = ("gaupol-main", "gaupol-translation")[doc]
 331         self.recent_manager.add_full(
 332             util.path_to_uri(path), {
 333                 "mime_type": "text/plain",
 334                 "app_name": "gaupol",
 335                 "app_exec": "gaupol %F",
 336                 "groups": [group],})

 338     def connect_to_view_signals(self, view):
 339         """Connect to signals emitted by view."""
 340 
 341         selection = view.get_selection()
 342         selection.connect("changed", self.on_view_selection_changed)
 343         view.connect_after("move-cursor", self.on_view_move_cursor)
 344         view.connect("button-press-event", self.on_view_button_press_event)
 345 
 346         for i, column in enumerate(view.get_columns()):
 347             renderer = column.get_cell_renderers()[0]
 348             method = self.on_view_renderer_edited
 349             renderer.connect("edited", method)
 350             method = self.on_view_renderer_editing_started
 351             renderer.connect("editing-started", method)
 352             method = self.on_view_renderer_editing_canceled
 353             renderer.connect("editing-canceled", method)
 354             button = util.get_parent(column.get_widget(), gtk.Button)
 355             method = self.on_view_header_button_press_event
 356             button.connect("button-press-event", method)

 358     @util.gc_collected
 359     @util.ignore_exceptions(Default)
 360     def on_append_file_activate(self, *args):
 361         """Append subtitles from file to the current project."""
 362 
 363         util.set_cursor_busy(self.window)
 364         dialog = AppendDialog(self.window)
 365         util.set_cursor_normal(self.window)
 366         response = self.run_dialog(dialog)
 367         if response != gtk.RESPONSE_OK:
 368             return dialog.destroy()
 369         path = dialog.get_filenames()[0]
 370         encoding = dialog.get_encoding()
 371         dialog.destroy()
 372         while gtk.events_pending():
 373             gtk.main_iteration()
 374 
 375         encodings = self._get_encodings(encoding)
 376         temp = self._open_file(path, encodings, cons.DOCUMENT.MAIN, False)
 377         util.set_cursor_busy(self.window)
 378         current = self.get_current_page()
 379         self._pre_append_shift(current, temp)
 380         rows = self._append_subtitles(current, temp)
 381         current.view.set_focus(rows[0], None)
 382         current.view.select_rows(rows)
 383         current.view.scroll_to_row(rows[0])
 384         basename = temp.get_main_basename()
 385         fields = {'amount': len(rows), 'filename': basename}
 386         self.push_message(
 387             _('Appended %(amount)d subtitles from "%(filename)s"') % fields)
 388         util.set_cursor_normal(self.window)

 390     def on_new_project_activate(self, *args):
 391         """Create a new project."""
 392 
 393         util.set_cursor_busy(self.window)
 394         self.counter += 1
 395         page = Page(self.counter)
 396         page.project.insert_blank_subtitles([0], register=None)
 397         self.add_new_page(page)
 398         util.set_cursor_normal(self.window)

 422

 486     def open_main_files(self, paths, encoding=None):
 487         """Open main files."""
 488 
 489         encodings = self._get_encodings(encoding)
 490         for path in sorted(paths):
 491             try:
 492                 page = self._open_file(path, encodings, cons.DOCUMENT.MAIN)
 493             except Default:
 494                 continue
 495             util.set_cursor_busy(self.window)
 496             self.add_new_page(page)
 497             self.add_to_recent_files(path, cons.DOCUMENT.MAIN)
 498             basename = page.get_main_basename()
 499             self.push_message(_('Opened main file "%s"') % basename)
 500             while gtk.events_pending():
 501                 gtk.main_iteration()
 502             util.set_cursor_normal(self.window)

 504     @util.ignore_exceptions(Default)
 505     def open_translation_file(self, path, encoding=None):
 506         """Open translation file."""
 507 
 508         encodings = self._get_encodings(encoding)
 509         page = self._open_file(path, encodings, cons.DOCUMENT.TRAN)
 510         util.set_cursor_busy(self.window)
 511         if not page.view.get_column(TTXT).get_visible():
 512             self.uim.get_action(TTXT.uim_path).activate()
 513         self.add_to_recent_files(path, cons.DOCUMENT.TRAN)
 514         basename = page.get_translation_basename()
 515         self.push_message(_('Opened translation file "%s"') % basename)
 516         util.set_cursor_normal(self.window)

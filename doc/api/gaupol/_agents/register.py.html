  19 """Managing revertable actions.
  20 
  21 To hook a method up with the undo/redo system, the following need to be done:
  22 
  23 (1) The last argument to the method must be keyword argument 'register' with a
  24     default value of -1. This argument indicates which of doing, undoing or
  25     redoing is in process.
  26 
  27 (2) At the end of the method, self.register_action(...) needs to be called with
  28     keyword arguments specified in RevertableAction.__init__, where they're
  29     directly passed to. Calling this method takes care of the management of
  30     undo and redo stacks.
  31 
  32 (3) The method should be marked with the 'revertable' decorator. This decorator
  33     takes care of emitting an action signal once the method has been run, cuts
  34     the undo and redo stacks if needed and defaults the 'register' keyword
  35     argument to cons.REGISTER.DO.
  36 
  37 Each method marked as revertable should match exactly one action in the undo
  38 and redo stacks. Hence, if the method calls other revertable methods, the
  39 resulting action needs to be grouped as one using self.group_actions(...).
  40 
  41 If a revertable method needs to be performed without the possibility of
  42 reverting, the 'register' keyword argument can be given a value of None. This
  43 way it will not be in any way processed by the undo/redo system.
  44 """
  45 
  46 
  47 import functools
  48 
  49 from gaupol import cons, util
  50 from gaupol.base import Delegate
  51 
  52 
  53 def revertable(function):
  54     """Decorator to handle handle function with the action reversion system."""
  55 
  56     @functools.wraps(function)
  57     def wrapper(*args, **kwargs):
  58         project = args[0]
  59         main_changed = project.main_changed
  60         tran_changed = project.tran_changed
  61         if not "register" in kwargs:
  62             kwargs["register"] = cons.REGISTER.DO
  63         register = kwargs["register"]
  64         if register is None:
  65             return function(*args, **kwargs)
  66         blocked = project.block(register.signal)
  67         value = function(*args, **kwargs)
  68         if blocked:
  69             project.cut_reversion_stacks()
  70             project.unblock(register.signal)
  71             if project.main_changed != main_changed or \
               project.tran_changed != tran_changed:
  72                 project.emit_action_signal(register)
  73         return value
  74 
  75     return wrapper
  76 
  77 
  78 class RevertableAction(object):
  79 
  80     """Action that can be reverted (undone or redone).
  81 
  82     Instance variables:
  83 
  84         description:   Short one line description
  85         docs:          List of DOCUMENT constants
  86         register:      REGISTER constant
  87         revert_args:   Arguments passed to the revert method
  88         revert_kwargs: Keyword arguments passed to the revert method
  89         revert_method: Method called for reversion
  90     """
  91 
  92     def __init__(self, register, docs, description,
  93         revert_method, revert_args=None, revert_kwargs=None):
  94 
  95         revert_args = (revert_args if revert_args is not None else [])
  96         revert_kwargs = (revert_kwargs if revert_kwargs is not None else {})
  97 
  98         self.description   = description
  99         self.docs          = docs
 100         self.register      = register
 101         self.revert_args   = revert_args
 102         self.revert_kwargs = revert_kwargs
 103         self.revert_method = revert_method
 104 
 105     def revert(self):
 106         """Call the revert method."""
 107 
 108         register = None
 109         if self.register.shift == 1:
 110             register = cons.REGISTER.UNDO
 111         if self.register.shift == -1:
 112             register = cons.REGISTER.REDO
 113         self.revert_kwargs["register"] = register
 114         self.revert_method(*self.revert_args, **self.revert_kwargs)
 115 
 116 
 117 class RevertableActionGroup(object):
 118 
 119     """Group of revertable actions.
 120 
 121     Instance variables:
 122 
 123         actions:     List of actions in group
 124         description: Short one line description
 125     """
 126 
 127     def __init__(self, actions, description):
 128 
 129         self.actions     = actions
 130         self.description = description
 131 
 132 
 133 class RegisterAgent(Delegate):
 134 
 135     """Managing revertable actions.
 136 
 137     Instance variables:
 138 
 139         _do_description: The original description of the reverted action
 140     """
 141 
 142     # pylint: disable-msg=E0203,W0201
 143 
 144     def __init__(self, master):
 145 
 146         Delegate.__init__(self, master)
 147 
 148         self._do_description = None
 149 
 150         util.connect(self, self, "notify::undo_limit")
 151 
 152     def _break_action_group(self, stack, index=0):
 153         """Break the action group in stack into individual actions.
 154 
 155         Return the amount of actions broken into.
 156         """
 157         action_group = stack.pop(index)
 158         for action in reversed(action_group.actions):
 159             stack.insert(index, action)
 160         return len(action_group.actions)
 161 
 162     def _get_destination_stack(self, register):
 163         """Get the stack where the registered action will be placed."""
 164 
 165         if register.shift == 1:
 166             return self.undoables
 167         if register.shift == -1:
 168             return self.redoables
 169         raise ValueError
 170 
 171     def _get_source_stack(self, register):
 172         """Get the stack where the action to register is."""
 173 
 174         if register.shift == 1:
 175             return self.redoables
 176         if register.shift == -1:
 177             return self.undoables
 178         raise ValueError
 179 
 180     def _on_notify_undo_limit(self, *args):
 181         """Cut reversion stacks if limit set."""
 182 
 183         if self.undo_limit is not None:
 184             self.cut_reversion_stacks()
 185 
 186     def _revert_multiple(self, count, register):
 187         """Revert multiple actions."""
 188 
 189         self.block(register.signal)
 190         stack = self._get_source_stack(register)
 191         for i in range(count):
 192             sub_count = 1
 193             if isinstance(stack[0], RevertableActionGroup):
 194                 description = stack[0].description
 195                 sub_count = self._break_action_group(stack)
 196             for j in range(sub_count):
 197                 self._do_description = stack[0].description
 198                 stack.pop(0).revert()
 199             if sub_count > 1:
 200                 self.group_actions(register, sub_count, description)
 201         self.unblock(register.signal)
 202         self.cut_reversion_stacks()
 203         self.emit_action_signal(register, count)
 204 
 205     def _shift_changed_value(self, action, shift):
 206         """Shift the values of the changed attributes."""
 207 
 208         if cons.DOCUMENT.MAIN in action.docs:
 209             self.main_changed += shift
 210         if cons.DOCUMENT.TRAN in action.docs:
 211             self.tran_changed += shift
 212 
 213         if action.docs == [cons.DOCUMENT.TRAN]:
 214             self.tran_active = True
 215 
 216     def can_redo(self):
 217         """Return True if something can be redone."""
 218 
 219         return bool(self.redoables)
 220 
 221     def can_undo(self):
 222         """Return True if something can be undone."""
 223 
 224         return bool(self.undoables)
 225 
 226     def cut_reversion_stacks(self):
 227         """Cut the undo and redo stacks to their maximum lengths."""
 228 
 229         if self.undo_limit is not None:
 230             del self.redoables[self.undo_limit:]
 231             del self.undoables[self.undo_limit:]
 232 
 233     def emit_action_signal(self, register, count=1):
 234         """Emit an action signal for register."""
 235 
 236         if register is not None:
 237             action = self._get_destination_stack(register)[0]
 238             self.emit(register.signal, action)
 239 
 240     def group_actions(self, register, count, description):
 241         """Group the registered actions as one item in the stack."""
 242 
 243         if register is not None:
 244             actions = []
 245             stack = self._get_destination_stack(register)
 246             for i in range(count):
 247                 actions.append(stack.pop(0))
 248             action_group = RevertableActionGroup(actions, description)
 249             stack.insert(0, action_group)
 250 
 251     def redo(self, count=1):
 252         """Redo actions."""
 253 
 254         if count > 1 or isinstance(self.redoables[0], RevertableActionGroup):
 255             return self._revert_multiple(count, cons.REGISTER.REDO_MULTIPLE)
 256         self._do_description = self.redoables[0].description
 257         self.redoables.pop(0).revert()
 258 
 259     def register_action(self, *args, **kwargs):
 260         """Register action done, undone or redone.
 261 
 262         See RevertableAction.__init__ for arguments.
 263         """
 264         action = RevertableAction(*args, **kwargs)
 265         if action.register is None:
 266             return
 267         if action.register == cons.REGISTER.DO:
 268             self.undoables.insert(0, action)
 269             self.redoables = []
 270         elif action.register == cons.REGISTER.UNDO:
 271             self.redoables.insert(0, action)
 272             action.description = self._do_description
 273         elif action.register == cons.REGISTER.REDO:
 274             self.undoables.insert(0, action)
 275             action.description = self._do_description
 276         self._shift_changed_value(action, action.register.shift)
 277 
 278     def set_action_description(self, register, description):
 279         """Set the description of the most recent action."""
 280 
 281         if register is not None:
 282             stack = self._get_destination_stack(register)
 283             stack[0].description = description
 284 
 285     def undo(self, count=1):
 286         """Undo actions."""
 287 
 288         if count > 1 or isinstance(self.undoables[0], RevertableActionGroup):
 289             return self._revert_multiple(count, cons.REGISTER.UNDO_MULTIPLE)
 290         self._do_description = self.undoables[0].description
 291         self.undoables.pop(0).revert()

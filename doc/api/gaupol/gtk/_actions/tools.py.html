  27 class AppendFileAction(UIMAction):
  28 
  29     """Append subtitles from file to the current project."""
  30 
  31     action_item = (
  32         "append_file",
  33         None,
  34         _("_Append File..."),
  35         None,
  36         _("Append subtitles from file to the current project"),)
  37 
  38     paths = ["/ui/menubar/tools/append_file"]
  39 
  40     @classmethod
  41     def is_doable(cls, application, page):
  42         """Return True if action can be done."""
  43 
  44         return (page is not None)

  47 class SplitProjectAction(UIMAction):
  48 
  49     """Split the current project in two."""
  50 
  51     action_item = (
  52         "split_project",
  53         None,
  54         _("_Split Project..."),
  55         None,
  56         _("Split the current project in two"),)
  57 
  58     paths = ["/ui/menubar/tools/split_project"]
  59 
  60     @classmethod
  61     def is_doable(cls, application, page):
  62         """Return True if action can be done."""
  63 
  64         if page is not None:
  65             return len(page.project.times) > 1
  66         return False

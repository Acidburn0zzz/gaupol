  30 class ClearTextsAction(UIMAction):
  31 
  32     """Clear the selected texts."""
  33 
  34     action_item = (
  35         "clear_texts",
  36         gtk.STOCK_CLEAR,
  37         _("C_lear"),
  38         "C",
  39         _("Clear the selected texts"),)
  40 
  41     paths = ["/ui/menubar/edit/clear"]
  42 
  43     @classmethod
  44     def is_doable(cls, application, page):
  45         """Return True if action can be done."""
  46 
  47         if page is not None:
  48             selection = bool(page.view.get_selected_rows())
  49             focus = page.view.get_focus()[1] in (MTXT, TTXT)
  50             return bool(selection and focus)
  51         return False

 116 class EditValueAction(UIMAction):
 117 
 118     """Edit the focused cell."""
 119 
 120     action_item = (
 121         "edit_value",
 122         gtk.STOCK_EDIT,
 123         _("_Edit"),
 124         "Return",
 125         _("Edit the focused cell"),)
 126 
 127     paths = ["/ui/menubar/edit/edit", "/ui/view_popup/edit"]
 128 
 129     @classmethod
 130     def is_doable(cls, application, page):
 131         """Return True if action can be done."""
 132 
 133         if page is not None:
 134             row, col = page.view.get_focus()
 135             if None in (row, col):
 136                 return False
 137             return (col != NO)
 138         return False

 141 class InsertSubtitlesAction(UIMAction):
 142 
 143     """Insert blank subtitles."""
 144 
 145     action_item = (
 146         "insert_subtitles",

 154     @classmethod
 155     def is_doable(cls, application, page):

 188 class MergeSubtitlesAction(UIMAction):
 189 
 190     """Merge the selected subtitles."""
 191 
 192     action_item = (
 193         "merge_subtitles",
 194         None,
 195         _("_Merge Subtitles"),
 196         "M",
 197         _("Merge the selected subtitles"),)
 198 
 199     paths = ["/ui/menubar/edit/merge", "/ui/view_popup/merge"]
 200 
 201     @classmethod
 202     def is_doable(cls, application, page):
 203         """Return True if action can be done."""
 204 
 205         if page is not None:
 206             rows = page.view.get_selected_rows()
 207             if (len(rows) > 1) and (rows[-1] == rows[0] + len(rows) - 1):
 208                 return True
 209         return False

  19 """Text parser for tag-aware editing."""
  20 
  21 
  22 from gaupol.finder import Finder
  23 
  24 
  25 class Parser(Finder):
  26 
  27     """Text parser for tag-aware editing.
  28 
  29     Instance variables:
  30 
  31         re_tag:  Regular expression object to match any tag
  32         margins: Start tag, end tag that every line is wrapped in
  33         tags:    List of lists of tag, position
  34 
  35     The purpose of the Parser is to split text to the actual text and its tags,
  36     allowing the text to be edited while keeping the tags separate and intact.
  37     Parser can be used by first setting text to it, then performing operations
  38     via defined methods and finally getting the full text back.
  39 
  40     The margin system is only used if no other tags are found in the text and
  41     if the text has at least two lines. Either margins or tags will always be
  42     empty.
  43     """
  44 
  45     def __init__(self, re_tag=None):
  46         """Initialize Parser.
  47 
  48         re_tag should be a regular expression object.
  49         """
  50         Finder.__init__(self)
  51 
  52         self.re_tag  = re_tag
  53         self.margins = None
  54         self.tags    = None
  55 
  56     def _set_margins(self, text):
  57         """Find the margin tags in text if such exist."""
  58 
  59         lines = text.split("\n")
  60         line = lines[0]
  61         start_tag = ""
  62         while True:
  63             match = self.re_tag.match(line)
  64             if match is None:
  65                 break
  66             a, z = match.span()
  67             start_tag += line[a:z]
  68             line = line[z:]
  69         if not start_tag:
  70             return
  71 
  72         end_tag = ""
  73         while True:
  74             iterator = self.re_tag.finditer(line)
  75             match = None
  76             while True:
  77                 try:
  78                     match = iterator.next()
  79                 except StopIteration:
  80                     break
  81             if match is None:
  82                 break
  83             a, z = match.span()
  84             if z != len(line):
  85                 return
  86             end_tag = line[a:z] + end_tag
  87             line = line[:a]
  88 
  89         if all(list(x.startswith(start_tag) for x in lines)):
  90             if all(list(x.endswith(end_tag) for x in lines)):
  91                 self.margins = [start_tag, end_tag]
  92 
  93     def _shift_tags(self, pos, shift, orig_text):
  94         """Shift all the tags after position."""
  95 
  96         if not shift or not self.tags:
  97             return
  98 
  99         # Shift tags at position if at a closing tag and shift is positive,
 100         # otherwise shift only tags after position. This should keep tags at
 101         # the very outer edges of words.
 102         opening = True
 103         if shift > 0:
 104             if pos < len(orig_text):
 105                 if orig_text[pos].isspace():
 106                     opening = False
 107             elif pos == len(orig_text):
 108                 opening = False
 109 
 110         # Get length of tags before position.
 111         pos_with_tags = pos
 112         for pos, tag in self.tags:
 113             if opening and pos <= pos_with_tags:
 114                 pos_with_tags += len(tag)
 115             elif pos < pos_with_tags:
 116                 pos_with_tags += len(tag)
 117 
 118         # Shift tags.
 119         for i in range(len(self.tags)):
 120             if opening and self.tags[i][0] > pos_with_tags:
 121                 self.tags[i][0] += shift
 122             elif self.tags[i][0] >= pos_with_tags:
 123                 self.tags[i][0] += shift
 124             if self.tags[i][0] < 0:
 125                 self.tags[i][0] = 0
 126 
 127     def get_text(self):
 128         """Reassemble the text and return it."""
 129 
 130         text = self.text[:]
 131         for pos, tag in self.tags:
 132             text = text[:pos] + tag + text[pos:]
 133         if self.margins:
 134             text = text.replace("\n", "%s\n%s" % tuple(self.margins[::-1]))
 135             text = self.margins[0] + text + self.margins[1]
 136         return text
 137 
 138     def replace(self, next=True):
 139         """Replace the current match.
 140 
 141         next should be True to finish at end of match, False for beginning.
 142         """
 143         a = self.match_span[0]
 144         orig_text = self.text[:]
 145         Finder.replace(self, next)
 146         shift = len(self.text) - len(orig_text)
 147         self._shift_tags(a, shift, orig_text)
 148 
 149     def set_text(self, text, next=True):
 150         """Set the text and parse it.
 151 
 152         next should be True to start at beginning, False for end.
 153         """
 154         Finder.set_text(self, text, next)
 155 
 156         self.margins = []
 157         self.tags = []
 158         if self.re_tag is None:
 159             return
 160         if text.count("\n"):
 161             self._set_margins(text)
 162         if not self.margins:
 163             for match in self.re_tag.finditer(text):
 164                 a, z = match.span()
 165                 self.tags.append([a, text[a:z]])
 166         self.text = self.re_tag.sub("", text)

  19 """Basic subtitle data editing."""
  20 
  21 
  22 from __future__ import division
  23 from gettext import gettext as _
  24 
  25 from gaupol import cons, util
  26 from gaupol.base import Delegate, notify_frozen
  27 from .index import SHOW, HIDE, DURN
  28 from .register import revertable
  29 
  30 
  31 class EditAgent(Delegate):
  32 
  33     """Basic subtitle data editing."""
  34 
  35     # pylint: disable-msg=E0203,W0201
  36 
  37     @revertable
  38     def clear_texts(self, rows, doc, register=-1):
  39         """Clear texts."""
  40 
  41         new_texts = [u""] * len(rows)
  42         self.replace_texts(rows, doc, new_texts, register=register)
  43         self.set_action_description(register, _("Clearing texts"))
  44 
  45     def copy_texts(self, rows, doc):
  46         """Copy texts to the clipboard."""
  47 
  48         data = []
  49         texts = self.get_texts(doc)
  50         for row in range(min(rows), max(rows) + 1):
  51             item = (texts[row] if row in rows else None)
  52             data.append(item)
  53         self.clipboard.data = data
  54 
  55     @revertable
  56     def cut_texts(self, rows, doc, register=-1):
  57         """Cut texts to the clipboard."""
  58 
  59         self.copy_texts(rows, doc)
  60         self.clear_texts(rows, doc, register=register)
  61         self.set_action_description(register, _("Cutting texts"))
  62 
  63     @revertable
  64     @notify_frozen
  65     def insert_blank_subtitles(self, rows, register=-1):
  66         """Insert blank subtitles."""
  67 
  68         mode = self.get_mode()
  69         all_rows = util.get_sorted_unique(rows)
  70         for rows in util.get_ranges(all_rows):
  71             if mode == cons.MODE.TIME:
  72                 start = 0.0
  73                 if min(rows) > 0:
  74                     start = self.times[min(rows) - 1][HIDE]
  75                     start = self.calc.time_to_seconds(start)
  76                 duration = 3.0
  77                 if min(rows) < len(self.times):
  78                     end = self.times[min(rows)][SHOW]
  79                     end = self.calc.time_to_seconds(end)
  80                     duration = (end - start) / len(rows)
  81             if mode == cons.MODE.FRAME:
  82                 start = 0
  83                 if min(rows) > 0:
  84                     start = self.frames[min(rows) - 1][HIDE]
  85                 duration = self.calc.seconds_to_frame(3.0)
  86                 if min(rows) < len(self.times):
  87                     end = self.frames[min(rows)][SHOW]
  88                     duration = int((end - start) / len(rows))
  89             for i, row in enumerate(rows):
  90                 time, frame = self.expand_positions(
  91                     start + (i * duration), start + ((i + 1) * duration))
  92                 self.times.insert(row, time)
  93                 self.frames.insert(row, frame)
  94                 self.main_texts.insert(row, u"")
  95                 self.tran_texts.insert(row, u"")
  96 
  97         self.register_action(
  98             register=register,
  99             docs=[cons.DOCUMENT.MAIN, cons.DOCUMENT.TRAN],
 100             description=_("Inserting subtitles"),
 101             revert_method=self.remove_subtitles,
 102             revert_args=[all_rows],)
 103 
 104         self.emit("subtitles-inserted", all_rows)
 105 
 106     @revertable
 107     def merge_subtitles(self, rows, register=-1):
 108         """Merge subtitles."""
 109 
 110         rows = sorted(rows)
 111         positions = self.get_positions()
 112         show = positions[rows[0]][SHOW]
 113         hide = positions[rows[-1]][HIDE]
 114         time, frame = self.expand_positions(show, hide)
 115         main_texts = list(self.main_texts[x] for x in rows)
 116         tran_texts = list(self.tran_texts[x] for x in rows)
 117         for texts in (main_texts, tran_texts):
 118             while "" in texts:
 119                 texts.remove("")
 120         main_text = "\n".join(main_texts)
 121         tran_text = "\n".join(tran_texts)
 122 
 123         self.remove_subtitles(rows, register=register)
 124         self.insert_subtitles(
 125             [rows[0]], [time], [frame], [main_text], [tran_text],
 126             register=register)
 127         self.group_actions(register, 2, _("Merging subtitles"))
 128 
 129     @revertable
 130     def paste_texts(self, row, doc, register=-1):
 131         """Paste texts from the clipboard.
 132 
 133         Return the rows that were pasted into.
 134         """
 135         data = self.clipboard.data
 136         excess = len(data) - (len(self.times) - row)
 137         if excess > 0:
 138             rows = range(len(self.times), len(self.times) + excess)
 139             self.insert_blank_subtitles(rows, register=register)
 140 
 141         rows = []
 142         new_texts = []
 143         for i, text in enumerate(data):
 144             if text is not None:
 145                 rows.append(row + i)
 146                 new_texts.append(text)
 147         self.replace_texts(rows, doc, new_texts, register=register)
 148         if excess > 0:
 149             self.group_actions(register, 2, "")
 150         self.set_action_description(register, _("Pasting texts"))
 151         return rows
 152 
 153     @revertable
 154     @notify_frozen
 155     def remove_subtitles(self, rows, register=-1):
 156         """Remove subtitles."""
 157 
 158         times = []
 159         frames = []
 160         main_texts = []
 161         tran_texts = []
 162         rows = sorted(rows)
 163         for row in reversed(rows):
 164             times.insert(0, self.times.pop(row))
 165             frames.insert(0, self.frames.pop(row))
 166             main_texts.insert(0, self.main_texts.pop(row))
 167             tran_texts.insert(0, self.tran_texts.pop(row))
 168 
 169         self.register_action(
 170             register=register,
 171             docs=[cons.DOCUMENT.MAIN, cons.DOCUMENT.TRAN],
 172             description=_("Removing subtitles"),
 173             revert_method=self.insert_subtitles,
 174             revert_args=[rows, times, frames, main_texts, tran_texts],)
 175 
 176         self.emit("subtitles-removed", rows)
 177 
 178     @revertable
 179     def split_subtitle(self, row, register=-1):
 180         """Split subtitle in two."""
 181 
 182         positions = self.get_positions()
 183         show = positions[row][SHOW]
 184         hide = positions[row][HIDE]
 185         middle = self.calc.get_middle(show, hide)
 186         time_1, frame_1 = self.expand_positions(show, middle)
 187         time_2, frame_2 = self.expand_positions(middle, hide)
 188         main_text = self.main_texts[row]
 189         tran_text = self.tran_texts[row]
 190 
 191         self.remove_subtitles([row], register=register)
 192         self.insert_subtitles(
 193             [row, row + 1],
 194             [time_1, time_2],
 195             [frame_1, frame_2],
 196             [main_text, u""],
 197             [tran_text, u""],
 198             register=register)
 199         self.group_actions(register, 2, _("Splitting subtitle"))

  19 """Dialog for displaying credits and information.
  20 
  21 Module variables:
  22 
  23     _LICENSE:     Unwrapped license notice

  28 
  29 import os
  30 import textwrap
  31 from gettext import gettext as _
  32 
  33 import gtk
  34 
  35 from gaupol import urls, util, __version__
  36 from gaupol.gtk import paths
  37 
  38 
  39 _LOGO = os.path.join(paths.DATA_DIR, "icons", "logo.png")
  40 
  41 # Translators: This is a special message that shouldn't be translated
  42 # literally. It is used in the about dialog to give credits to the translators.
  43 # Thus, you should translate it to your name and email address.  You can also
  44 # include other translators who have contributed to this translation; in that
  45 # case, please write them on separate lines seperated by newlines (\n).
  46 _TRANSLATORS = _("translator-credits")
  47 
  48 _LICENSE = \
"""Gaupol is free software; you can redistribute it and/or modify it under
  49 the terms of the GNU General Public License as published by the Free
  50 Software Foundation; either version 2 of the License, or (at your option)
  51 any later version.""".replace("\n" "") + "\n\n" + \
"""Gaupol is distributed in the hope that it will be useful, but WITHOUT ANY
  52 WARRANTY; without even the implied warranty of MERCHANTABILITY or FITNESS
  53 FOR A PARTICULAR PURPOSE. See the GNU General Public License for more
  54 details.""".replace("\n" "") + "\n\n" + \
"""You should have received a copy of the GNU General Public License along
  55 with Gaupol; if not, write to the Free Software Foundation, Inc., 51
  56 Franklin Street, Fifth Floor, Boston, MA 02110-1301, USA."""
  57 
  58 
  59 class AboutDialog(gtk.AboutDialog):
  60 
  61     """Dialog for displaying credits and information."""
  62 
  63     def __init__(self, parent):
  64 
  65         gtk.AboutDialog.__init__(self)
  66 
  67         self.set_name("Gaupol")
  68         self.set_version(__version__)
  69         self.set_copyright(u"Copyright \xa9 2005-2007 Osmo Salomaa")
  70         self.set_comments(_("Subtitle editor"))
  71         self.set_website_label(urls.HOMEPAGE)
  72         self.set_authors(["Osmo Salomaa <otsaloma@cc.hut.fi>"])
  73         self.set_artists(["Osmo Salomaa <otsaloma@cc.hut.fi>"])
  74         self.set_translator_credits(_TRANSLATORS)
  75         self.set_license(_LICENSE)
  76         self.set_logo(gtk.gdk.pixbuf_new_from_file(_LOGO))
  77 
  78         gtk.about_dialog_set_url_hook(self._on_url_clicked)
  79         self.set_wrap_license(True)
  80         self.set_transient_for(parent)
  81 
  82     def _on_url_clicked(self, *args):
  83         """Open website in a web browser."""
  84 
  85         util.browse_url(urls.HOMEPAGE)

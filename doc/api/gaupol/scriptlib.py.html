  19 """Script-specific regular expression patterns.
  20 
  21 Module variables:
  22 
  23     CAP_AFTERS:    Tuple of tuples of script, display name, pattern
  24     CLAUSE_SEPS:   Tuple of tuples of script, display name, pattern
  25     DIALOGUE_SEPS: Tuple of tuples of script, display name, pattern
  26 
  27 Regular expression patterns in 'CLAUSE_SEPS' and 'DIALOGUE_SEPS' are required
  28 to have named groups 'before' and 'after'. Space (or whatever) between those
  29 groups acts as the separator. Either of the groups can be empty.
  30 """
  31 
  32 
  33 from gettext import gettext as _
  34 
  35 
  36 CAP_AFTERS = (
  37     ("latin", _("Latin"), r"((?<!(.\.|\..))\.|\?|\!)(\'|\")?( |$)"),)
  38 
  39 CLAUSE_SEPS = (
  40     ("latin", _("Latin"),
  41      r"(?P<before>" + \
         r"(" + \
             r"((?<!(.\.|\..))\.|[\?\!])[\'\"]?" + \
         r"|" + \
             r"(?<!-)" + \
             r" [^ ]+?(\.{2,}|[\,\;\:])[\'\"]?" + \
             r"( [^ ]+?([^\.].\.{3,}|[\,\;\:])[\'\"]?)*" + \
             r"( [^ ]+?([^\.].\.|[\?\!])[\'\"]?)?" + \
             r"( [^ ]+$)?" + \
         r")" + \
     r")" + \
     r"( |$)" + \
     r"(?P<after>)"),
  42     ("latin-english", _("Latin (English)"),
  43      r"(?P<before>" + \
         r"(" + \
             r"(" + \
                 r"(?<!\b([Dd]r|[Jj]r|[Mm]r|[Mm]s|[Ss]r|[Ss]t))" + \
                 r"(?<!\b[Mm]rs)" + \
             r"(?<!(.\.|\..))\.|[\?\!])[\'\"]?" + \
         r"|" + \
             r"(?<!-)" + \
             r" [^ ]+?(\.{2,}|[\,\;\:])[\'\"]?" + \
             r"( [^ ]+?([^\.].\.{3,}|[\,\;\:])[\'\"]?)*" + \
             r"( [^ ]+?([^\.].\.|[\?\!])[\'\"]?)?" + \
             r"( [^ ]+$)?" + \
         r")" + \
     r")" + \
     r"( |$)" + \
     r"(?P<after>)"),)
  44 
  45 DIALOGUE_SEPS = (
  46     ("latin", _("Latin"), r"(?P<before>) (?P<after>- )"),)
  47 
  48 
  49 def get_capitalize_after(script):
  50     """Get the regular expression to capitalize after for script.
  51 
  52     Return regular expression pattern, flags.
  53     """
  54     for seq in CAP_AFTERS:
  55         if seq[0] == script:
  56             return seq[2]
  57     raise ValueError
  58 
  59 def get_clause_separator(script):
  60     """Get the regular expression for a clause separator for script.
  61 
  62     Return regular expression pattern.
  63     """
  64     for seq in CLAUSE_SEPS:
  65         if seq[0] == script:
  66             return seq[2]
  67     raise ValueError
  68 
  69 def get_dialogue_separator(script):
  70     """Get the regular expression for a dialogue separator for script.
  71 
  72     Return regular expression pattern.
  73     """
  74     for seq in DIALOGUE_SEPS:
  75         if seq[0] == script:
  76             return seq[2]
  77     raise ValueError

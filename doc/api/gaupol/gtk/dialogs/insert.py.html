  19 """Dialog for inserting new subtitles."""
  20 
  21 
  22 import gtk
  23 
  24 from gaupol import util
  25 from gaupol.gtk import conf
  26 from .glade import GladeDialog
  27 
  28 
  29 class InsertDialog(GladeDialog):
  30 
  31     """Dialog for inserting new subtitles.
  32 
  33     Instance variables:
  34 
  35         _amount_spin:    gtk.SpinButton
  36         _position_combo: gtk.ComboBox
  37         _position_label: gtk.Label
  38         application:     Associated application
  39     """
  40 
  41     def __init__(self, parent, application):
  42 
  43         GladeDialog.__init__(self, "insert-dialog")
  44 
  45         self._amount_spin    = self._glade_xml.get_widget("amount_spin")
  46         self._position_combo = self._glade_xml.get_widget("position_combo")
  47         self._position_label = self._glade_xml.get_widget("position_label")
  48         self.application     = application
  49 
  50         self._init_data()
  51         self._init_signal_handlers()
  52         self._dialog.set_transient_for(parent)
  53         self._dialog.set_default_response(gtk.RESPONSE_OK)
  54 
  55     def _init_data(self):
  56         """Initialize default values for widgets."""
  57 
  58         self._amount_spin.set_value(conf.subtitle_insert.amount)
  59         active = (0 if conf.subtitle_insert.above else 1)
  60         self._position_combo.set_active(active)
  61 
  62         page = self.application.get_current_page()
  63         if not page.project.times:
  64             self._position_combo.set_sensitive(False)
  65             self._position_label.set_sensitive(False)
  66 
  67     def _init_signal_handlers(self):
  68         """Initialize signal handlers."""
  69 
  70         util.connect(self, self, "response")
  71 
  72     def _insert_subtitles(self, amount, above):
  73         """Insert subtitles."""
  74 
  75         row = 0
  76         page = self.application.get_current_page()
  77         if page.project.times:
  78             row = page.view.get_selected_rows()[0]
  79             if not above:
  80                 row += 1
  81         rows = range(row, row + amount)
  82         page.project.insert_blank_subtitles(rows)
  83 
  84     def _on_response(self, dialog, response):
  85         """Save values and insert subtitles."""
  86 
  87         amount = self._amount_spin.get_value_as_int()
  88         above = (self._position_combo.get_active() == 0)
  89         conf.subtitle_insert.amount = amount
  90         conf.subtitle_insert.above = above
  91         if response == gtk.RESPONSE_OK:
  92             self._insert_subtitles(amount, above)

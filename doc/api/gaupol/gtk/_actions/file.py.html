  29 class CloseProjectAction(UIMAction):
  30 
  31     """Close project."""
  32 
  33     action_item = (
  34         "close_project",
  35         gtk.STOCK_CLOSE,
  36         _("_Close"),
  37         "<control>W",
  38         _("Close the current project"),)
  39 
  40     paths = ["/ui/menubar/file/close"]
  41 
  42     @classmethod
  43     def is_doable(cls, application, page):
  44         """Return True if action can be done."""
  45 
  46         return (page is not None)

  49 class EditHeadersAction(UIMAction):
  50 
  51     """Edit file headers."""
  52 
  53     action_item = (
  54         "edit_headers",
  55         gtk.STOCK_PROPERTIES,
  56         _("_Headers"),
  57         "<alt>Return",
  58         _("Edit file headers"),)
  59 
  60     paths = ["/ui/menubar/file/headers"]
  61 
  62     @classmethod
  63     def is_doable(cls, application, page):
  64         """Return True if action can be done."""
  65 
  66         if page is None:
  67             return False
  68         main_file = page.project.main_file
  69         tran_file = page.project.tran_file
  70         if main_file is not None:
  71             if main_file.has_header:
  72                 return True
  73         if tran_file is not None:
  74             if tran_file.has_header:
  75                 return True
  76         return False

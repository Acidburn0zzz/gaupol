  19 """Editing subtitle data."""

  21 
  22 from gettext import ngettext
  23 
  24 from gaupol.base import Delegate
  25 from gaupol.gtk import util
  26 from gaupol.gtk.dialogs import HeaderDialog, InsertDialog, PreferencesDialog

  28 
  29 class EditAgent(Delegate):
  30 
  31     """Editing subtitle data.
  32 
  33     Instance variables:
  34 
  35         _pref_dialog: PreferencesDialog or None
  36     """
  37 
  38     # pylint: disable-msg=E0203,W0201
  39 
  40     def __init__(self, master):
  41 
  42         Delegate.__init__(self, master)
  43 
  44         self._pref_dialog = None
  45 
  46     def _on_pref_dialog_response(self, *args):
  47         """Destroy the preferences dialog."""
  48 
  49         self._pref_dialog.destroy()
  50         self._pref_dialog = None
  51 
  52     def _sync_clipboards(self, page):
  53         """Synchronize all clipboards to match that of page."""
  54 
  55         data = page.project.clipboard.data
  56         self.clipboard.data = data
  57         for item in self.pages:
  58             item.project.clipboard.data = data
  59         text = page.project.clipboard.get_data_as_string()
  60         self.x_clipboard.set_text(text)
  61         self.update_gui()
  62 
  63     def on_clear_texts_activate(self, *args):
  64         """Clear the selected texts."""
  65 
  66         page = self.get_current_page()
  67         rows = page.view.get_selected_rows()
  68         col = page.view.get_focus()[1]
  69         doc = page.text_column_to_document(col)
  70         page.project.clear_texts(rows, doc)
  71 
  72     def on_copy_texts_activate(self, *args):
  73         """Copy the selected texts to the clipboard."""
  74 
  75         page = self.get_current_page()
  76         rows = page.view.get_selected_rows()
  77         col = page.view.get_focus()[1]
  78         doc = page.text_column_to_document(col)
  79         page.project.copy_texts(rows, doc)
  80         self._sync_clipboards(page)
  81 
  82     def on_cut_texts_activate(self, *args):
  83         """Cut the selected texts to the clipboard."""
  84 
  85         page = self.get_current_page()
  86         rows = page.view.get_selected_rows()
  87         col = page.view.get_focus()[1]
  88         doc = page.text_column_to_document(col)
  89         page.project.cut_texts(rows, doc)
  90         self._sync_clipboards(page)
  91 
  92     def on_edit_headers_activate(self, *args):
  93         """Edit file headers."""
  94 
  95         dialog = HeaderDialog(self.window, self)
  96         self.flash_dialog(dialog)
  97         self.update_gui()
  98 
  99     def on_edit_preferences_activate(self, *args):
 100         """Configure Gaupol."""
 101 
 102         if self._pref_dialog is None:
 103             self._pref_dialog = PreferencesDialog()
 104             util.connect(self, "_pref_dialog", "response")
 105             self._pref_dialog.show()
 106         self._pref_dialog.present()
 107 
 108     def on_edit_value_activate(self, *args):
 109         """Edit the focused cell."""
 110 
 111         view = self.get_current_page().view
 112         row, column = view.get_cursor()
 113         view.set_cursor(row, column, True)
 114 
 115     def on_insert_subtitles_activate(self, *args):
 116         """Insert blank subtitles."""
 117 
 118         dialog = InsertDialog(self.window, self)
 119         self.flash_dialog(dialog)
 120 
 121     def on_invert_selection_activate(self, *args):
 122         """Invert the current selection."""
 123 
 124         page = self.get_current_page()
 125         rows = set(range(0, len(page.project.times)))
 126         rows.difference_update(set(page.view.get_selected_rows()))
 127         page.view.select_rows(rows)
 128 
 129     def on_merge_subtitles_activate(self, *args):
 130         """Merge the selected subtitles."""
 131 
 132         page = self.get_current_page()
 133         rows = page.view.get_selected_rows()
 134         page.project.merge_subtitles(rows)
 135 
 136     def on_paste_texts_activate(self, *args):
 137         """Paste texts from the clipboard."""
 138 
 139         page = self.get_current_page()
 140         rows = page.view.get_selected_rows()
 141         col = page.view.get_focus()[1]
 142         doc = page.text_column_to_document(col)
 143         length = len(page.project.times)
 144         rows = page.project.paste_texts(rows[0], doc)
 145         count = len(page.project.times) - length
 146         if count > 0:
 147             message = ngettext(
 148                 "Inserted %d subtitle to fit clipboard contents",
 149                 "Inserted %d subtitles to fit clipboard contents",
 150                 count ) % count
 151             self.push_message(message)
 152 
 153     def on_project_action_done(self, *args):
 154         """Update GUI after doing action."""
 155 
 156         self.update_gui()
 157         self.emit("page-changed", self.get_current_page())
 158 
 159     def on_project_action_redone(self, *args):
 160         """Update GUI after redoing action."""
 161 
 162         page = self.get_current_page()
 163         row = page.view.get_focus()[0]
 164         if row is not None:
 165             page.view.scroll_to_row(row)
 166         self.update_gui()
 167         self.emit("page-changed", page)
 168 
 169     def on_project_action_undone(self, *args):
 170         """Update GUI after undoing action."""
 171 
 172         page = self.get_current_page()
 173         row = page.view.get_focus()[0]
 174         if row is not None:
 175             page.view.scroll_to_row(row)
 176         self.update_gui()
 177         self.emit("page-changed", page)
 178

 238     def redo(self, count=1):
 239         """Redo actions."""
 240 
 241         util.set_cursor_busy(self.window)
 242         page = self.get_current_page()
 243         page.project.redo(count)
 244         util.set_cursor_normal(self.window)
 245 
 246     def undo(self, count=1):
 247         """Undo actions."""
 248 
 249         util.set_cursor_busy(self.window)
 250         page = self.get_current_page()
 251         page.project.undo(count)
 252         util.set_cursor_normal(self.window)

  19 """Message dialog classes."""
  20 
  21 
  22 import gtk
  23 
  24 
  25 class ErrorDialog(gtk.MessageDialog):
  26 
  27     """Base class for error dialogs."""
  28 
  29     def __init__(self, parent, title, message=None):
  30 
  31         gtk.MessageDialog.__init__(
  32             self, parent, gtk.DIALOG_MODAL | gtk.DIALOG_DESTROY_WITH_PARENT,
  33             gtk.MESSAGE_ERROR, gtk.BUTTONS_NONE, title)
  34 
  35         if message is not None:
  36             self.format_secondary_text(message)
  37

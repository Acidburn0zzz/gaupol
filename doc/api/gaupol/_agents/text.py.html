  32 class TextAgent(Delegate):
  33 
  34     """Text corrections."""

  36     # pylint: disable-msg=E0203,W0201
  37 
  38     @revertable
  39     def capitalize(self, rows, doc, pattern, register=-1):
  40         """Capitalize texts following match of pattern.
  41 
  42         rows can be None to process all rows.
  43         Raise re.error if pattern sucks.
  44         Return changed rows.
  45         """
  46         finder = Finder()
  47         finder.set_regex(pattern)
  48         re_alphanum = re.compile(r"\w", re.UNICODE)
  49 
  50         new_rows = []
  51         new_texts = []
  52         rows = rows or range(len(self.times))
  53         texts = self.get_texts(doc)
  54         for rows in util.get_ranges(rows):
  55             cap_next = False
  56             for row in rows:
  57                 text = texts[row]
  58                 if cap_next or row == 0:
  59                     match = re_alphanum.search(text)
  60                     if match is not None:
  61                         a = match.start()
  62                         text = text[:a] + text[a:].capitalize()
  63                     cap_next = False
  64                 finder.set_text(text)
  65                 while True:
  66                     try:
  67                         z = finder.next()[1]
  68                     except StopIteration:
  69                         break
  70                     match = re_alphanum.search(text[z:])
  71                     if match is not None:
  72                         z = z + match.start()
  73                         text = text[:z] + text[z:].capitalize()
  74                         continue
  75                     cap_next = True
  76                     break
  77                 if text != texts[row]:
  78                     new_rows.append(row)
  79                     new_texts.append(text)
  80 
  81         if new_rows:
  82             self.replace_texts(new_rows, doc, new_texts, register=register)
  83             self.set_action_description(register, _("Capitalizing texts"))
  84         return new_rows
  85 
  86     @revertable
  87     def format_lines(self, rows, doc, dialogue_pattern, clause_pattern,
  88         ok_dialogue, ok_clauses, max_length, length_func, legal_length=None,
  89         legal_lines=None, require_reduction=False, register=-1):
  90         """Split or merge lines.
  91 
  92         rows can be None to process all rows.
  93         ok_* are line counts that are not attempted to be reduced.
  94         legal_length is the maximum line length of skipped subtitles.
  95         legal_lines is the maximum line count of skipped subtitles.
  96         require_reduction should be True to skip unreducible line counts.
  97         Raise re.error if pattern sucks.
  98         Return changed rows.
  99         """
 100         liner = Liner(self.get_tag_regex(doc))
 101         liner.re_dialogue = re.compile(dialogue_pattern, re.UNICODE)
 102         liner.re_clause = re.compile(clause_pattern, re.UNICODE)
 103         liner.ok_dialogue = ok_dialogue
 104         liner.ok_clauses = ok_clauses
 105         liner.max_length = max_length
 106         liner.set_length_function(length_func)
 107 
 108         new_rows = []
 109         new_texts = []
 110         rows = rows or range(len(self.times))
 111         texts = self.get_texts(doc)
 112         for row in rows:
 113             text = texts[row]
 114             lines = text.split("\n")
 115             if legal_length is None and legal_lines is None:
 116                 liner.set_text(text)
 117                 text = liner.format()
 118             elif legal_length is not None:
 119                 if any(list(length_func(x) > legal_length for x in lines)):
 120                     liner.set_text(text)
 121                     text = liner.format()
 122                 elif legal_lines is not None:
 123                     if len(lines) > legal_lines:
 124                         liner.set_text(text)
 125                         text = liner.format()
 126                         if require_reduction:
 127                             if not len(text.split("\n")) < len(lines):
 128                                 continue
 129             elif legal_lines is not None:
 130                 if len(lines) > legal_lines:
 131                     liner.set_text(text)
 132                     text = liner.format()
 133                     if require_reduction:
 134                         if not len(text.split("\n")) < len(lines):
 135                             continue
 136             if text != texts[row]:
 137                 new_rows.append(row)
 138                 new_texts.append(text)
 139 
 140         if new_rows:
 141             self.replace_texts(new_rows, doc, new_texts, register=register)
 142             self.set_action_description(register, _("Formatting lines"))
 143         return new_rows

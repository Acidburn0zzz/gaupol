  19 """Base class for subtitle tag libraries."""
  20 
  21 
  22 class TagLibrary(object):
  23 
  24     """Base class for subtitle tag libraries.
  25 
  26     Class variables:
  27 
  28         tag:         Regular expression pattern, flags
  29         italic_tag:  Regular expression pattern, flags
  30         decode_tags: List of tuples of pattern, flags, replacement [,count]
  31         encode_tags: List of tuples of pattern, flags, replacement [,count]
  32 
  33     decode_tags is a list of regular expressions that convert tags to the
  34     internal format. encode_tags convert from the internal format to the
  35     class's format. The fourth item in the decode_tags and encode_tags tuple is
  36     an optional integer that tells how many times the substitution should be
  37     performed (default is one).
  38 
  39     pre- and post-, -decode and -encode functions can be used to perform
  40     arbitrary tasks in tag conversion. pre-methods are run before regular
  41     exressions and post-methods after.
  42     """
  43 
  44     tag = None
  45     italic_tag = None
  46 
  47     @classmethod
  48     def pre_decode(cls, text):
  49         """Convert tags."""
  50 
  51         return text
  52 
  53     decode_tags = []
  54 
  55     @classmethod
  56     def post_decode(cls, text):
  57         """Convert tags."""
  58 
  59         return text
  60 
  61     @classmethod
  62     def pre_encode(cls, text):
  63         """Convert tags."""
  64 
  65         return text
  66 
  67     encode_tags = [
  68         # Remove all tags.
  69         (r"<.*?>", 0,
  70             r"")]
  71 
  72     @classmethod
  73     def post_encode(cls, text):
  74         """Convert tags."""
  75 
  76         return text
  77 
  78     @classmethod
  79     def italicize(cls, text):
  80         """Italicize text."""
  81 
  82         return text
  83 
  84     @classmethod
  85     def remove_redundant(cls, text):
  86         """Remove redundant tags from text."""
  87 
  88         return text

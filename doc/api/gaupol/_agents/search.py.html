  19 """Searching for and replacing text."""

  21 
  22 from gettext import gettext as _
  23 
  24 from gaupol import cons
  25 from gaupol.base import Delegate

  29 
  30 class SearchAgent(Delegate):
  31 
  32     """Finding and replacing text.
  33 
  34     Instance variables:
  35 
  36         _docs:         List of the target documents
  37         _finder:       Finder
  38         _match_doc:    Document of the last match
  39         _match_passed: True if the position of last match has been passed
  40         _match_row:    The row of the last match
  41         _match_span:   The start and end positions of the last match
  42         _rows:         The target rows
  43         _wrap:         True to wrap search
  44     """
  45 
  46     # pylint: disable-msg=E0203,W0201
  47 
  48     def __init__(self, master):
  49 
  50         Delegate.__init__(self, master)
  51 
  52         self._docs         = [cons.DOCUMENT.MAIN, cons.DOCUMENT.TRAN]
  53         self._finder       = Finder()
  54         self._match_doc    = None
  55         self._match_passed = False
  56         self._match_row    = None
  57         self._match_span   = None
  58         self._rows         = None
  59         self._wrap         = True
  60 
  61     def _find(self, row, doc, pos, next):
  62         """Find pattern starting from position.
  63 
  64         pos can be None for beginning or end.
  65         next should be True to find next, False to find previous.
  66         Raise StopIteration if no match.
  67         Return tuple of row, document, match span.
  68         """
  69         find = (self._next_in_document if next else self._previous_in_document)
  70         self._match_row = row
  71         self._match_doc = doc
  72         self._match_passed = False
  73         rows = self._rows or range(len(self.times))
  74         while True:
  75             try:
  76                 # Return match in document after position.
  77                 return find(row, doc, pos)
  78             except ValueError:
  79                 pass
  80             # Proceed to next document or raise StopIteration.
  81             self._match_passed = True
  82             doc = self._get_document(doc, next)
  83             row = (min(rows) if next else max(rows))
  84             pos = None
  85 
  86     def _get_document(self, doc, next):
  87         """Get the document to proceed to.
  88 
  89         next should be True to find next, False to find previous.
  90         Raise StopIteration if nowhere to proceed.
  91         """
  92         if len(self._docs) == 1:
  93             if self._wrap:
  94                 return doc
  95             raise StopIteration
  96         if next and doc == cons.DOCUMENT.MAIN:
  97             return cons.DOCUMENT.TRAN
  98         if next and doc == cons.DOCUMENT.TRAN:
  99             if self._wrap:
 100                 return cons.DOCUMENT.MAIN
 101             raise StopIteration
 102         if not next and doc == cons.DOCUMENT.MAIN:
 103             if self._wrap:
 104                 return cons.DOCUMENT.TRAN
 105             raise StopIteration
 106         if not next and doc == cons.DOCUMENT.TRAN:
 107             return cons.DOCUMENT.MAIN
 108         raise ValueError
 109 
 110     def _next_in_document(self, row, doc, pos=None):
 111         """Find the next match in document starting from position.
 112 
 113         pos can be None to start from beginning.
 114         Raise StopIteration if no matches at all anywhere.
 115         Raise ValueError if no match in this document after position.
 116         Return tuple of row, document, match span.
 117         """
 118         rows = self._rows or range(len(self.times))
 119         texts = self.get_texts(doc)
 120         for row in range(row, max(rows) + 1):
 121             # Avoid resetting finder's match span.
 122             if texts[row] != self._finder.text:
 123                 self._finder.set_text(texts[row])
 124             self._finder.pos = 0
 125             if pos is not None:
 126                 self._finder.pos = pos
 127             try:
 128                 match_span = self._finder.next()
 129             except StopIteration:
 130                 # Raise StopIteration if a full loop around all target
 131                 # documents and rows has been made with no matches.
 132                 if doc == self._match_doc:
 133                     if row == self._match_row:
 134                         if self._match_passed:
 135                             raise StopIteration
 136                 pos = None
 137                 continue
 138             self._match_row = row
 139             self._match_doc = doc
 140             self._match_span = match_span
 141             self._match_passed = False
 142             return row, doc, match_span
 143         # Raise ValueError if no match found in this document after position.
 144         raise ValueError
 145 
 146     def _previous_in_document(self, row, doc, pos=None):
 147         """Find the previous match in document starting from position.
 148 
 149         pos can be None to start from end.
 150         Raise StopIteration if no matches at all anywhere.
 151         Raise ValueError if no match in this document before position.
 152         Return tuple of row, document, match span.
 153         """
 154         rows = self._rows or range(len(self.times))
 155         texts = self.get_texts(doc)
 156         for row in reversed(range(min(rows), row + 1)):
 157             # Avoid resetting finder's match span.
 158             if texts[row] != self._finder.text:
 159                 self._finder.set_text(texts[row])
 160             self._finder.pos = len(self._finder.text)
 161             if pos is not None:
 162                 self._finder.pos = pos
 163             try:
 164                 match_span = self._finder.previous()
 165             except StopIteration:
 166                 # Raise StopIteration if a full loop around all target
 167                 # documents and rows has been made with no matches.
 168                 if doc == self._match_doc:
 169                     if row == self._match_row:
 170                         if self._match_passed:
 171                             raise StopIteration
 172                 pos = None
 173                 continue
 174             self._match_row = row
 175             self._match_doc = doc
 176             self._match_span = match_span
 177             self._match_passed = False
 178             return row, doc, match_span
 179         # Raise ValueError if no match found in this document after position.
 180         raise ValueError
 181 
 182     def find_next(self, row, doc, pos=None):
 183         """Find the next match starting from position.
 184 
 185         pos can be None to start from beginning.
 186         Raise StopIteration if no matches exist.
 187         Return tuple of row, document, match span.
 188         """
 189         return self._find(row, doc, pos, True)
 190 
 191     def find_previous(self, row, doc, pos=None):
 192         """Find the previous match starting from position.
 193 
 194         pos can be None to start from end.
 195         Raise StopIteration if no matches exist.
 196         Return tuple of row, document, match span.
 197         """
 198         return self._find(row, doc, pos, False)
 199 
 200     @revertable
 201     def replace(self, register=-1):
 202         """Replace the current match."""
 203 
 204         row = self._match_row
 205         doc = self._match_doc
 206         self._finder.replace()
 207         self.set_text(row, doc, self._finder.text, register=register)
 208         self.set_action_description(register, _("Replacing"))
 209 
 210     @revertable
 211     def replace_all(self, register=-1):
 212         """Replace all matches of pattern.
 213 
 214         Return the amount of substitutions made.
 215         """
 216         new_rows = [[], []]
 217         new_texts = [[], []]
 218         count = 0
 219         for doc in self._docs:
 220             texts = self.get_texts(doc)
 221             for row, text in enumerate(texts):
 222                 self._finder.set_text(text)
 223                 sub_count = self._finder.replace_all()
 224                 if sub_count > 0:
 225                     new_rows[doc].append(row)
 226                     new_texts[doc].append(self._finder.text)
 227                     count += sub_count
 228         if count > 0:
 229             self.replace_both_texts(new_rows, new_texts, register=register)
 230             self.set_action_description(register, _("Replacing all"))
 231         return count
 232 
 233     def set_find_regex(self, pattern, flags=0):
 234         """Set the regular expression pattern to find.
 235 
 236         DOTALL, MULTILINE and UNICODE are automatically added to flags.
 237         Raise re.error if pattern sucks.
 238         """
 239         self._finder.set_regex(unicode(pattern), flags)
 240 
 241     def set_find_replacement(self, replacement):
 242         """Set the replacement."""
 243 
 244         self._finder.replacement = unicode(replacement)
 245 
 246     def set_find_string(self, pattern, ignore_case=False):
 247         """Set string pattern to find."""
 248 
 249         self._finder.pattern = unicode(pattern)
 250         self._finder.ignore_case = ignore_case
 251         self._finder.is_regex = False
 252 
 253     def set_find_target(self, rows=None, docs=None, wrap=True):
 254         """Set the target to find in.
 255 
 256         rows can be None to target all rows.
 257         docs can be None to target all documents.
 258         """
 259         self._rows = rows or []
 260         self._docs = docs or [cons.DOCUMENT.MAIN, cons.DOCUMENT.TRAN]
 261         self._wrap = wrap

  19 """Model for subtitle data."""
  20 
  21 
  22 import types
  23 
  24 from gaupol import cons
  25 from gaupol._agents import *
  26 from gaupol._agents import AGENTS
  27 from gaupol.base import Observable
  28 from gaupol.calculator import Calculator
  29 from gaupol.clipboard import Clipboard
  30 
  31 
  32 class Project(Observable):
  33 
  34     """Model for subtitle data.
  35 
  36     Instance variables:
  37 
  38         _delegations: Dictionary mapping method names to agents
  39         calc:         Position calculator
  40         clipboard:    Internal clipboard
  41         framerate:    FRAMERATE constant
  42         frames:       List of lists of show, hide, duration frames
  43         main_changed: Integer, status of main document
  44         main_file:    The main SubtitleFile
  45         main_texts:   List of the main document's texts
  46         redoables:    Stack of redoable Actions
  47         times:        List of lists of show, hide, duration times
  48         tran_active:  True if the translation document is active
  49         tran_changed: Integer, status of the translation document
  50         tran_file:    The translation SubtitleFile
  51         tran_texts:   List of the translation document's texts
  52         undoables:    Stack of undoable Actions
  53         undo_limit:   Maximum size of the undo/redo stacks or None for no limit
  54         video_path:   Path to the video file
  55 
  56     Signals:
  57 
  58         action-done (project, action)
  59         action-redone (project, action)
  60         action-undone (project, action)
  61         main-file-opened (project, main_file)
  62         main-file-saved (project, main_file)
  63         main-texts-changed (project, rows)
  64         positions-changed (project, rows)
  65         subtitles-changed (project, rows)
  66         subtitles-inserted (project, rows)
  67         subtitles-removed (project, rows)
  68         translation-file-opened (project, tran_file)
  69         translation-file-saved (project, tran_file)
  70         translation-texts-changed (project, rows)
  71 
  72     See gaupol._agents for project methods provided by agents.
  73     """
  74 
  75     _signals = [
  76         "action-done",
  77         "action-redone",
  78         "action-undone",
  79         "main-file-opened",
  80         "main-file-saved",
  81         "main-texts-changed",
  82         "positions-changed",
  83         "subtitles-inserted",
  84         "subtitles-removed",
  85         "subtitles-changed",
  86         "translation-file-opened",
  87         "translation-file-saved",
  88         "translation-texts-changed",]
  89 
  90     def __getattr__(self, name):
  91 
  92         return self._delegations[name].__getattribute__(name)
  93 
  94     def __init__(self, framerate=cons.FRAMERATE.FR_23_976, undo_limit=None):
  95 
  96         Observable.__init__(self)
  97 
  98         self._delegations = {}
  99         self.calc         = Calculator(framerate)
 100         self.clipboard    = Clipboard()
 101         self.framerate    = framerate
 102         self.frames       = []
 103         self.main_changed = 0
 104         self.main_file    = None
 105         self.main_texts   = []
 106         self.redoables    = []
 107         self.times        = []
 108         self.tran_active  = False
 109         self.tran_changed = 0
 110         self.tran_file    = None
 111         self.tran_texts   = []
 112         self.undo_limit   = undo_limit
 113         self.undoables    = []
 114         self.video_path   = None
 115 
 116         self._init_delegations()
 117 
 118     def _init_delegations(self):
 119         """Initialize the delegation mappings."""
 120 
 121         for agent in (eval(x)(self) for x in AGENTS):
 122             for attr_name in (x for x in dir(agent) if not x.startswith("_")):
 123                 attr = getattr(agent, attr_name)
 124                 if type(attr) is types.MethodType:
 125                     if attr_name in self._delegations:
 126                         raise ValueError
 127                     self._delegations[attr_name] = agent

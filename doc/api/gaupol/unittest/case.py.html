  19 """Base class for unit test cases."""
  20 
  21 
  22 from __future__ import with_statement
  23 import os
  24 import tempfile
  25 
  26 from .samples import MICRODVD_TEXT, SUBRIP_TEXT
  27 
  28 
  29 class TestCase(object):
  30 
  31     """Base class for unit test cases.
  32 
  33     Instance variables:
  34 
  35         files: List of the paths of temporary files created
  36     """
  37 
  38     # pylint: disable-msg=C0103
  39 
  40     def __init__(self):
  41 
  42         self.files = []
  43 
  44     def get_microdvd_path(self):
  45         """Get path to a temporary MicroDVD file."""
  46 
  47         # pylint: disable-msg=E0602
  48         fd, path = tempfile.mkstemp(prefix="gaupol.", suffix=".sub")
  49         with os.fdopen(fd, "w") as fobj:
  50             fobj.write(MICRODVD_TEXT)
  51         self.files.append(path)
  52         return path
  53 
  54     def get_project(self):
  55         """Get a new project."""
  56 
  57         from gaupol.project import Project
  58         project = Project()
  59         project.open_main(self.get_subrip_path(), "ascii")
  60         project.open_translation(self.get_microdvd_path(), "ascii")
  61         return project
  62 
  63     def get_subrip_path(self):
  64         """Get path to a temporary SubRip file."""
  65 
  66         # pylint: disable-msg=E0602
  67         fd, path = tempfile.mkstemp(prefix="gaupol.", suffix=".srt")
  68         with os.fdopen(fd, "w") as fobj:
  69             fobj.write(SUBRIP_TEXT)
  70         self.files.append(path)
  71         return path
  72 
  73     def is_regex(self, obj):
  74         """Return True if obj is a regular expression object."""
  75 
  76         is_regex = True
  77         for name in ("pattern", "flags", "match", "sub"):
  78             is_regex = is_regex and hasattr(obj, name)
  79         return is_regex
  80 
  81     def setUp(self):
  82         """Compatibility alias for 'setup_method'."""
  83 
  84         self.setup_method(None)
  85 
  86     def setup_method(self, method):
  87         """Set proper state for executing tests in method."""
  88 
  89         pass
  90 
  91     def tearDown(self):
  92         """Compatibility alias for 'teardown_method'."""
  93 
  94         self.teardown_method(None)
  95 
  96     def teardown_method(self, method):
  97         """Remove state set for executing tests in method."""
  98 
  99         for path in self.files:
 100             try:
 101                 os.remove(path)
 102             except OSError:
 103                 pass
 104         self.files = []

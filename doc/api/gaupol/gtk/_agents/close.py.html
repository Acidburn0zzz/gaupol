  19 """Closing projects and quitting Gaupol."""
  20 
  21 
  22 from gettext import gettext as _
  23 
  24 import gtk
  25 
  26 from gaupol import util
  27 from gaupol.base import Delegate
  28 from gaupol.gtk import conf, cons
  29 from gaupol.gtk.dialogs import MultiCloseDialog, WarningDialog
  30 from gaupol.gtk.errors import Default
  31 
  32 
  33 class CloseAgent(Delegate):
  34 
  35     """Closing projects and quitting Gaupol."""
  36 
  37     # pylint: disable-msg=E0203,W0201
  38 
  39     def _close_all_pages(self):
  40         """Close all pages after confirmation.
  41 
  42         Raise Default to abort.
  43         """
  44         count = 0
  45         for page in self.pages:
  46             if page.project.main_changed:
  47                 unsaved = page
  48                 count += 1
  49             if page.project.tran_active and page.project.tran_changed:
  50                 unsaved = page
  51                 count += 1
  52         if count == 1:
  53             self._confirm_page(unsaved)
  54         elif count > 1:
  55             self._confirm_multiple(self.pages)
  56         while self.pages:
  57             self.close(self.pages[-1], False)
  58 
  59     def _confirm_main(self, page):
  60         """Confirm closing main document.
  61 
  62         Raise Default to abort.
  63         """
  64         doc = cons.DOCUMENT.MAIN
  65         basename = page.get_main_basename()
  66         response = self._show_close_warning_dialog(doc, basename)
  67         if response == gtk.RESPONSE_YES:
  68             self.save_main(page)
  69         elif response != gtk.RESPONSE_NO:
  70             raise Default
  71 
  72     def _confirm_multiple(self, pages):
  73         """Confirm closing pages.
  74 
  75         Raise Default to abort.
  76         """
  77         dialog = MultiCloseDialog(self.application, pages)
  78         response = self.flash_dialog(dialog)
  79         if response not in (gtk.RESPONSE_YES, gtk.RESPONSE_NO):
  80             raise Default
  81 
  82     def _confirm_page(self, page):
  83         """Confirm closing page.
  84 
  85         Raise Default to abort.
  86         """
  87         confirm_main = page.project.main_changed
  88         confirm_tran = page.project.tran_changed and page.project.tran_active
  89         if confirm_main and confirm_tran:
  90             self._confirm_multiple([page])
  91         elif confirm_main:
  92             self._confirm_main(page)
  93         elif confirm_tran:
  94             self._confirm_translation(page)
  95 
  96     def _confirm_translation(self, page):
  97         """Confirm closing translation document.
  98 
  99         Raise Default to abort.
 100         """
 101         doc = cons.DOCUMENT.TRAN
 102         basename = page.get_translation_basename()
 103         response = self._show_close_warning_dialog(doc, basename)
 104         if response == gtk.RESPONSE_YES:
 105             self.save_translation(page)
 106         elif response != gtk.RESPONSE_NO:
 107             raise Default
 108 
 109     def _save_window_geometry(self):
 110         """Save the geometry of main and output windows."""
 111 
 112         if not conf.application_window.maximized:
 113             domain = conf.application_window
 114             domain.size = self.window.get_size()
 115             domain.position = self.window.get_position()
 116         if not conf.output_window.maximized:
 117             domain = conf.output_window
 118             domain.size = self.output_window.get_size()
 119             domain.position = self.output_window.get_position()
 120 
 121     def _show_close_warning_dialog(self, doc, basename):
 122         """Show a warning dialog when trying to close a project.
 123 
 124         Return response.
 125         """
 126         if doc == cons.DOCUMENT.MAIN:
 127             title = _('Save changes to main document "%s" before closing?') \
                % basename
 128         elif doc == cons.DOCUMENT.TRAN:
 129             title = _('Save changes to translation document "%s" before '
 130                 'closing?') % basename
 131         message = _("If you don't save, changes will be permanently lost.")
 132         dialog = WarningDialog(self.window, title, message)
 133         dialog.add_button(_("Close _Without Saving"), gtk.RESPONSE_NO)
 134         dialog.add_button(gtk.STOCK_CANCEL, gtk.RESPONSE_CANCEL)
 135         dialog.add_button(gtk.STOCK_SAVE, gtk.RESPONSE_YES)
 136         dialog.set_default_response(gtk.RESPONSE_YES)
 137         return self.flash_dialog(dialog)
 138 
 139     def close(self, page, confirm=True):
 140         """Close page (without confirmation)."""
 141 
 142         if confirm:
 143             self._confirm_page(page)
 144         index = self.pages.index(page)
 145         if self.notebook.get_current_page() == index:
 146             self.notebook.next_page()
 147         self.notebook.remove_page(index)
 148         self.pages.remove(page)
 149         self.emit("page-closed", page)
 150 
 151     @util.ignore_exceptions(Default)
 152     def on_close_all_projects_activate(self, *args):
 153         """Close all open projects."""
 154 
 155         self._close_all_pages()
 156 
 157     @util.ignore_exceptions(Default)
 158     def on_close_project_activate(self, *args):
 159         """Close project."""
 160 
 161         self.close(self.get_current_page())
 162 
 163     @util.ignore_exceptions(Default)
 164     def on_page_close_request(self, page, *args):
 165         """Close project."""
 166 
 167         self.close(page)
 168 
 169     def on_quit_activate(self, *args):
 170         """Quit Gaupol."""
 171 
 172         try:
 173             self._close_all_pages()
 174         except Default:
 175             return
 176         self._save_window_geometry()
 177         try:
 178             gtk.main_quit()
 179         except RuntimeError:
 180             raise SystemExit(1)
 181 
 182     def on_window_delete_event(self, *args):
 183         """Quit Gaupol."""
 184 
 185         self.on_quit_activate()
 186         return True

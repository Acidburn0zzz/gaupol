  19 """MPsub file."""
  20 
  21 
  22 import codecs
  23 import re
  24 
  25 from gaupol import cons
  26 from gaupol.calculator import Calculator
  27 from ._subfile import SubtitleFile
  28 
  29 
  30 class MPsub(SubtitleFile):
  31 
  32     """MPsub file.
  33 
  34     Class variables:
  35 
  36         mode: MODE constant
  37     """
  38 
  39     format = cons.FORMAT.MPSUB
  40     framerate  = None
  41     has_header = True
  42     identifier = r"^FORMAT=(TIME|[\d\.]+)\s*$", 0
  43     mode = cons.MODE.TIME
  44 
  45     def _clean_lines(self, all_lines, re_time_line):
  46         """Return lines without blank lines preceding time lines."""
  47 
  48         lines = ["\n"]
  49         for line in all_lines:
  50             if re_time_line.match(line) is not None:
  51                 if not lines[-1].strip():
  52                     lines[-1] = line
  53                     continue
  54             lines.append(line)
  55         return lines
  56 
  57     def _get_mpsub_frames(self, shows, hides):
  58         """Get MPsub style shows and hides as frames."""
  59 
  60         shows = shows[:]
  61         hides = hides[:]
  62         for i in reversed(range(1, len(shows))):
  63             hides[i] = str(hides[i] - shows[i])
  64             shows[i] = str(shows[i] - hides[i - 1])
  65         hides[0] = str(hides[0] - shows[0])
  66         return shows, hides
  67 
  68     def _get_mpsub_times(self, shows, hides):
  69         """Get MPsub style shows and hides as times."""
  70 
  71         calc = Calculator()
  72         shows = list(calc.time_to_seconds(x) for x in shows)
  73         hides = list(calc.time_to_seconds(x) for x in hides)
  74         for i in reversed(range(1, len(shows))):
  75             hides[i] -= shows[i]
  76             shows[i] -= hides[i - 1]
  77         hides[0] -= shows[0]
  78 
  79         deviation = 0
  80         for i in range(len(shows)):
  81             orig_show = shows[i]
  82             shows[i] = round(shows[i] - deviation, 2)
  83             deviation = deviation + shows[i] - orig_show
  84             orig_hide = hides[i]
  85             hides[i] = round(hides[i] - deviation, 2)
  86             deviation = deviation + hides[i] - orig_hide
  87 
  88         shows = list("%.2f" % x for x in shows)
  89         hides = list("%.2f" % x for x in hides)
  90         return shows, hides
  91 
  92     def _read_components(self, lines, re_time_line):
  93         """Read and return shows, hides and texts."""
  94 
  95         shows = []
  96         hides = []
  97         texts = []
  98         for line in lines:
  99             match = re_time_line.match(line)
 100             if match is not None:
 101                 show = float(match.group(1))
 102                 show = (show + hides[-1] if hides else show)
 103                 shows.append(show)
 104                 hides.append(shows[-1] + float(match.group(2)))
 105                 texts.append(u"")
 106             elif texts:
 107                 texts[-1] += line
 108 
 109         calc = Calculator()
 110         if self.mode == cons.MODE.TIME:
 111             shows = list(calc.seconds_to_time(x) for x in shows)
 112             hides = list(calc.seconds_to_time(x) for x in hides)
 113         elif self.mode == cons.MODE.FRAME:
 114             shows = list(int(round(x, 0)) for x in shows)
 115             hides = list(int(round(x, 0)) for x in hides)
 116         texts = list(x[:-1] for x in texts)
 117         return shows, hides, texts
 118 
 119     def _read_header(self, lines, re_time_line):
 120         """Read header and return leftover lines."""
 121 
 122         header = ""
 123         while re_time_line.match(lines[0]) is None:
 124             header += lines.pop(0)
 125         self.header = header[:-1]
 126         return lines
 127 
 128     def read(self):
 129         """Read file.
 130 
 131         Raise IOError if reading fails.
 132         Raise UnicodeError if decoding fails.
 133         Raise ValueError if bad FORMAT line.
 134         Return shows, hides, texts.
 135         """
 136         re_time_line = re.compile(r"^([\d\.]+) ([\d\.]+)\s*$")
 137         lines = self._read_lines()
 138         lines = self._clean_lines(lines, re_time_line)
 139         lines = self._read_header(lines, re_time_line)
 140         return self._read_components(lines, re_time_line)
 141 
 142     def set_header(self, header):
 143         """Parse and set header and mode.
 144 
 145         Raise ValueError if bad FORMAT line.
 146         """
 147         mode = None
 148         for line in header.split("\n"):
 149             if line.startswith("FORMAT="):
 150                 mode = line[7:].strip()
 151         if not mode in (cons.FRAMERATE.mpsub_names + ["TIME"]):
 152             raise ValueError
 153 
 154         self.header = header
 155         self.framerate = None
 156         self.mode = cons.MODE.TIME
 157         if mode in cons.FRAMERATE.mpsub_names:
 158             index = cons.FRAMERATE.mpsub_names.index(mode)
 159             self.framerate = cons.FRAMERATE.members[index]
 160             self.mode = cons.MODE.FRAME
 161 
 162     def write(self, shows, hides, texts):
 163         """Write file.
 164 
 165         Raise IOError if writing fails.
 166         Raise UnicodeError if encoding fails.
 167         """
 168         method = (self._get_mpsub_times, self._get_mpsub_frames)[self.mode]
 169         shows, hides = method(shows, hides)
 170         texts = list(x.replace("\n", self.newline.value) for x in texts)
 171 
 172         fobj = codecs.open(self.path, "w", self.encoding)
 173         try:
 174             fobj.write(self.header + self.newline.value * 2)
 175             for i in range(len(shows)):
 176                 fobj.write("%s %s%s%s%s%s" % (
 177                     shows[i], hides[i], self.newline.value,
 178                     texts[i], self.newline.value, self.newline.value))
 179         finally:
 180             fobj.close()

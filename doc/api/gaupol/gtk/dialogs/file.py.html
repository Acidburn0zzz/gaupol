  34 class _FileDialog(gtk.FileChooserDialog, Runner):
  35 
  36     """Base class for dialogs for selecting text files.
  37 
  38     Instance variables:
  39 
  40         _encodings: List of tuples: Python name, display name
  41     """
  42 
  43     # pylint: disable-msg=W0231
  44     def __init__(self, *args, **kwargs):
  45 
  46         gtk.FileChooserDialog.__init__(self, *args, **kwargs)
  47 
  48         self._encodings = []
  49 
  50         self._init_encodings()
  51         self._init_filters()
  52 
  53     def _init_encoding_data(self):
  54         """Initialize data in the encoding combo box."""
  55 
  56         # pylint: disable-msg=W0612
  57         store = self._encoding_combo.get_model()
  58         store.clear()
  59         for encoding, name in self._encodings:
  60             store.append([name])
  61         self._encoding_combo.set_row_separator_func(util.separate_combo)
  62 
  63     def _init_encodings(self, custom=None):
  64         """Initialize the encodings."""
  65 
  66         self._encodings = []
  67         locale = enclib.get_locale_python_name()
  68         if locale is not None:
  69             name = enclib.get_locale_long_name()
  70             self._encodings.append((locale, name))
  71 
  72         encodings = list(conf.encoding.visibles)
  73         if custom is not None:
  74             encodings.append(custom)
  75         for encoding in encodings:
  76             if encoding != locale and enclib.is_valid(encoding):
  77                 name = enclib.get_long_name(encoding)
  78                 self._encodings.append((encoding, name))
  79 
  80         def sort(x, y):
  81             return cmp(x[1], y[1])
  82         a = (0 if locale is None else 1)
  83         self._encodings[a:] = sorted(self._encodings[a:], sort)
  84         self._encodings = util.get_unique(self._encodings)
  85         if not self._encodings:
  86             name = enclib.get_long_name("utf_8")
  87             self._encodings.append(("utf_8", name))
  88 
  89         self._init_special_encodings()
  90         self._encodings.append((util.COMBO_SEP, util.COMBO_SEP))
  91         self._encodings.append(("other", _("Other...")))
  92 
  93     def _init_filters(self):
  94         """Initialize the file filters."""
  95 
  96         file_filter = gtk.FileFilter()
  97         file_filter.set_name(_("All files"))
  98         file_filter.add_pattern("*")
  99         self.add_filter(file_filter)
 100 
 101         file_filter = gtk.FileFilter()
 102         file_filter.set_name(_("Plain text files"))
 103         file_filter.add_mime_type("text/plain")
 104         self.add_filter(file_filter)
 105 
 106         for format in cons.FORMAT.members:
 107             pattern = "*" + format.extension
 108             fields = {'format': format.display_name, 'extension': pattern}
 109             name = _("%(format)s (%(extension)s)") % fields
 110             file_filter = gtk.FileFilter()
 111             file_filter.set_name(name)
 112             file_filter.add_pattern(pattern)
 113             self.add_filter(file_filter)
 114 
 115     def _init_special_encodings(self):
 116         """Initialize any possible special encodings."""
 117 
 118         pass
 119 
 120     def _on_encoding_combo_changed(self, *args):
 121         """Show the encoding selection dialog."""
 122 
 123         if self.get_encoding() != "other":
 124             return
 125         encoding = None
 126         dialog = AdvEncodingDialog(self)
 127         response = self.run_dialog(dialog)
 128         if response == gtk.RESPONSE_OK:
 129             encoding = dialog.get_encoding()
 130             conf.encoding.visibles = dialog.get_visible_encodings()
 131         dialog.destroy()
 132         self.set_encoding(encoding)
 133 
 134     def get_encoding(self):
 135         """Get the encoding or None."""
 136 
 137         index = self._encoding_combo.get_active()
 138         if index < 0:
 139             return None
 140         return self._encodings[index][0]
 141 
 142     def set_encoding(self, encoding):
 143         """Set the encoding."""
 144 
 145         for i, seq in enumerate(self._encodings):
 146             if seq[0] == encoding:
 147                 self._encoding_combo.set_active(i)
 148                 return
 149         if encoding is not None and enclib.is_valid(encoding):
 150             self._init_encodings(encoding)
 151             self._init_encoding_data()
 152             return self.set_encoding(encoding)
 153         self._encoding_combo.set_active(0)

 156 class OpenDialog(_FileDialog):
 157 
 158     """Dialog for selecting subtitle files to open.
 159 
 160     Instance variables:
 161 
 162         _encoding_combo: gtk.ComboBox
 163         _smart_check:    gtk.CheckButton
 164     """
 165 
 166     def __init__(self, doc, title, parent):
 167 
 168         _FileDialog.__init__(
 169             self, title, parent, gtk.FILE_CHOOSER_ACTION_OPEN,
 170             (gtk.STOCK_CANCEL, gtk.RESPONSE_CANCEL,
 171              gtk.STOCK_OPEN, gtk.RESPONSE_OK))
 172 
 173         self._encoding_combo = None
 174         self._smart_check    = None
 175 
 176         self._init_extra_widget(doc)
 177         self._init_encoding_data()
 178         self._init_data(doc)
 179         self._init_signal_handlers()
 180         self.set_default_response(gtk.RESPONSE_OK)
 181 
 182     def _init_data(self, doc):
 183         """Initialize default values for widgets."""
 184 
 185         self.set_select_multiple(doc == cons.DOCUMENT.MAIN)
 186         if os.path.isdir(conf.file.directory):
 187             self.set_current_folder(conf.file.directory)
 188         self.set_encoding(conf.file.encoding)
 189         self._smart_check.set_active(conf.file.smart_tran)
 190         if all(conf.open_dialog.size):
 191             self.set_default_size(*conf.open_dialog.size)
 192 
 193     def _init_extra_widget(self, doc):
 194         """Initialize the extra widget."""
 195 
 196         glade_xml = util.get_glade_xml("open-box", "extra_widget")
 197         self._encoding_combo = glade_xml.get_widget("encoding_combo")
 198         self._smart_check = glade_xml.get_widget("smart_check")
 199         self._smart_check.props.visible = (doc == cons.DOCUMENT.TRAN)
 200         self.set_extra_widget(glade_xml.get_widget("extra_widget"))
 201 
 202     def _init_signal_handlers(self):
 203         """Initialize signal handlers."""
 204 
 205         util.connect(self, self, "response")
 206         util.connect(self, "_encoding_combo", "changed")
 207 
 208     def _init_special_encodings(self):
 209         """Initialize encoding auto-detection entries."""
 210 
 211         if util.chardet_available():
 212             self._encodings.append((util.COMBO_SEP, util.COMBO_SEP))
 213             self._encodings.append(("auto", _("Auto-detected")))
 214 
 215     def _on_response(self, dialog, response):
 216         """Save settings."""
 217 
 218         conf.file.encoding = self.get_encoding()
 219         conf.file.directory = self.get_current_folder()
 220         conf.file.smart_tran = self._smart_check.get_active()
 221         conf.open_dialog.size = self.get_size()

 240 class SaveDialog(_FileDialog):
 241 
 242     """Dialog for selecting a subtitle file to save.
 243 
 244     Instance variables:
 245 
 246         _encoding_combo: gtk.ComboBox
 247         _format_combo:   gtk.ComboBox
 248         _newline_combo:  gtk.ComboBox
 249     """
 250 
 251     def __init__(self, title, parent):
 252 
 253         _FileDialog.__init__(
 254             self, title, parent, gtk.FILE_CHOOSER_ACTION_SAVE,
 255             (gtk.STOCK_CANCEL, gtk.RESPONSE_CANCEL,
 256              gtk.STOCK_SAVE, gtk.RESPONSE_OK))
 257 
 258         self._encoding_combo = None
 259         self._format_combo   = None
 260         self._newline_combo  = None
 261 
 262         self._init_extra_widget()
 263         self._init_format_data()
 264         self._init_encoding_data()
 265         self._init_newline_data()
 266         self._init_data()
 267         self._init_signal_handlers()
 268         self.set_default_response(gtk.RESPONSE_OK)
 269 
 270     def _init_data(self):
 271         """Initialize default values for widgets."""
 272 
 273         self.set_do_overwrite_confirmation(True)
 274         if os.path.isdir(conf.file.directory):
 275             self.set_current_folder(conf.file.directory)
 276         self.set_encoding(conf.file.encoding)
 277         self.set_format(conf.file.format)
 278         self.set_newline(conf.file.newline)
 279 
 280     def _init_extra_widget(self):
 281         """Initialize the extra widget."""
 282 
 283         glade_xml = util.get_glade_xml("save-box", "extra_widget")
 284         self._encoding_combo = glade_xml.get_widget("encoding_combo")
 285         self._format_combo = glade_xml.get_widget("format_combo")
 286         self._newline_combo = glade_xml.get_widget("newline_combo")
 287         self.set_extra_widget(glade_xml.get_widget("extra_widget"))
 288 
 289     def _init_format_data(self):
 290         """Initialize data in the format combo box."""
 291 
 292         store = self._format_combo.get_model()
 293         store.clear()
 294         for name in cons.FORMAT.display_names:
 295             store.append([name])
 296         self._format_combo.set_active(0)
 297 
 298     def _init_newline_data(self):
 299         """Initialize data in the newline combo box."""
 300 
 301         store = self._newline_combo.get_model()
 302         store.clear()
 303         for name in cons.NEWLINE.display_names:
 304             store.append([name])
 305         self._newline_combo.set_active(0)
 306 
 307     def _init_signal_handlers(self):
 308         """Initialize signal handlers."""
 309 
 310         util.connect(self, self, "response")
 311         util.connect(self, "_format_combo", "changed")
 312         util.connect(self, "_encoding_combo", "changed")
 313 
 314         def update(*args):
 315             self._format_combo.emit("changed")
 316         # pylint: disable-msg=E1101
 317         save_button = self.action_area.get_children()[0]
 318         save_button.connect("event", update)
 319 
 320     def _on_format_combo_changed(self, combo_box):
 321         """Change the extension of the current filename."""
 322 
 323         path = self.get_filename()
 324         if path is None:
 325             return
 326         self.unselect_filename(path)
 327         dirname = os.path.dirname(path)
 328         basename = os.path.basename(path)
 329         for extension in cons.FORMAT.extensions:
 330             if basename.endswith(extension):
 331                 basename = basename[:-len(extension)]
 332                 break
 333         format = cons.FORMAT.members[combo_box.get_active()]
 334         basename += format.extension
 335         path = os.path.join(dirname, basename)
 336         self.set_current_name(basename)
 337         self.set_filename(path)
 338 
 339     def _on_response(self, dialog, response):
 340         """Save settings."""
 341 
 342         conf.file.directory = self.get_current_folder()
 343         conf.file.encoding = self.get_encoding()
 344         conf.file.format = self.get_format()
 345         conf.file.newline = self.get_newline()
 346 
 347     def get_format(self):
 348         """Get the format."""
 349 
 350         index = self._format_combo.get_active()
 351         return cons.FORMAT.members[index]
 352 
 353     def get_newline(self):
 354         """Get the newline."""
 355 
 356         index = self._newline_combo.get_active()
 357         return cons.NEWLINE.members[index]
 358 
 359     def set_format(self, format):
 360         """Set the format."""
 361 
 362         self._format_combo.set_active(format)
 363 
 364     def set_name(self, path):
 365         """Set either filename or current name."""
 366 
 367         if os.path.isfile(path):
 368             return self.set_filename(path)
 369         return self.set_current_name(path)
 370 
 371     def set_newline(self, newline):
 372         """Set the newline."""
 373 
 374         self._newline_combo.set_active(newline)

 377 class VideoDialog(gtk.FileChooserDialog):
 378 
 379     """Dialog for selecting a video file."""
 380 
 381     def __init__(self, parent):
 382 
 383         gtk.FileChooserDialog.__init__(
 384             self, _("Select Video"), parent, gtk.FILE_CHOOSER_ACTION_OPEN,
 385             (gtk.STOCK_CANCEL, gtk.RESPONSE_CANCEL,
 386              _("_Select"), gtk.RESPONSE_OK))
 387 
 388         self._init_filters()
 389 
 390     def _init_filters(self):
 391         """Intialize the file filters."""
 392 
 393         file_filter = gtk.FileFilter()
 394         file_filter.add_pattern("*")
 395         file_filter.set_name(_("All files"))
 396         self.add_filter(file_filter)
 397 
 398         file_filter = gtk.FileFilter()
 399         file_filter.add_mime_type("video/*")
 400         file_filter.set_name(_("Video files"))
 401         self.add_filter(file_filter)
 402         self.set_filter(file_filter)

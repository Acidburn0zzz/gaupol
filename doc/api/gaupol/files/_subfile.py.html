  19 """Base class for subtitle files.
  20 
  21 Module variables:
  22 
  23     _GLOBAL_HEADER_DIR: Directory path to global header templates
  24     _LOCAL_HEADER_DIR:  Directory path to local header templates
  25 """
  26 
  27 
  28 import codecs
  29 import os
  30 
  31 from gaupol import cons, paths, util
  32 
  33 
  34 _GLOBAL_HEADER_DIR = os.path.join(paths.DATA_DIR, "headers")
  35 _LOCAL_HEADER_DIR = os.path.join(paths.PROFILE_DIR, "headers")
  36 
  37 
  38 class SubtitleFile(object):
  39 
  40     """Base class for subtitle files.
  41 
  42     Class variables:
  43 
  44         format:     FORMAT constant
  45         has_header: True if file has a header
  46         identifier: Regular expression pattern, flags
  47         mode:       MODE constant
  48 
  49     Instance variables:
  50 
  51         encoding: Character encoding
  52         header:   Header string
  53         newline:  NEWLINE constant
  54         path:     Filepath
  55     """
  56 
  57     format = None
  58     has_header = None
  59     identifier = None, None
  60     mode = None
  61 
  62     def __init__(self, path, encoding, newline=None):
  63 
  64         self.encoding = encoding
  65         self.header   = ""
  66         self.newline  = newline
  67         self.path     = path
  68 
  69         if self.has_header:
  70             self.header = self.get_template_header()
  71 
  72     def _read_lines(self):
  73         """Read file to a unicoded list of lines.
  74 
  75         All newlines are converted to '\\n'.
  76         All blank lines from the end are removed.
  77         Raise IOError if reading fails.
  78         Raise UnicodeError if decoding fails.
  79         Return a list of the lines.
  80         """
  81         fobj = codecs.open(self.path, "rU", self.encoding)
  82         try:
  83             lines = fobj.readlines()
  84             chars = fobj.newlines
  85         finally:
  86             fobj.close()
  87         while lines and lines[-1] == "\n":
  88             lines.pop(-1)
  89         if isinstance(chars, tuple):
  90             chars = chars[0]
  91         index = cons.NEWLINE.values.index(chars)
  92         self.newline = cons.NEWLINE.members[index]
  93         return lines
  94 
  95     def get_template_header(self):
  96         """Read and return the header from a template file."""
  97 
  98         @util.exceptional(IOError, util.handle_read_io, 0)
  99         @util.exceptional(UnicodeError, util.handle_read_unicode, 0, 1)
 100         def read(path, encoding):
 101             return util.read(path, encoding)
 102 
 103         # pylint: disable-msg=E1101
 104         basename = self.format.name.lower() + ".txt"
 105         path = os.path.join(_LOCAL_HEADER_DIR, basename)
 106         if os.path.isfile(path):
 107             return read(path, None)
 108         path = os.path.join(_GLOBAL_HEADER_DIR, basename)
 109         if os.path.isfile(path):
 110             return read(path, "ascii")
 111         return ""
 112 
 113     def read(self):
 114         """Read file.
 115 
 116         Raise IOError if reading fails.
 117         Raise UnicodeError if decoding fails.
 118         Return shows, hides, texts.
 119         """
 120         raise NotImplementedError
 121 
 122     def write(self, shows, hides, texts):
 123         """Write file.
 124 
 125         Raise IOError if writing fails.
 126         Raise UnicodeError if encoding fails.
 127         """
 128         raise NotImplementedError

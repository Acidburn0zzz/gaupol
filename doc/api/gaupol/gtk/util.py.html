  19 """Miscellaneous functions for GTK widgets.
  20 
  21 Module variables:
  22 
  23     COMBO_SEP:     String rendered as a separator in combo boxes
  24     EXTRA:         Extra width to add to size calculations
  25     BUSY_CURSOR:   gtk.gdk.Cursor
  26     HAND_CURSOR:   gtk.gdk.Cursor
  27     INSERT_CURSOR: gtk.gdk.Cursor
  28     NORMAL_CURSOR: gtk.gdk.Cursor
  29 
  30 When setting dialog sizes based on their content, we get the size request of
  31 the scrolled window component and add the surroundings to that. For this to
  32 work neatly we should add some extra to adapt to different widget sizes in
  33 different themes. Let the EXTRA constant very vaguely account for that.
  34 """

  59 def get_glade_xml(name, root=None, directory=None):
  60     """Get gtk.glade.XML object from Glade file path.
  61 
  62     name should be Glade XML file's basename without extension.
  63     directory can be None for the default global directory.
  64     Raise RuntimeError if unable to load Glade XML file.
  65     """
  66     if directory is None:
  67         directory = os.path.join(paths.DATA_DIR, "glade")
  68     path = os.path.join(directory, "%s.glade" % name)
  69     try:
  70         return gtk.glade.XML(path, root)
  71     except RuntimeError:
  72         print "Failed to load Glade XML file '%s'." % path
  73         raise

  75 def get_event_box(widget):
  76     """Get an event box if it is a parent of widget."""
  77 
  78     return get_parent(widget, gtk.EventBox)
  79 
  80 def get_parent(child, parent_type):
  81     """Get the first parent of widget that is of given type."""
  82 
  83     parent = child.get_parent()
  84     while not isinstance(parent, parent_type):
  85         parent = parent.get_parent()
  86     return parent

  88 def get_text_view_size(text_view):
  89     """Get the size desired by text view.
  90 
  91     Return width, height.
  92     """
  93     text_buffer = text_view.get_buffer()
  94     start, end = text_buffer.get_bounds()
  95     text = text_buffer.get_text(start, end)
  96     label = gtk.Label(text)
  97     return label.size_request()
  98 
  99 def get_tree_view_size(tree_view):
 100     """Get the size desired by tree view.
 101 
 102     Return width, height.
 103     """
 104     scroller = tree_view.get_parent()
 105     policy = scroller.get_policy()
 106     scroller.set_policy(gtk.POLICY_NEVER, gtk.POLICY_NEVER)
 107     width, height = scroller.size_request()
 108     scroller.set_policy(*policy)
 109     return width, height

 111 def idle_method(function):
 112     """Decorator for functions to be run in main loop while idle.
 113 
 114     Threads doing GUI operations should use this decorator so that all GUI
 115     operations get done in the main loop.
 116     """
 117     def do_idle(args, kwargs):
 118         function(*args, **kwargs)
 119         return False
 120 
 121     @functools.wraps(function)
 122     def wrapper(*args, **kwargs):
 123         gobject.idle_add(do_idle, args, kwargs)
 124 
 125     return wrapper

 127 def prepare_text_view(text_view):
 128     """Connect text view to font and length margin updates."""
 129 
 130     # pylint: disable-msg=E0102
 131     def update(*args):
 132         if conf.editor.show_lengths_edit:
 133             return lengthlib.connect_text_view(text_view)
 134         return lengthlib.disconnect_text_view(text_view)
 135     conf.editor.connect("notify::show_lengths_edit", update)
 136     update()
 137 
 138     def update(*args):
 139         font = ("" if conf.editor.use_default_font else conf.editor.font)
 140         set_widget_font(text_view, font)
 141     conf.editor.connect("notify::use_default_font", update)
 142     conf.editor.connect("notify::font", update)
 143     update()

 145 def resize_dialog(dialog, width, height, max_width=0.6, max_height=0.6):
 146     """Resize dialog in a smart manner.
 147 
 148     width and height should be desired sizes in pixels.
 149     max_width and max_height should be between 0 and 1.
 150     """
 151     # Set maximum based on percentage of screen space.
 152     width = min(width, int(max_width * gtk.gdk.screen_width()))
 153     height = min(height, int(max_height * gtk.gdk.screen_height()))
 154 
 155     # Set minimum based on dialog content.
 156     size = dialog.size_request()
 157     width = max(size[0], width)
 158     height = max(size[1], height)
 159 
 160     dialog.set_default_size(width, height)
 161 
 162 def resize_message_dialog(
 163     dialog, width, height, max_width=0.5, max_height=0.5):
 164     """Resize message dialog in a smart manner.
 165 
 166     width and height should be desired sizes in pixels.
 167     max_width and max_height should be between 0 and 1.
 168     """
 169     resize_dialog(dialog, width, height, max_width, max_height)

 171 def separate_combo(store, itr):
 172     """Separator function for combo box models."""
 173 
 174     return store.get_value(itr, 0) == COMBO_SEP

 176 def set_button(button, text, stock=None):
 177     """Set the label and the image on button."""
 178 
 179     if stock is not None:
 180         image = gtk.Button(stock=stock).get_image()
 181         button.set_image(image)
 182     alignment = button.get_children()[0]
 183     hbox = alignment.get_children()[0]
 184     image = hbox.get_children()[0]
 185     if stock is None:
 186         hbox.remove(image)
 187     button.set_label(text)
 188     button.set_use_underline(True)

 190 def set_cursor_busy(window):
 191     """Set cursor busy when above window."""
 192 
 193     while gtk.events_pending():
 194         gtk.main_iteration()
 195     window.window.set_cursor(BUSY_CURSOR)
 196     while gtk.events_pending():
 197         gtk.main_iteration()
 198 
 199 def set_cursor_normal(window):
 200     """Set cursor normal when above window."""
 201 
 202     while gtk.events_pending():
 203         gtk.main_iteration()
 204     window.window.set_cursor(NORMAL_CURSOR)
 205     while gtk.events_pending():
 206         gtk.main_iteration()

 208 def set_label_font(label, font):
 209     """Set label font."""
 210 
 211     context = label.get_pango_context()
 212     font_desc = context.get_font_description()
 213     custom_font_desc = pango.FontDescription(font)
 214     font_desc.merge(custom_font_desc, True)
 215 
 216     attr = pango.AttrFontDesc(font_desc, 0, -1)
 217     attr_list = pango.AttrList()
 218     attr_list.insert(attr)
 219     label.set_attributes(attr_list)
 220 
 221 def set_widget_font(widget, font):
 222     """Set widget font."""
 223 
 224     context = widget.get_pango_context()
 225     font_desc = context.get_font_description()
 226     custom_font_desc = pango.FontDescription(font)
 227     font_desc.merge(custom_font_desc, True)
 228     widget.modify_font(font_desc)

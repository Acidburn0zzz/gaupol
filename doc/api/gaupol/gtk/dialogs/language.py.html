  19 """Dialog for configuring spell-check."""
  20 
  21 
  22 import gobject
  23 import gtk
  24 
  25 try:
  26     import enchant
  27 except Exception:
  28     pass
  29 
  30 from gaupol import langlib
  31 from gaupol.gtk import conf, cons, util
  32 from gaupol.gtk.index import *
  33 from .glade import GladeDialog
  34 
  35 
  36 class LanguageDialog(GladeDialog):
  37 
  38     """Dialog for configuring spell-check.
  39 
  40     Instance variables:
  41 
  42         _all_radio:     gtk.RadioButton, target all documents
  43         _current_radio: gtk.RadioButton, target current document
  44         _langs:         List of languages, er... locales
  45         _main_radio:    gtk.RadioButton, target main text column
  46         _tran_radio:    gtk.RadioButton, target translation text column
  47         _tree_view:     gtk.TreeView, languages
  48     """
  49 
  50     def __init__(self, parent):
  51 
  52         GladeDialog.__init__(self, "language-dialog")
  53 
  54         self._langs = []
  55 
  56         self._all_radio     = self._glade_xml.get_widget("all_radio")
  57         self._current_radio = self._glade_xml.get_widget("current_radio")
  58         self._main_radio    = self._glade_xml.get_widget("main_radio")
  59         self._tran_radio    = self._glade_xml.get_widget("tran_radio")
  60         self._tree_view     = self._glade_xml.get_widget("tree_view")
  61 
  62         self._init_langs()
  63         self._init_tree_view()
  64         self._init_data()
  65         self._init_signal_handlers()
  66         self._init_sizes()
  67         self._dialog.set_transient_for(parent)
  68         self._dialog.set_default_response(gtk.RESPONSE_CLOSE)
  69 
  70     def _init_data(self):
  71         """Initialize default values for widgets."""
  72 
  73         if conf.spell_check.lang in self._langs:
  74             row = self._langs.index(conf.spell_check.lang)
  75             self._tree_view.get_selection().select_path(row)
  76 
  77         col = conf.spell_check.col
  78         self._main_radio.set_active(col == MTXT)
  79         self._tran_radio.set_active(col == TTXT)
  80 
  81         target = conf.spell_check.target
  82         self._all_radio.set_active(target == cons.TARGET.ALL)
  83         self._current_radio.set_active(target == cons.TARGET.CURRENT)
  84 
  85     def _init_langs(self):
  86         """Initialize the list of available languages."""
  87 
  88         for locale in langlib.LOCALES:
  89             try:
  90                 enchant.Dict(locale)
  91                 self._langs.append(locale)
  92             except enchant.Error:
  93                 pass
  94         self._langs.sort()
  95 
  96     def _init_signal_handlers(self):
  97         """Initialize signal handlers."""
  98 
  99         def on_target_toggled(*args):
 100             self._save_target()
 101         self._all_radio.connect("toggled", on_target_toggled)
 102         self._current_radio.connect("toggled", on_target_toggled)
 103 
 104         def on_column_toggled(*args):
 105             self._save_column()
 106         self._main_radio.connect("toggled", on_column_toggled)
 107         self._tran_radio.connect("toggled", on_column_toggled)
 108 
 109         selection = self._tree_view.get_selection()
 110         selection.connect("changed", self._on_tree_view_selection_changed)
 111 
 112     def _init_sizes(self):
 113         """Initialize widget sizes."""
 114 
 115         width, height = util.get_tree_view_size(self._tree_view)
 116         width = width + 42 + util.EXTRA
 117         height = height + 259 + util.EXTRA
 118         util.resize_dialog(self, width, height, 0.5, 0.5)
 119 
 120     def _init_tree_view(self):
 121         """Initialize the tree view."""
 122 
 123         self._tree_view.columns_autosize()
 124         selection = self._tree_view.get_selection()
 125         selection.set_mode(gtk.SELECTION_SINGLE)
 126         store = gtk.ListStore(gobject.TYPE_STRING)
 127         self._tree_view.set_model(store)
 128         column = gtk.TreeViewColumn("", gtk.CellRendererText(), text=0)
 129         self._tree_view.append_column(column)
 130         for lang in self._langs:
 131             store.append([langlib.get_long_name(lang)])
 132 
 133     def _on_tree_view_selection_changed(self, selection):
 134         """Save the language."""
 135 
 136         store, itr = selection.get_selected()
 137         if itr is not None:
 138             row = store.get_path(itr)[0]
 139             conf.spell_check.lang = self._langs[row]
 140 
 141     def _save_column(self):
 142         """Save the column."""
 143 
 144         main = self._main_radio.get_active()
 145         col = (MTXT if main else TTXT)
 146         conf.spell_check.col = col
 147 
 148     def _save_target(self):
 149         """Save the target."""
 150 
 151         each = self._all_radio.get_active()
 152         target = (cons.TARGET.ALL if each else cons.TARGET.CURRENT)
 153         conf.spell_check.target = target

  19 """String and regular expression finder and replacer."""
  20 
  21 
  22 import re
  23 
  24 
  25 class Finder(object):
  26 
  27     """String and regular expression finder and replacer.
  28 
  29     Instance variables:
  30 
  31         ignore_case: True to ignore case
  32         match:       Regular expression match object
  33         match_span:  Tuple of start and end position
  34         pattern:     String or regular expression object to find
  35         pos:         Current offset in text
  36         replacement: Replacement string
  37         text:        Target text
  38     """
  39 
  40     def __init__(self):
  41 
  42         self.ignore_case = False
  43         self.match       = None
  44         self.match_span  = None
  45         self.pattern     = None
  46         self.pos         = None
  47         self.replacement = None
  48         self.text        = None
  49 
  50     def next(self):
  51         """Find the next match of pattern.
  52 
  53         Raise StopIteration if no next match found.
  54         Return tuple of match start, end position.
  55         """
  56         if isinstance(self.pattern, basestring):
  57             text = self.text
  58             pattern = self.pattern
  59             if self.ignore_case:
  60                 text = text.lower()
  61                 pattern = pattern.lower()
  62             try:
  63                 index = text.index(pattern, self.pos)
  64             except ValueError:
  65                 raise StopIteration
  66             self.match_span = (index, index + len(pattern))
  67         else:
  68             match = self.pattern.search(self.text, self.pos)
  69             if match is None:
  70                 raise StopIteration
  71             # Avoid getting stuck with zero-length regular expressions.
  72             if match.span() == self.match_span == (self.pos, self.pos):
  73                 if self.pos == len(self.text):
  74                     raise StopIteration
  75                 self.pos += 1
  76                 return self.next()
  77             self.match = match
  78             self.match_span = match.span()
  79 
  80         self.pos = self.match_span[1]
  81         return self.match_span
  82 
  83     def previous(self):
  84         """Find the previous match of pattern.
  85 
  86         Raise StopIteration if no previous match found.
  87         Return tuple of match start, end position.
  88         """
  89         if isinstance(self.pattern, basestring):
  90             text = self.text
  91             pattern = self.pattern
  92             if self.ignore_case:
  93                 text = text.lower()
  94                 pattern = pattern.lower()
  95             try:
  96                 index = text.rindex(pattern, 0, self.pos)
  97             except ValueError:
  98                 raise StopIteration
  99             self.match_span = (index, index + len(pattern))
 100         else:
 101             iterator = self.pattern.finditer(self.text)
 102             match = None
 103             while True:
 104                 try:
 105                     candidate = iterator.next()
 106                     if candidate.end() > self.pos:
 107                         raise StopIteration
 108                 except StopIteration:
 109                     break
 110                 match = candidate
 111             if match is None:
 112                 raise StopIteration
 113             # Avoid getting stuck with zero-length regular expressions.
 114             if match.span() == self.match_span == (self.pos, self.pos):
 115                 if self.pos == 0:
 116                     raise StopIteration
 117                 self.pos -= 1
 118                 return self.previous()
 119             self.match = match
 120             self.match_span = match.span()
 121 
 122         self.pos = self.match_span[0]
 123         return self.match_span
 124 
 125     def replace(self, next=True):
 126         """Replace the current match.
 127 
 128         next should be True to finish at end of match, False for beginning.
 129         """
 130         a, z = self.match_span
 131         orig_length = len(self.text)
 132         replacement = self.replacement
 133         if not isinstance(self.pattern, basestring):
 134             replacement = self.match.expand(self.replacement)
 135         self.text = self.text[:a] + replacement + self.text[z:]
 136 
 137         shift = len(self.text) - orig_length
 138         self.pos = ((z + shift) if next else a)
 139 
 140         # Adapt match span to new to text length to avoid getting stuck with
 141         # zero-length regular expressions.
 142         if next and self.match_span[0] == self.match_span[1]:
 143             self.match_span = (self.pos, self.pos)
 144 
 145     def replace_all(self):
 146         """Replace all occurences of pattern.
 147 
 148         Return the amount of substitutions made.
 149         """
 150         self.pos = 0
 151         self.match = None
 152         self.match_span = None
 153         count = 0
 154         while True:
 155             try:
 156                 self.next()
 157             except StopIteration:
 158                 self.pos = len(self.text)
 159                 self.match_span = None
 160                 return count
 161             self.replace()
 162             count += 1
 163         return count
 164 
 165     def set_regex(self, pattern, flags=0):
 166         """Set and use regular expression.
 167 
 168         DOTALL, MULTILINE and UNICODE are automatically added to flags.
 169         IGNORECASE is automatically added to flags if self.ignore_case is True.
 170         Raise re.error if pattern sucks.
 171         """
 172         if self.ignore_case:
 173             flags = flags | re.IGNORECASE
 174         flags = flags | re.DOTALL | re.MULTILINE | re.UNICODE
 175         self.pattern = re.compile(pattern, flags)
 176 
 177     def set_text(self, text, next=True):
 178         """Set the text.
 179 
 180         next should be True to start at beginning, False for end.
 181         """
 182         self.text = text
 183         self.match = None
 184         self.match_span = None
 185         self.pos = (0 if next else len(text))

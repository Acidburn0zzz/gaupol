  29 class ActivateNextProjectAction(UIMAction):
  30 
  31     """Activate the project in the next tab."""
  32 
  33     action_item = (
  34         "activate_next_project",
  35         None,
  36         _("_Next"),
  37         "<control>Page_Down",
  38         _("Activate the project in the next tab"),)
  39 
  40     paths = ["/ui/menubar/projects/next"]
  41 
  42     @classmethod
  43     def is_doable(cls, application, page):
  44         """Return True if action can be done."""
  45 
  46         if page is not None:
  47             index = application.pages.index(page) + 1
  48             return (index in range(len(application.pages)))
  49         return False

  19 """Decorators for testing and timing methods and functions."""
  20 
  21 
  22 import copy
  23 import functools
  24 import time
  25 
  26 
  27 def benchmark(function):
  28     """Decorator for benchmarking functions and methods."""
  29 
  30     @functools.wraps(function)
  31     def wrapper(*args, **kwargs):
  32         start = time.time()
  33         value = function(*args, **kwargs)
  34         end = time.time()
  35         print "%.3f %s" % (end - start, function.__name__)
  36         return value
  37 
  38     return wrapper
  39 
  40 def reversion_test(function):
  41     """Decorator for testing reversions of one action."""
  42 
  43     @functools.wraps(function)
  44     def wrapper(*args, **kwargs):
  45         project = args[0].project
  46         values = (
  47             project.times,
  48             project.frames,
  49             project.main_texts,
  50             project.tran_texts,)
  51         orig_values = copy.deepcopy(values)
  52         value = function(*args, **kwargs)
  53         new_values = copy.deepcopy(values)
  54         assert new_values != orig_values
  55         for i in range(2):
  56             project.undo()
  57             assert values == orig_values
  58             project.redo()
  59             assert values == new_values
  60         return value
  61 
  62     return wrapper

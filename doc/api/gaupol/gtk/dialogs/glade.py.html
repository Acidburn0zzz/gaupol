  19 """Wrapper for Glade constructed dialogs."""
  20 
  21 
  22 from gaupol.base import Delegate
  23 from gaupol.gtk import util
  24 from gaupol.gtk.runner import Runner
  25 
  26 
  27 class GladeDialog(Delegate, Runner):
  28 
  29     """Wrapper for Glade constructed dialogs.
  30 
  31     Instance variables:
  32 
  33         _dialog:    gtk.Dialog
  34         _glade_xml: gtk.glade.XML
  35     """
  36 
  37     # pylint: disable-msg=W0231
  38     def __init__(self, name):
  39 
  40         glade_xml = util.get_glade_xml(name)
  41         dialog = glade_xml.get_widget("dialog")
  42         Delegate.__init__(self, dialog)
  43 
  44         self._dialog    = dialog
  45         self._glade_xml = glade_xml
  46 
  47     def run(self):
  48         """Show the dialog, run it and return response."""
  49 
  50         self._dialog.show()
  51         return self._dialog.run()

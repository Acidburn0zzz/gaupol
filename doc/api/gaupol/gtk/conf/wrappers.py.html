  33 class Config(configobj.ConfigObj):
  34 
  35     """Reading, writing and storing configurations.
  36 
  37     This is a convenience wrapper around ConfigObj. Methods defined in this
  38     class are as error-tolerant as possible. Validation includes extensions to
  39     handle Gaupol's constants, which can be defined in the spec file as
  40     CONSTANT and CONSTANT_list for each constant.
  41     """
  42 
  43     def __init__(self, config_file, spec_file):
  44         """Initialize a Config object.
  45 
  46         Raise ConfigParseError if parsing config_file fails.
  47         """
  48         try:
  49             configobj.ConfigObj.__init__(
  50                 self, config_file, configspec=spec_file,
  51                 encoding=util.get_default_encoding(),
  52                 write_empty_values=True)
  53         except IOError:
  54             util.handle_read_io(sys.exc_info(), config_file)
  55             # pylint: disable-msg=W0233
  56             configobj.ConfigObj.__init__(
  57                 self, None, configspec=spec_file,
  58                 encoding=util.get_default_encoding())
  59         except configobj.ConfigObjError, obj:
  60             print "Errors parsing configuration file '%s':" % config_file
  61             for error in obj.errors:
  62                 print "  Line %d: %s" % (error.line_number, error.message)
  63             raise ConfigParseError(obj)
  64 
  65         validator = self.__init_validator()
  66         spec = self.__init_spec(validator, spec_file)
  67         self.__remove_crap(spec)
  68         self.__validate(validator, spec)
  69 
  70     def __init_spec(self, validator, spec_file):
  71         """Initialize and return spec ConfigObj."""
  72 
  73         spec = configobj.ConfigObj(None, configspec=spec_file)
  74         result = spec.validate(validator)
  75         if result is True:
  76             return spec
  77         print "The spec is erroneous!"
  78         for error in configobj.flatten_errors(spec, result):
  79             sections, option, result = error
  80             for section in sections:
  81                 print "  %s.%s: %s" % (section, option, result)
  82         raise SystemExit
  83 
  84     def __init_validator(self):
  85         """Initialize and return validator."""
  86 
  87         validator = validate.Validator()
  88 
  89         def check_constant(name, value):
  90             index = validator.check("integer", value)
  91             members = getattr(cons, name).members
  92             try:
  93                 return members[index]
  94             except IndexError:
  95                 raise validate.VdtValueError(value)
  96 
  97         def check_constant_list(name, value):
  98             lst = validator.check("int_list", value)
  99             members = getattr(cons, name).members
 100             try:
 101                 return list(members[x] for x in lst)
 102             except IndexError:
 103                 raise validate.VdtValueError(value)
 104 
 105         for name in (x for x in dir(cons) if x.isupper()):
 106             func = functools.partial(check_constant, name)
 107             validator.functions[name] = func
 108             func = functools.partial(check_constant_list, name)
 109             validator.functions["%s_list" % name] = func
 110 
 111         return validator
 112 
 113     def __remove_crap(self, spec):
 114         """Remove sections and options not in spec."""
 115 
 116         sections = set(self.keys())
 117         sections -= set(spec.keys())
 118         if sections:
 119             print "Discarding unrecognized configuration sections:"
 120         for section in sections:
 121             print "  %s" % section
 122             del self[section]
 123 
 124         first = True
 125         for section in self.iterkeys():
 126             options = set(self[section].keys())
 127             options -= set(spec[section].keys())
 128             if options and first:
 129                 print "Discarding unrecognized configuration options:"
 130                 first = False
 131             for option in options:
 132                 print "  %s.%s" % (section, option)
 133                 del self[section][option]
 134 
 135     def __validate(self, validator, spec):
 136         """Validate options according to spec."""
 137 
 138         result = self.validate(validator, preserve_errors=True)
 139         if result is True:
 140             return
 141         print "Discarding erroneous configuration options:"
 142         for error in configobj.flatten_errors(self, result):
 143             sections, option, result = error
 144             for section in sections:
 145                 print "  %s.%s: %s" % (section, option, result)
 146                 self[section][option] = spec[section][option]
 147                 self[section].defaults.append(option)
 148 
 149     def write_to_file(self):
 150         """Write configurations to file."""
 151 
 152         try:
 153             util.makedirs(os.path.dirname(self.filename))
 154             configobj.ConfigObj.write(self)
 155         except (IOError, OSError):
 156             util.handle_write_io(sys.exc_info(), self.filename)

 159 class Container(Observable):
 160 
 161     """Configuration data container.
 162 
 163     Instance variables:
 164 
 165         root: A dictionary in a ConfigObj
 166 
 167     This class can be a configuration section or a container for the entire
 168     configuration data if there are no sections. This class is instantiated
 169     with a 'root', which is a lowest (sub)dictionary of a ConfigObj instance.
 170     This class provides convenient attribute access and notifications for
 171     configuration variables as per the Observable API.
 172     """
 173 
 174     __slots__ = ["root"]
 175 
 176     def __getattr__(self, name):
 177 
 178         return self.root[name]
 179 
 180     def __init__(self, root):
 181 
 182         Observable.__init__(self)
 183 
 184         self.root = root
 185         for name in root.keys():
 186             self._add_signal("notify::%s" % name)
 187 
 188     def __setattr__(self, name, value):
 189 
 190         if name in self.__slots__ or name in Observable.__slots__:
 191             return Observable.__setattr__(self, name, value)
 192         if value != self.root[name]:
 193             self.root[name] = value
 194             self.emit("notify::" + name)

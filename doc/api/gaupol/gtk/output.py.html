  19 """Output window."""
  20 
  21 
  22 from gettext import gettext as _
  23 
  24 import gtk
  25 
  26 from gaupol.gtk import conf, util
  27 
  28 
  29 class OutputWindow(gtk.Window):
  30 
  31     """Output window.
  32 
  33     Instance variables:
  34 
  35         _close_button: gtk.Button
  36         _text_view:    gtk.TextView
  37     """
  38 
  39     def __init__(self):
  40 
  41         gtk.Window.__init__(self)
  42 
  43         self._close_button = None
  44         self._text_view    = None
  45 
  46         self.set_border_width(12)
  47         self.set_title(_("Output"))
  48         self._init_widgets()
  49         self._init_sizes()
  50         self._init_signal_handlers()
  51         self._init_keys()
  52 
  53     def _init_keys(self):
  54         """Initialize keyboard shortcuts."""
  55 
  56         # Add Ctrl+W as an accelerator to close the window.
  57         accel_group = gtk.AccelGroup()
  58         method = self._on_close_key_pressed
  59         accel_group.connect_group(
  60             119, gtk.gdk.CONTROL_MASK, gtk.ACCEL_MASK, method)
  61         self.add_accel_group(accel_group)
  62 
  63     def _init_signal_handlers(self):
  64         """Initialize signal handlers."""
  65 
  66         util.connect(self, "_close_button", "clicked")
  67         util.connect(self, self, "delete-event")
  68         util.connect(self, self, "notify::visible")
  69         util.connect(self, self, "window-state-event")
  70 
  71     def _init_sizes(self):
  72         """Initialize widget sizes."""
  73 
  74         self.resize(*conf.output_window.size)
  75         self.move(*conf.output_window.position)
  76         if conf.output_window.maximized:
  77             self.maximize()
  78 
  79     def _init_widgets(self):
  80         """Initialize widgets."""
  81 
  82         self._text_view = gtk.TextView()
  83         self._text_view.set_wrap_mode(gtk.WRAP_WORD)
  84         self._text_view.set_cursor_visible(False)
  85         self._text_view.set_editable(False)
  86         util.set_widget_font(self._text_view, "monospace")
  87 
  88         scroller = gtk.ScrolledWindow()
  89         scroller.set_policy(gtk.POLICY_AUTOMATIC, gtk.POLICY_AUTOMATIC)
  90         scroller.set_shadow_type(gtk.SHADOW_ETCHED_IN)
  91         scroller.add(self._text_view)
  92 
  93         self._close_button = gtk.Button(stock=gtk.STOCK_CLOSE)
  94         button_box = gtk.HButtonBox()
  95         button_box.set_layout(gtk.BUTTONBOX_END)
  96         button_box.pack_start(self._close_button, False, False)
  97 
  98         vbox = gtk.VBox(spacing=12)
  99         vbox.pack_start(scroller, True, True)
 100         vbox.pack_start(button_box, False, False)
 101         vbox.show_all()
 102         self.add(vbox)
 103 
 104     def _on_close_button_clicked(self, *args):
 105         """Hide the window."""
 106 
 107         self._save_geometry()
 108         self.hide()
 109 
 110     def _on_close_key_pressed(self, *args):
 111         """Hide the window."""
 112 
 113         self._save_geometry()
 114         self.hide()
 115 
 116     def _on_delete_event(self, *args):
 117         """Hide the window."""
 118 
 119         self._save_geometry()
 120         self.hide()
 121         return True
 122 
 123     def _on_notify_visible(self, *args):
 124         """Save visibility."""
 125 
 126         conf.output_window.show = self.props.visible
 127 
 128     def _on_window_state_event(self, window, event):
 129         """Remember window maximization."""
 130 
 131         state = event.new_window_state
 132         maximized = bool(state & gtk.gdk.WINDOW_STATE_MAXIMIZED)
 133         conf.output_window.maximized = maximized
 134 
 135     def _save_geometry(self):
 136         """Save window geometry."""
 137 
 138         if not conf.output_window.maximized:
 139             conf.output_window.size = self.get_size()
 140             conf.output_window.position = self.get_position()
 141 
 142     def set_output(self, output):
 143         """Set output to the text view."""
 144 
 145         text_buffer = self._text_view.get_buffer()
 146         text_buffer.set_text(output)
 147         mark = text_buffer.get_insert()
 148         self._text_view.scroll_to_mark(mark, 0)

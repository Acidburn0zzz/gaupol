  19 """Dialog for warning when closing multiple documents."""
  20 
  21 
  22 import gobject
  23 import gtk
  24 
  25 from gaupol.gtk import util
  26 from gaupol.gtk.errors import Default
  27 from .glade import GladeDialog
  28 
  29 
  30 class MultiCloseDialog(GladeDialog):
  31 
  32     """Dialog for warning when closing multiple documents.
  33 
  34     Instance variables:
  35 
  36         _main_pages:     Pages with main documents to save
  37         _main_tree_view: gtk.TreeView
  38         _tran_pages:     Pages with translation documents to save
  39         _tran_tree_view: gtk.TreeView
  40         application:     Associated Application
  41     """
  42 
  43     def __init__(self, parent, application, pages):
  44 
  45         GladeDialog.__init__(self, "multiclose-dialog")
  46         self._main_tree_view = self._glade_xml.get_widget("main_tree_view")
  47         self._tran_tree_view = self._glade_xml.get_widget("tran_tree_view")
  48 
  49         self._main_pages = []
  50         self._tran_pages = []
  51         self.application = application
  52 
  53         self._init_data(pages)
  54         self._init_main_tree_view()
  55         self._init_tran_tree_view()
  56         self._init_signal_handlers()
  57         self._init_sizes()
  58         self._dialog.set_transient_for(parent)
  59         self._dialog.set_default_response(gtk.RESPONSE_YES)
  60 
  61     def _close_pages(self):
  62         """Close all handled pages."""
  63 
  64         pages = set(self._main_pages)
  65         pages.update(set(self._tran_pages))
  66         for page in pages:
  67             self.application.close(page, False)
  68 
  69     def _init_data(self, pages):
  70         """Initialize the page lists."""
  71 
  72         for page in pages:
  73             if page.project.main_changed:
  74                 self._main_pages.append(page)
  75             if page.project.tran_active and page.project.tran_changed:
  76                 self._tran_pages.append(page)
  77 
  78     def _init_main_tree_view(self):
  79         """Initialize the main tree view."""
  80 
  81         store = self._init_tree_view(self._main_tree_view)
  82         for page in self._main_pages:
  83             store.append([True, page.get_main_basename()])
  84         if len(store) == 0:
  85             self._glade_xml.get_widget("main_vbox").hide()
  86 
  87     def _init_signal_handlers(self):
  88         """Initialize signal handlers."""
  89 
  90         util.connect(self, self, "response")
  91 
  92     def _init_sizes(self):
  93         """Initialize widget sizes."""
  94 
  95         main_width   = 0
  96         main_height  = 0
  97         tran_width   = 0
  98         tran_height  = 0
  99         height_extra = 136
 100         get_size = util.get_tree_view_size
 101         if self._main_pages:
 102             main_width, main_height = get_size(self._main_tree_view)
 103             height_extra += 32
 104         if self._tran_pages:
 105             tran_width, tran_height = get_size(self._tran_tree_view)
 106             height_extra += 32
 107 
 108         width = max(main_width, tran_width) + 88 + util.EXTRA
 109         height = main_height + tran_height + height_extra + util.EXTRA
 110         util.resize_message_dialog(self._dialog, width, height)
 111 
 112     def _init_tran_tree_view(self):
 113         """Initialize the translation tree view."""
 114 
 115         store = self._init_tree_view(self._tran_tree_view)
 116         for page in self._tran_pages:
 117             store.append([True, page.get_translation_basename()])
 118         if len(store) == 0:
 119             self._glade_xml.get_widget("tran_vbox").hide()
 120 
 121     def _init_tree_view(self, tree_view):
 122         """Initialize tree view and return its list store."""
 123 
 124         tree_view.set_headers_visible(False)
 125         store = gtk.ListStore(gobject.TYPE_BOOLEAN, gobject.TYPE_STRING)
 126         tree_view.set_model(store)
 127         selection = tree_view.get_selection()
 128         selection.set_mode(gtk.SELECTION_SINGLE)
 129         selection.unselect_all()
 130 
 131         renderer = gtk.CellRendererToggle()
 132         renderer.props.activatable = True
 133         renderer.connect("toggled", self._on_tree_view_cell_toggled, store)
 134         column = gtk.TreeViewColumn("", renderer, active=0)
 135         tree_view.append_column(column)
 136 
 137         renderer = gtk.CellRendererText()
 138         column = gtk.TreeViewColumn("", renderer, text=1)
 139         tree_view.append_column(column)
 140         return store
 141 
 142     def _on_response(self, dialog, response):
 143         """Save the selected documents and close pages."""
 144 
 145         if response == gtk.RESPONSE_YES:
 146             try:
 147                 self._save_documents()
 148             except Default:
 149                 self.stop_emission("response")
 150                 return self.response(gtk.RESPONSE_CANCEL)
 151             self._close_pages()
 152 
 153     def _on_tree_view_cell_toggled(self, renderer, row, store):
 154         """Toggle the save check button value."""
 155 
 156         store[row][0] = not store[row][0]
 157         store = self._main_tree_view.get_model()
 158         mains = list(x for x in range(len(store)) if store[x][0])
 159         store = self._tran_tree_view.get_model()
 160         trans = list(x for x in range(len(store)) if store[x][0])
 161         sensitive = bool(mains or trans)
 162         self._dialog.set_response_sensitive(gtk.RESPONSE_YES, sensitive)
 163 
 164     def _save_documents(self):
 165         """Save the selected documents.
 166 
 167         Raise Default if something goes wrong.
 168         """
 169         store = self._main_tree_view.get_model()
 170         for i in (x for x in range(len(store)) if store[x][0]):
 171             self.application.save_main(self._main_pages[i])
 172 
 173         store = self._tran_tree_view.get_model()
 174         for i in (x for x in range(len(store)) if store[x][0]):
 175             self.application.save_translation(self._tran_pages[i])

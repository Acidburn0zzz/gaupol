  19 """Formatting text."""
  20 
  21 
  22 import re
  23 from gettext import gettext as _
  24 
  25 from gaupol.base import Delegate
  26 from gaupol.parser import Parser
  27 from gaupol.tags import *
  28 from .register import revertable
  29 
  30 
  31 class FormatAgent(Delegate):
  32 
  33     """Formatting text."""
  34 
  35     # pylint: disable-msg=E0203,W0201
  36 
  37     @revertable
  38     def change_case(self, rows, doc, method, register=-1):
  39         """Change the case of text.
  40 
  41         method should be 'title', 'capitalize', 'upper' or 'lower'.
  42         """
  43         new_texts = []
  44         parser = Parser(self.get_tag_regex(doc))
  45         texts = self.get_texts(doc)
  46         for row in rows:
  47             parser.set_text(texts[row])
  48             parser.text = getattr(parser.text, method)()
  49             new_texts.append(parser.get_text())
  50 
  51         self.replace_texts(rows, doc, new_texts, register=register)
  52         self.set_action_description(register, _("Changing case"))
  53 
  54     @revertable
  55     def toggle_dialogue_lines(self, rows, doc, register=-1):
  56         """Toggle dialogue lines on text."""
  57 
  58         re_tag = self.get_tag_regex(doc)
  59         parser = Parser(re_tag)
  60         texts = self.get_texts(doc)
  61         dialoguize = False
  62         for row in rows:
  63             for line in texts[row].split("\n"):
  64                 # Strip all tags from line.
  65                 # If leftover doesn't start with "-",
  66                 # dialogue lines should be added.
  67                 if re_tag is not None:
  68                     line = re_tag.sub("", line)
  69                 if not line.startswith("-"):
  70                     dialoguize = True
  71                     break
  72 
  73         new_texts = []
  74         for row in rows:
  75             parser.set_text(texts[row])
  76             parser.set_regex(r"^-\s*")
  77             parser.replacement = ""
  78             parser.replace_all()
  79             if dialoguize:
  80                 parser.set_regex(r"^")
  81                 parser.replacement = r"- "
  82                 parser.replace_all()
  83             new_texts.append(parser.get_text())
  84 
  85         self.replace_texts(rows, doc, new_texts, register=register)
  86         self.set_action_description(register, _("Toggling dialogue lines"))
  87 
  88     @revertable
  89     def toggle_italicization(self, rows, doc, register=-1):
  90         """Toggle the italicization of text."""
  91 
  92         re_tag = self.get_tag_regex(doc)
  93         format = eval(self.get_format_class_name(doc))
  94         re_italic_tag = re.compile(*format.italic_tag)
  95         texts = self.get_texts(doc)
  96         italicize = False
  97         for row in rows:
  98             # Remove tags from the start of the text,
  99             # ending after all tags are removed
 100             # or when an italic tag is found.
 101             text = texts[row][:]
 102             if re_tag is not None:
 103                 while re_tag.match(text) is not None:
 104                     if re_italic_tag.match(text) is not None:
 105                         break
 106                     text = re_tag.sub("", text, 1)
 107             # If there is no italic tag at the start of the text,
 108             # texts should be italicized.
 109             if re_italic_tag.match(text) is None:
 110                 italicize = True
 111                 break
 112 
 113         new_texts = []
 114         for row in rows:
 115             text = re_italic_tag.sub("", texts[row])
 116             if italicize:
 117                 text = format.italicize(text)
 118             new_texts.append(text)
 119 
 120         self.replace_texts(rows, doc, new_texts, register=register)
 121         self.set_action_description(register, _("Toggling italicization"))

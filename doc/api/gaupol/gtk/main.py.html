  36 def _check_dependencies():
  37     """Check existance and versions of dependencies."""
  38 
  39     if sys.version_info[:3] < (2, 5, 0):
  40         print "Python 2.5 or greater is required to run Gaupol."
  41         raise SystemExit(1)
  42 
  43     try:
  44         import gtk
  45         if gtk.pygtk_version < (2, 10, 0):
  46             raise ImportError
  47     except ImportError:
  48         print "PyGTK 2.10.0 or greater is required to run Gaupol."
  49         raise SystemExit(1)
  50 
  51     try:
  52         import gtk.glade
  53     except ImportError:
  54         print "Glade support in PyGTK is required to run Gaupol."
  55         raise SystemExit(1)
  56 
  57     try:
  58         import enchant
  59     except ImportError:
  60         print "PyEnchant not found;"
  61         print "spell-checking not possible."
  62 
  63     try:
  64         import chardet
  65     except ImportError:
  66         print "Universal Encoding Detector not found;"
  67         print "character encoding auto-detection not possible."

  69 def _list_encodings():
  70     """List all available character encodings."""
  71 
  72     from gaupol import enclib, util
  73     if util.chardet_available():
  74         print "auto"
  75     for seq in enclib.get_valid_encodings():
  76         print seq[0]

  78 def _move_eggs():
  79     """Move eggs to sys.path so that they are importable."""
  80 
  81     try:
  82         import enchant
  83         return
  84     except Exception:
  85         pass
  86     try:
  87         import pkg_resources
  88         pkg_resources.require("pyenchant")
  89     except Exception:
  90         pass

  92 def _parse_args(args):
  93     """Parse and return options and arguments."""
  94 
  95     import optparse
  96     parser = optparse.OptionParser(
  97         formatter=optparse.IndentedHelpFormatter(2, 42),
  98         usage=_("gaupol [OPTION...] [FILE...] [+NUM]"),)
  99 
 100     parser.add_option(
 101         "-c", "--config-file",
 102         action="store",
 103         type="string",
 104         metavar=_("FILE"),
 105         dest="config_file",
 106         default=None,
 107         help=_("set the configuration file to be used"),)
 108 
 109     parser.add_option(
 110         "-e", "--encoding",
 111         action="store",
 112         type="string",
 113         metavar=_("ENCODING"),
 114         dest="encoding",
 115         default=None,
 116         help=_("set the encoding used to open files"),)
 117 
 118     parser.add_option(
 119         "--list-encodings",
 120         action="store_true",
 121         dest="list_encodings",
 122         default=False,
 123         help=_("list all available character encodings"),)
 124 
 125     parser.add_option(
 126         "-t", "--translation-file",
 127         action="store",
 128         type="string",
 129         metavar=_("FILE"),
 130         dest="translation_file",
 131         default=None,
 132         help=_("open translation file"),)
 133 
 134     parser.add_option(
 135         "--version",
 136         action="store_true",
 137         dest="version",
 138         default=False,
 139         help=_("show version number and exit"),)
 140 
 141     parser.add_option(
 142         "-v", "--video-file",
 143         action="store",
 144         type="string",
 145         metavar=_("FILE"),
 146         dest="video_file",
 147         default=None,
 148         help=_("select video file"),)
 149 
 150     return parser.parse_args(args)

 152 def _prepare_configuration(path):
 153     """Set the configuration file to use."""
 154 
 155     if path is not None:
 156         from gaupol.gtk import paths
 157         paths.CONFIG_FILE = os.path.abspath(path)
 158     from gaupol.gtk import conf
 159     atexit.register(conf.write)
 160 
 161 def _prepare_gettext():
 162     """Assign gettext domains."""
 163 
 164     import gtk.glade
 165     from gaupol.gtk import paths
 166     locale.setlocale(locale.LC_ALL, "")
 167     gettext.bindtextdomain("gaupol", paths.LOCALE_DIR)
 168     gettext.textdomain("gaupol")
 169     gtk.glade.bindtextdomain("gaupol", paths.LOCALE_DIR)
 170     gtk.glade.textdomain("gaupol")
 171 
 172 def _prepare_ui():
 173     """Prepare user interface stuff."""
 174 
 175     import gobject
 176     gobject.threads_init()
 177 
 178     import gtk
 179     from gaupol.gtk import paths
 180     rc_file = os.path.join(paths.DATA_DIR, "gtkrc")
 181     gtk.rc_add_default_file(rc_file)
 182 
 183     from gaupol.gtk.dialogs import debug
 184     sys.excepthook = debug.show
 185 
 186 def _show_version():
 187     """Show the version number."""
 188 
 189     from gaupol import __version__
 190     print "gaupol %s" % __version__
 191 
 192 def _start(opts, args):
 193     """Start application and open files given as arguments."""
 194 
 195     from gaupol.gtk.app import Application
 196     application = Application()
 197     if args:
 198         row = None
 199         re_jump = re.compile(r"\+\d*")
 200         for arg in (x for x in args if re_jump.match(x) is not None):
 201             row = (max(0, int(arg[1:]) - 1) if arg[1:] else -1)
 202             args.remove(arg)
 203         paths = list(os.path.abspath(x) for x in args)
 204         application.open_main_files(paths, opts.encoding)
 205         page = application.get_current_page()
 206         if opts.translation_file:
 207             path = os.path.abspath(opts.translation_file)
 208             application.open_translation_file(path, opts.encoding)
 209         if opts.video_file and os.path.isfile(opts.video_file):
 210             path = os.path.abspath(opts.video_file)
 211             page.project.video_path = path
 212         if row is not None:
 213             row = min(len(page.project.times) - 1, row)
 214             page.view.set_focus(row)
 215     import gtk
 216     gtk.main()
 217 
 218 def main(args):
 219     """Parse arguments and start application."""
 220 
 221     _move_eggs()
 222     _check_dependencies()
 223     _prepare_gettext()
 224     opts, args = _parse_args(args)
 225     if opts.list_encodings:
 226         return _list_encodings()
 227     if opts.version:
 228         return _show_version()
 229     _prepare_configuration(opts.config_file)
 230     _prepare_ui()
 231     _start(opts, args)

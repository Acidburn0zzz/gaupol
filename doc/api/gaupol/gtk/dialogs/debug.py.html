  42 class DebugDialog(GladeDialog):
  43 
  44     """Dialog for displaying a traceback in case of an unhandled exception.
  45 
  46     Instance variables:
  47 
  48         _code_lines: List of lines of code displayed
  49         _text_view:  gtk.TextView displaying the traceback
  50     """
  51 
  52     def __init__(self):
  53 
  54         GladeDialog.__init__(self, "debug-dialog")
  55 
  56         self._code_lines = []
  57         self._text_view  = self._glade_xml.get_widget("text_view")
  58 
  59         self._init_text_tags()
  60         self._init_signal_handlers()
  61         self._dialog.set_default_response(gtk.RESPONSE_CLOSE)
  62 
  63     def _init_signal_handlers(self):
  64         """Initialize signal handlers."""
  65 
  66         util.connect(self, self, "response")
  67         util.connect(self, "_text_view", "motion-notify-event")
  68 
  69     def _init_text_tags(self):
  70         """Initialize tags for the text buffer."""
  71 
  72         text_buffer = self._text_view.get_buffer()
  73         text_buffer.create_tag("bold", weight=pango.WEIGHT_BOLD)
  74         text_buffer.create_tag("large", scale=pango.SCALE_LARGE)
  75         text_buffer.create_tag("margin_12", left_margin=12)
  76         text_buffer.create_tag("margin_24", left_margin=24)
  77         text_buffer.create_tag("monospace", family="monospace")
  78 
  79     def _insert_link(self, path, lineno, *tags):
  80         """Insert path into the text view."""
  81 
  82         text_buffer = self._text_view.get_buffer()
  83         tag = text_buffer.create_tag(None, foreground="blue")
  84         tag.props.underline = pango.UNDERLINE_SINGLE
  85         tag.connect("event", self._on_text_view_link_tag_event)
  86         path = os.path.abspath(path)
  87         if path.startswith(os.getcwd()):
  88             path = path.replace(os.getcwd(), "")[len(os.sep):]
  89         tag.set_data("path", os.path.abspath(path))
  90         tag.set_data("lineno", lineno)
  91         end_iter = text_buffer.get_end_iter()
  92         tag_table = text_buffer.get_tag_table()
  93         tags = tuple(tag_table.lookup(x) for x in tags)
  94         text_buffer.insert_with_tags(end_iter, path, tag, *tags)
  95 
  96     def _insert_text(self, text, *tags):
  97         """Insert text with tags to the text view."""
  98 
  99         text_buffer = self._text_view.get_buffer()
 100         end_iter = text_buffer.get_end_iter()
 101         text_buffer.insert_with_tags_by_name(end_iter, text, *tags)
 102 
 103     def _on_response(self, dialog, response):
 104         """Do not send response if reporting bug."""
 105 
 106         if response == gtk.RESPONSE_YES:
 107             util.browse_url(urls.BUG_REPORT)
 108             self.stop_emission("response")
 109 
 110     def _on_text_view_link_tag_event(self, tag, text_view, event, itr):
 111         """Open linked file in editor."""
 112 
 113         if event.type == gtk.gdk.BUTTON_RELEASE and event.button == 1:
 114             text_buffer = self._text_view.get_buffer()
 115             if not text_buffer.get_selection_bounds():
 116                 self._open_link(tag)
 117 
 118     def _on_text_view_motion_notify_event(self, text_view, event):
 119         """Change mouse pointer when hovering over a link."""
 120 
 121         window = gtk.TEXT_WINDOW_WIDGET
 122         x = int(event.x)
 123         y = int(event.y)
 124         x, y = text_view.window_to_buffer_coords(window, x, y)
 125         window = text_view.get_window(gtk.TEXT_WINDOW_TEXT)
 126         for tag in text_view.get_iter_at_location(x, y).get_tags():
 127             if tag.get_data("path") is not None:
 128                 window.set_cursor(util.HAND_CURSOR)
 129                 text_view.window.get_pointer()
 130                 return
 131         window.set_cursor(util.INSERT_CURSOR)
 132         text_view.window.get_pointer()
 133 
 134     def _open_link(self, tag):
 135         """Open linked file in editor."""
 136 
 137         def on_editor_exit(pid, return_value):
 138             if return_value != 0:
 139                 self._show_editor_error_dialog()
 140         path = util.shell_quote_path(tag.get_data("path"))
 141         process = util.start_process("%s %s +%d" % (
 142             conf.debug.editor, path, tag.get_data("lineno")))
 143         gobject.child_watch_add(process.pid, on_editor_exit)
 144         tag.props.foreground = "purple"
 145 
 146     def _print_platform(self):
 147         """Print platform information."""
 148 
 149         def insert_text(text, *tags):
 150             self._insert_text(text, "margin_12", *tags)
 151         insert_text("System: ", "bold")
 152         insert_text(platform.system())
 153         insert_text("\n")
 154         insert_text("Desktop environment: ", "bold")
 155         insert_text(util.get_desktop_environment() or "?")
 156         insert_text("\n")
 157 
 158     def _print_traceback(self, exctype, value, tb, limit=None):
 159         """Print up to limit stack trace entries from traceback."""
 160 
 161         def insert_text(text, *tags):
 162             self._insert_text(text, "margin_12", *tags)
 163 
 164         i = 0
 165         while tb is not None and (limit is None or i < limit):
 166             frame = tb.tb_frame
 167             lineno = tb.tb_lineno
 168             code = frame.f_code
 169             filename = code.co_filename
 170             name = code.co_name
 171             insert_text("File: ", "bold")
 172             self._insert_link(filename, lineno)
 173             insert_text("\n")
 174             insert_text("Line: ", "bold")
 175             insert_text(str(lineno))
 176             insert_text("\n")
 177             insert_text("In: ", "bold")
 178             insert_text(name)
 179             insert_text("\n")
 180             line = linecache.getline(filename, lineno).strip()
 181             if line:
 182                 self._code_lines.append(line)
 183                 insert_text("\n%s\n\n" % line, "monospace", "margin_24")
 184             tb = tb.tb_next
 185             i += 1
 186 
 187         exception = traceback.format_exception_only(exctype, value)[0]
 188         message = ""
 189         if exception.count(" "):
 190             exception, message = exception.split(" ", 1)
 191         insert_text(exception + " ", "bold")
 192         insert_text(message)
 193         insert_text("\n")
 194 
 195     def _print_versions(self):
 196         """Print version information."""
 197 
 198         def insert_text(text, *tags):
 199             self._insert_text(text, "margin_12", *tags)
 200         insert_text("Gaupol: ", "bold")
 201         insert_text(__version__)
 202         insert_text("\n")
 203         insert_text("Python: ", "bold")
 204         insert_text("%d.%d.%d" % sys.version_info[:3])
 205         insert_text("\n")
 206         insert_text("GTK: ", "bold")
 207         insert_text("%d.%d.%d" % gtk.gtk_version)
 208         insert_text("\n")
 209         insert_text("PyGTK: ", "bold")
 210         insert_text("%d.%d.%d" % gtk.pygtk_version)
 211         insert_text("\n")
 212         insert_text("PyEnchant: ", "bold")
 213         insert_text(util.get_enchant_version() or "N/A")
 214         insert_text("\n")
 215         insert_text("Universal Encoding Detector: ", "bold")
 216         insert_text(util.get_chardet_version() or "N/A")
 217         insert_text("\n")
 218 
 219     def _resize(self):
 220         """Resize dialog based on the text view's content."""
 221 
 222         # Get required width and height to display text.
 223         text_buffer = self._text_view.get_buffer()
 224         bounds = text_buffer.get_bounds()
 225         text = text_buffer.get_text(*bounds)
 226         label = gtk.Label(text)
 227         text_width, height = label.size_request()
 228 
 229         # Get required width to display code.
 230         attrs = pango.AttrList()
 231         attrs.insert(pango.AttrFamily("monospace", 0, -1))
 232         label.set_attributes(attrs)
 233         label.set_text("\n".join(self._code_lines))
 234         code_width = label.size_request()[0]
 235 
 236         # Set dialog size.
 237         width = max(text_width, code_width) + 150 + util.EXTRA
 238         height = height + 160 + util.EXTRA
 239         util.resize_message_dialog(self, width, height)
 240 
 241     def _show_editor_error_dialog(self):
 242         """Show an error dialog after failing to open editor."""
 243 
 244         title = _('Failed to open editor "%s"') % conf.debug.editor
 245         message = _('To change the editor, edit option "editor" under ' \
            'section "debug" in configuration file "%s".') % conf.CONFIG_FILE
 246         dialog = ErrorDialog(self._dialog, title, message)
 247         dialog.add_button(gtk.STOCK_OK, gtk.RESPONSE_OK)
 248         self.flash_dialog(dialog)
 249 
 250     def set_text(self, exctype, value, tb):
 251         """Set text to the text view."""
 252 
 253         self._insert_text("Traceback", "large", "bold")
 254         self._insert_text("\n\n")
 255         self._print_traceback(exctype, value, tb, 100)
 256         self._insert_text("Platform", "large", "bold")
 257         self._insert_text("\n\n")
 258         self._print_platform()
 259         self._insert_text("\n")
 260         self._insert_text("Versions", "large", "bold")
 261         self._insert_text("\n\n")
 262         self._print_versions()
 263 
 264         self._resize()
 265 
 266 
 267 def show(exctype, value, tb):
 268     """Show exception traceback in dialog.
 269 
 270     This function is usable as sys.excepthook.
 271     """
 272     traceback.print_exception(exctype, value, tb)
 273     if not isinstance(value, Exception):
 274         return
 275     try:
 276         dialog = DebugDialog()
 277         dialog.set_text(exctype, value, tb)
 278         response = dialog.run()
 279         dialog.destroy()
 280         if response == gtk.RESPONSE_NO:
 281             try:
 282                 gtk.main_quit()
 283             except RuntimeError:
 284                 raise SystemExit(1)
 285     except Exception:
 286         traceback.print_exc()

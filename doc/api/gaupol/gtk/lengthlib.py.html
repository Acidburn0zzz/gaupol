  19 """Functions to calculate line lengths and to show them in widgets.
  20 
  21 Module variables:
  22 
  23     _COUNTER: Instance of _Counter used
  24     RE_TAG:   Regular expression for a tag of any format
  25 """
  26 
  27 
  28 from __future__ import division
  29 import re
  30 
  31 import gtk
  32 import pango
  33 
  34 from gaupol.gtk import conf, cons
  35 
  36 
  37 RE_TAG = re.compile(r"(^[/\\_]|<.*?>|\{.*?\})")
  38 
  39 
  40 class _Counter(object):
  41 
  42     """Line length counter.
  43 
  44     Instance variables:
  45 
  46         _layout:   pango.Layout used to determine pixel sizes
  47         em_length: Length of the letter 'M' in the current font used
  48     """
  49 
  50     def __init__(self):
  51 
  52         self._layout     = None
  53         self.em_length   = None
  54         self.get_lengths = None
  55 
  56         self._update_em_length()
  57         self._update_length_func()
  58         self._init_signal_handlers()
  59 
  60     def _init_signal_handlers(self):
  61         """Initialize signal handlers."""
  62 
  63         for option in ("font", "length_unit", "use_default_font"):
  64             conf.connect(self, "editor", option)
  65 
  66     def _on_conf_editor_notify_font(self, *args):
  67         """Update the length of em."""
  68 
  69         self._update_em_length()
  70 
  71     def _on_conf_editor_notify_length_unit(self, *args):
  72         """Update the length function used."""
  73 
  74         self._update_length_func()
  75 
  76     def _on_conf_editor_notify_use_default_font(self, *args):
  77         """Update the length of em."""
  78 
  79         self._update_em_length()
  80 
  81     def _update_em_length(self):
  82         """Update the length of em."""
  83 
  84         self._layout = gtk.Label().get_layout().copy()
  85         if not conf.editor.use_default_font:
  86             font_desc = pango.FontDescription(conf.editor.font)
  87             self._layout.set_font_description(font_desc)
  88         self._layout.set_text("M")
  89         self.em_length = self._layout.get_pixel_size()[0]
  90 
  91     def _update_length_func(self):
  92         """Update the length function used."""
  93 
  94         if conf.editor.length_unit == cons.LENGTH_UNIT.CHAR:
  95             self.get_lengths = self.get_char_lengths
  96         elif conf.editor.length_unit == cons.LENGTH_UNIT.EM:
  97             self.get_lengths = self.get_em_lengths
  98 
  99     def get_char_lengths(self, text, strip, floor):
 100         """Get a list of line lengths measured in characters."""
 101 
 102         if strip:
 103             text = RE_TAG.sub("", text)
 104         return list(len(x) for x in text.split("\n"))
 105 
 106     def get_em_lengths(self, text, strip, floor):
 107         """Get a list of line lengths measured in ems."""
 108 
 109         if strip:
 110             text = RE_TAG.sub("", text)
 111         lengths = []
 112         for line in text.split("\n"):
 113             self._layout.set_text(line)
 114             length = self._layout.get_pixel_size()[0] / self.em_length
 115             if floor:
 116                 length = int(length)
 117             lengths.append(length)
 118         return lengths
 119 
 120 
 121 _COUNTER = _Counter()
 122 
 123 
 124 def _on_text_view_expose_event(text_view, event):
 125     """Calculate and show line lengths in the margin."""
 126 
 127     text_buffer = text_view.get_buffer()
 128     text = text_buffer.get_text(*text_buffer.get_bounds())
 129     lengths = get_lengths(text)
 130 
 131     layout = pango.Layout(text_view.get_pango_context())
 132     layout.set_markup("\n".join(list(str(x) for x in lengths)))
 133     layout.set_alignment(pango.ALIGN_RIGHT)
 134 
 135     width = layout.get_pixel_size()[0]
 136     text_view.set_border_window_size(gtk.TEXT_WINDOW_RIGHT, width + 4)
 137     y = -text_view.window_to_buffer_coords(gtk.TEXT_WINDOW_RIGHT, 2, 0)[1]
 138     window = text_view.get_window(gtk.TEXT_WINDOW_RIGHT)
 139     window.clear()
 140     text_view.style.paint_layout(
 141         window, gtk.STATE_NORMAL, True, None, text_view, None, 2, y, layout)
 142 
 143 def connect_text_view(text_view):
 144     """Connect text view to show line lengths in the margin."""
 145 
 146     layout = pango.Layout(text_view.get_pango_context())
 147     layout.set_text("8")
 148     width = layout.get_pixel_size()[0]
 149     text_view.set_border_window_size(gtk.TEXT_WINDOW_RIGHT, width + 4)
 150     handler_id = text_view.connect("expose-event", _on_text_view_expose_event)
 151     text_view.set_data("lengthlib_handler_id", handler_id)
 152     return handler_id
 153 
 154 def disconnect_text_view(text_view):
 155     """Disconnect text view from showing line lengths in the margin."""
 156 
 157     text_view.set_border_window_size(gtk.TEXT_WINDOW_RIGHT, 0)
 158     handler_id = text_view.get_data("lengthlib_handler_id")
 159     text_view.set_data("lengthlib_handler_id", None)
 160     return text_view.disconnect(handler_id)
 161 
 162 def func(text):
 163     """Get the length (float) of text without stripping tags.
 164 
 165     This is suitable for use as liner length_func.
 166     """
 167     return sum(_COUNTER.get_lengths(text, False, False))
 168 
 169 def get_lengths(text):
 170     """Get a list of floored line lengths without tags."""
 171 
 172     return _COUNTER.get_lengths(text, True, True)

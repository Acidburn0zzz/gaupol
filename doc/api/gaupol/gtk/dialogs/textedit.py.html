  19 """Dialog for editing the text of a single subtitle."""
  20 
  21 
  22 import gtk
  23 
  24 from gaupol.gtk import conf, util
  25 
  26 
  27 class TextEditDialog(gtk.Dialog):
  28 
  29     """Dialog for editing the text of a single subtitle.
  30 
  31     Instance variables:
  32 
  33         _text_view: gtk.TextView
  34     """
  35 
  36     def __init__(self, parent, text):
  37 
  38         gtk.Dialog.__init__(self)
  39 
  40         self._text_view = None
  41 
  42         self._init_dialog(parent)
  43         self._init_text_view(text)
  44         self._init_sizes()
  45         self.show_all()
  46 
  47     def _init_dialog(self, parent):
  48         """Initialize the dialog."""
  49 
  50         self.add_button(gtk.STOCK_CANCEL, gtk.RESPONSE_CANCEL)
  51         self.add_button(gtk.STOCK_OK, gtk.RESPONSE_OK)
  52         self.set_default_response(gtk.RESPONSE_OK)
  53         self.set_has_separator(False)
  54         self.set_transient_for(parent)
  55         self.set_border_width(6)
  56         self.set_modal(True)
  57 
  58     def _init_text_view(self, text):
  59         """Initialize the text view."""
  60 
  61         self._text_view = gtk.TextView()
  62         util.prepare_text_view(self._text_view)
  63         self._text_view.set_wrap_mode(gtk.WRAP_NONE)
  64         self._text_view.set_accepts_tab(False)
  65         text_buffer = self._text_view.get_buffer()
  66         text_buffer.set_text(text)
  67 
  68         scroller = gtk.ScrolledWindow()
  69         scroller.set_border_width(6)
  70         scroller.set_policy(gtk.POLICY_AUTOMATIC, gtk.POLICY_AUTOMATIC)
  71         scroller.set_shadow_type(gtk.SHADOW_IN)
  72         scroller.add(self._text_view)
  73         vbox = self.get_child()
  74         vbox.add(scroller)
  75 
  76     def _init_sizes(self):
  77         """Initialize widget sizes."""
  78 
  79         label = gtk.Label("\n".join(["M" * 36] * 5))
  80         if not conf.editor.use_default_font:
  81             util.set_label_font(label, conf.editor.font)
  82         width, height = label.size_request()
  83         self._text_view.set_size_request(width + 4, height + 7)
  84 
  85     def get_text(self):
  86         """Get the text."""
  87 
  88         text_buffer = self._text_view.get_buffer()
  89         bounds = text_buffer.get_bounds()
  90         return text_buffer.get_text(*bounds)

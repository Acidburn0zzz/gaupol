  19 """Opening subtitle files."""
  20 
  21 
  22 from gaupol import cons
  23 from gaupol.base import Delegate, notify_frozen
  24 from gaupol.determiner import FormatDeterminer
  25 from gaupol.files import *
  26 from .index import SHOW, HIDE, DURN
  27 
  28 
  29 class OpenAgent(Delegate):
  30 
  31     """Opening subtitle files.
  32 
  33     Instance variables:
  34 
  35         _sort_count: The amount of sort operations made
  36     """
  37 
  38     # pylint: disable-msg=E0203,W0201
  39 
  40     def __init__(self, master):
  41 
  42         Delegate.__init__(self, master)
  43 
  44         self._sort_count = 0
  45 
  46     def _adapt_translation(self, shows, hides, texts):
  47         """Open translation file data in an adaptive manner."""
  48 
  49         m = 0
  50         t = 0
  51         positions = self.get_positions(self.tran_file.mode)
  52         while t < len(shows):
  53             middle = self.calc.get_middle(shows[t], hides[t])
  54             time, frame = self.expand_positions(shows[t], hides[t])
  55             if m == len(self.times) or middle < positions[m][SHOW]:
  56                 self.insert_subtitles(
  57                     [m], [time], [frame], [u""], [texts[t]], register=None)
  58                 m += 1
  59                 t += 1
  60                 continue
  61             if middle > positions[m][HIDE]:
  62                 m += 1
  63                 continue
  64             self.tran_texts[m] = texts[t]
  65             m += 1
  66             t += 1
  67 
  68     def _sort(self, shows, hides, texts):
  69         """Sort data based on show positions.
  70 
  71         Return shows, hides, texts.
  72         """
  73         def sort(x, y):
  74             value = cmp(x[0], y[0])
  75             if value == -1:
  76                 self._sort_count += 1
  77             return value
  78         self._sort_count = 0
  79         data = list([shows[i], hides[i], texts[i]] for i in range(len(shows)))
  80         data.sort(sort)
  81         shows = list(x[0] for x in data)
  82         hides = list(x[1] for x in data)
  83         texts = list(x[2] for x in data)
  84         return shows, hides, texts
  85 
  86     @notify_frozen
  87     def open_main(self, path, encoding):
  88         """Open main file reading positions and texts.
  89 
  90         Raise IOError if reading fails.
  91         Raise UnicodeError if decoding fails.
  92         Raise FormatError if unable to detect the format.
  93         Return sort count.
  94         """
  95         format = FormatDeterminer(path, encoding).determine()
  96         main_file = eval(format.class_name)(path, encoding)
  97         shows, hides, texts = self._sort(*main_file.read())
  98         self.main_file = main_file
  99 
 100         self.times = []
 101         self.frames = []
 102         self.main_texts = []
 103         self.tran_texts = []
 104         for i in range(len(shows)):
 105             time, frame = self.expand_positions(shows[i], hides[i])
 106             self.times.append(time)
 107             self.frames.append(frame)
 108             self.main_texts.append(texts[i])
 109             self.tran_texts.append(u"")
 110 
 111         # Get framerate from MPsub header.
 112         if self.main_file.format == cons.FORMAT.MPSUB:
 113             if self.main_file.framerate is not None:
 114                 self.set_framerate(self.main_file.framerate, register=None)
 115 
 116         self.main_changed = 0
 117         self.tran_changed = 0
 118         self.tran_active = False
 119         self.emit("main-file-opened", self.main_file)
 120         return self._sort_count
 121 
 122     @notify_frozen
 123     def open_translation(self, path, encoding, smart=True):
 124         """Open translation file reading texts.
 125 
 126         Raise IOError if reading fails.
 127         Raise UnicodeError if decoding fails.
 128         Raise FileFormatError if unable to detect the format.
 129         Return sort count.
 130         """
 131         format = FormatDeterminer(path, encoding).determine()
 132         tran_file = eval(format.class_name)(path, encoding)
 133         shows, hides, texts = self._sort(*tran_file.read())
 134         self.tran_file = tran_file
 135 
 136         blocked = self.block("subtitles-inserted")
 137         self.tran_texts = [u""] * len(self.tran_texts)
 138         if smart:
 139             self._adapt_translation(shows, hides, texts)
 140         else:
 141             excess = len(texts) - len(self.main_texts)
 142             if excess > 0:
 143                 rows = range(len(self.times), len(self.times) + excess)
 144                 self.insert_blank_subtitles(rows, register=None)
 145             self.tran_texts[:len(texts)] = texts
 146         self.unblock("subtitles-inserted", blocked)
 147 
 148         self.tran_changed = 0
 149         self.tran_active = True
 150         self.emit("translation-file-opened", self.tran_file)
 151         return self._sort_count

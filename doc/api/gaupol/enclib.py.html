 140 def _translate(name):
 141     """Translate weird encoding name.
 142 
 143     Raise ValueError if not found.
 144     Return python name, display name, description.
 145     """
 146     name = _RE_ILLEGAL.sub("_", name.lower())
 147     if name in aliases:
 148         name = aliases[name]
 149 
 150     for seq in _ENCODINGS:
 151         if seq[0] == name:
 152             return seq
 153     raise ValueError
 154 
 155 def detect(path):
 156     """Detect the character encoding in file.
 157 
 158     Raise IOError if reading fails.
 159     Raise ValueError if not found.
 160     Return python name.
 161     """
 162     fobj = open(path, "r")
 163     detector = UniversalDetector()
 164     try:
 165         for line in fobj.readlines():
 166             detector.feed(line)
 167             if detector.done:
 168                 break
 169     finally:
 170         detector.close()
 171         fobj.close()
 172 
 173     encoding = detector.result["encoding"]
 174     if encoding is None:
 175         raise ValueError
 176     return _translate(encoding)[0]

 178 def get_display_name(python_name):
 179     """Get the display name for encoding.
 180 
 181     Raise ValueError if not found.
 182     """
 183     for seq in _ENCODINGS:
 184         if seq[0] == python_name:
 185             return seq[1]
 186     raise ValueError
 187 
 188 def get_locale_long_name():
 189     """Get the long name for locale encoding.
 190 
 191     Raise ValueError if not found.
 192     Return 'Current locale (Display name)'.
 193     """
 194     encoding = get_locale_python_name()
 195     return _("Current locale (%s)") % get_display_name(encoding)
 196 
 197 def get_locale_python_name():
 198     """Get the python name of the locale encoding or None."""
 199 
 200     name = locale.getdefaultlocale()[1]
 201     if name is not None:
 202         name = _RE_ILLEGAL.sub("_", name.lower())
 203         if name in aliases:
 204             name = aliases[name]
 205     return name
 206 
 207 def get_long_name(python_name):
 208     """Get the long name for encoding.
 209 
 210     Raise ValueError if not found.
 211     Return 'Description (Display name)'.
 212     """
 213     for seq in _ENCODINGS:
 214         if seq[0] == python_name:
 215             fields = {'description': seq[2], 'encoding': seq[1]}
 216             return _("%(description)s (%(encoding)s)") % fields
 217     raise ValueError
 218 
 219 def get_python_name(display_name):
 220     """Get the python name for encoding.
 221 
 222     Raise ValueError if not found.
 223     """
 224     for seq in _ENCODINGS:
 225         if seq[1] == display_name:
 226             return seq[0]
 227     raise ValueError
 228 
 229 def get_valid_encodings():
 230     """Get a list of valid encodings.
 231 
 232     Return list of tuples of python name, display name, description.
 233     """
 234     valid_encodings = list(_ENCODINGS)
 235     for i, seq in enumerate(reversed(valid_encodings)):
 236         if not is_valid(seq[0]):
 237             valid_encodings.pop(i)
 238     return valid_encodings
 239 
 240 def is_valid(python_name):
 241     """Return True is encoding is valid."""
 242 
 243     try:
 244         codecs.lookup(python_name)
 245         return True
 246     except LookupError:
 247         return False

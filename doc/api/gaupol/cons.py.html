 194 VIDEO_PLAYER = cons.Section()
 195 VIDEO_PLAYER.MPLAYER = cons.Member()
 196 VIDEO_PLAYER.MPLAYER.command = " ".join((
 197     "mplayer",
 198     "-identify",
 199     "-osdlevel 2",
 200     "-ss $SECONDS",
 201     "-sub $SUBFILE",
 202     "$VIDEOFILE",))
 203 VIDEO_PLAYER.MPLAYER.display_name = "MPlayer"
 204 VIDEO_PLAYER.VLC = cons.Member()
 205 VIDEO_PLAYER.VLC.command = " ".join((
 206     "vlc",
 207     "--start-time=$SECONDS",
 208     "--sub-file=$SUBFILE",
 209     "$VIDEOFILE",))
 210 VIDEO_PLAYER.VLC.display_name = "VLC"
 211 if sys.platform == "win32":
 212     VIDEO_PLAYER.MPLAYER.command[0:7] = r"%ProgramFiles%\mplayer\mplayer.exe"
 213     VIDEO_PLAYER.VLC.command[0:3] = r"%ProgramFiles%\VideoLAN\vlc\vlc.exe"
 214 VIDEO_PLAYER.finalize()

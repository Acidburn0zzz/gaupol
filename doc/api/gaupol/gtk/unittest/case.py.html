  19 """Base class for GTK unit test cases."""
  20 
  21 
  22 import gtk
  23 
  24 from gaupol import unittest
  25 from gaupol.gtk.app import Application
  26 from gaupol.gtk.page import Page
  27 
  28 
  29 class TestCase(unittest.TestCase):
  30 
  31     """Base class for GTK unit test cases."""
  32 
  33     APPLICATION = Application()
  34 
  35     def __init__(self):
  36 
  37         unittest.TestCase.__init__(self)
  38         while gtk.events_pending():
  39             gtk.main_iteration()
  40 
  41     def get_application(self):
  42         """Get a new application."""
  43 
  44         application = self.APPLICATION
  45         while application.pages:
  46             application.close(application.pages[0], False)
  47         application.add_new_page(self.get_page())
  48         application.add_new_page(self.get_page())
  49         return application
  50 
  51     def get_page(self):
  52         """Get a new page."""
  53 
  54         page = Page()
  55         page.project.open_main(self.get_subrip_path(), "ascii")
  56         page.project.open_translation(self.get_microdvd_path(), "ascii")
  57         return page

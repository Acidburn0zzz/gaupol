  19 """Subtitle file format determiner."""
  20 
  21 
  22 import re
  23 
  24 from gaupol import cons, util
  25 from gaupol.errors import FormatError
  26 from gaupol.files import *
  27 
  28 
  29 class FormatDeterminer(object):
  30 
  31     """Subtitle file format determiner.
  32 
  33     Instance variables:
  34 
  35         encoding: Character encoding to open the file with
  36         path:     Filepath
  37         re_ids:   List of tuples of format, regular expression
  38     """
  39 
  40     def __init__(self, path, encoding):
  41 
  42         self.encoding = encoding
  43         self.path     = path
  44         self.re_ids   = []
  45 
  46         for format in cons.FORMAT.members:
  47             re_id = re.compile(*eval(format.class_name).identifier)
  48             self.re_ids.append((format, re_id))
  49 
  50     def determine(self):
  51         """Determine the format of the file.
  52 
  53         Raise IOError if reading fails.
  54         Raise UnicodeError if decoding fails.
  55         Raise FormatError if unable to detect the format.
  56         Return FORMAT constant.
  57         """
  58         for line in util.readlines(self.path, self.encoding):
  59             for format, re_id in self.re_ids:
  60                 if re_id.search(line) is not None:
  61                     return format
  62         raise FormatError

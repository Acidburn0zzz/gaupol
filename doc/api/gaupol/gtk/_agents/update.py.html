  19 """Updating the application GUI."""
  20 
  21 
  22 import os
  23 from gettext import gettext as _
  24 
  25 import gobject
  26 import gtk
  27 
  28 from gaupol.base import Delegate
  29 from gaupol.gtk import conf, cons, util
  30 from gaupol.gtk._actions import *
  31 from gaupol.gtk._actions import ACTIONS
  32 
  33 
  34 class UpdateAgent(Delegate):
  35 
  36     """Updating the application GUI.
  37 
  38     Instance attributes:
  39 
  40         _message_id:  Statusbar message ID
  41         _message_tag: Statusbar GObject timeout tag
  42     """
  43 
  44     # pylint: disable-msg=E0203,W0201
  45 
  46     def __init__(self, master):
  47 
  48         Delegate.__init__(self, master)
  49 
  50         self._message_id  = None
  51         self._message_tag = None
  52 
  53     def _update_actions(self, page):
  54         """Update sensitivities of all UI manager actions for page."""
  55 
  56         for name in ACTIONS:
  57             cls = eval(name)
  58             doable = cls.is_doable(self, page)
  59             for path in cls.paths:
  60                 self.uim.get_action(path).set_sensitive(doable)
  61             for widget in cls.widgets:
  62                 getattr(self, widget).set_sensitive(doable)
  63 
  64     def _update_revert(self, page):
  65         """Update tooltips for undo and redo."""
  66 
  67         if page is not None and page.project.can_undo():
  68             tip = _('Undo "%s"') % page.project.undoables[0].description
  69             self.undo_button.set_tooltip(self.tooltips, tip)
  70             self.uim.get_action("/ui/menubar/edit/undo").props.tooltip = tip
  71         if page is not None and page.project.can_redo():
  72             tip = _('Redo "%s"') % page.project.redoables[0].description
  73             self.redo_button.set_tooltip(self.tooltips, tip)
  74             self.uim.get_action("/ui/menubar/edit/redo").props.tooltip = tip
  75 
  76     def _update_widgets(self, page):
  77         """Update the states of widgets."""
  78 
  79         if page is None:
  80             self.tooltips.disable()
  81             self.window.set_title("Gaupol")
  82             self.video_button.get_data("label").set_text("")
  83             self.push_message(None)
  84             return
  85         self.tooltips.enable()
  86         self.window.set_title(page.tab_label.get_text())
  87         self.uim.get_action(page.edit_mode.uim_path).set_active(True)
  88         path = cons.FRAMERATE.uim_paths[page.project.framerate]
  89         self.uim.get_action(path).set_active(True)
  90         self.framerate_combo.set_active(page.project.framerate)
  91         for i, path in enumerate(cons.COLUMN.uim_paths):
  92             visible = page.view.get_column(i).props.visible
  93             self.uim.get_action(path).set_active(visible)
  94         if page.project.video_path is not None:
  95             basename = os.path.basename(page.project.video_path)
  96             self.video_button.get_data("label").set_text(basename)
  97         else:
  98             self.video_button.get_data("label").set_text("")
  99         self.tooltips.set_tip(self.video_button, page.project.video_path)
 100 
 101     def on_activate_next_project_activate(self, *args):
 102         """Activate the project in the next tab."""
 103 
 104         self.notebook.next_page()
 105 
 106     def on_activate_previous_project_activate(self, *args):
 107         """Activate the project in the previous tab."""
 108 
 109         self.notebook.prev_page()
 110 
 111     def on_conf_application_window_notify_toolbar_style(self, *args):
 112         """Change the style of the main toolbar."""
 113 
 114         toolbar = self.uim.get_widget("/ui/main_toolbar")
 115         style = conf.application_window.toolbar_style
 116         if style == cons.TOOLBAR_STYLE.DEFAULT:
 117             return toolbar.unset_style()
 118         toolbar.set_style(style.value)
 119 
 120     def on_move_tab_left_activate(self, *args):
 121         """Move the current tab to the left."""
 122 
 123         page = self.get_current_page()
 124         scroller = page.view.get_parent()
 125         index = self.pages.index(page)
 126         self.notebook.reorder_child(scroller, index - 1)
 127 
 128     def on_move_tab_right_activate(self, *args):
 129         """Move the current tab to the right."""
 130 
 131         page = self.get_current_page()
 132         scroller = page.view.get_parent()
 133         index = self.pages.index(page)
 134         self.notebook.reorder_child(scroller, index + 1)
 135 
 136     def on_notebook_page_reordered(self, notebook, scroller, index):
 137         """Update the list of pages to match the new order."""
 138 
 139         view = scroller.get_child()
 140         for page in self.pages:
 141             if page.view == view:
 142                 break
 143         # pylint: disable-msg=W0631
 144         self.pages.remove(page)
 145         self.pages.insert(index, page)
 146         self.update_gui()
 147         self.emit("pages-reordered", page, index)
 148 
 149     def on_notebook_switch_page(self, notebook, pointer, index):
 150         """Update GUI for the page switched to."""
 151 
 152         if self.pages:
 153             page = self.pages[index]
 154             self.update_gui()
 155             page.view.grab_focus()
 156 
 157     def on_view_button_press_event(self, view, event):
 158         """Display a pop-up menu to edit data."""
 159 
 160         if event.button != 3:
 161             return False
 162         x = int(event.x)
 163         y = int(event.y)
 164         value = view.get_path_at_pos(x, y)
 165         if value is None:
 166             return False
 167         path, column, x, y = value
 168         if path[0] not in view.get_selected_rows():
 169             view.set_cursor(path[0], column)
 170             view.update_headers()
 171         menu = self.uim.get_widget("/ui/view_popup")
 172         menu.popup(None, None, None, event.button, event.time)
 173         return True
 174 
 175     def on_view_move_cursor(self, *args):
 176         """Update GUI after moving cursor in the view."""
 177 
 178         self.update_gui()
 179 
 180     def on_view_selection_changed(self, *args):
 181         """Update GUI after changing selection in the view."""
 182 
 183         self.update_gui()
 184 
 185     def on_window_window_state_event(self, window, event):
 186         """Save window maximization."""
 187 
 188         state = event.new_window_state
 189         maximized = bool(state & gtk.gdk.WINDOW_STATE_MAXIMIZED)
 190         conf.application_window.maximized = maximized
 191 
 192     def push_message(self, message, clear=True):
 193         """Push message to the statusbar."""
 194 
 195         if self._message_tag is not None:
 196             gobject.source_remove(self._message_tag)
 197         if self._message_id is not None:
 198             self.statusbar.remove(0, self._message_id)
 199         event_box = util.get_event_box(self.statusbar)
 200         self.tooltips.set_tip(event_box, message)
 201         if message is not None:
 202             self._message_id = self.statusbar.push(0, message)
 203             if clear:
 204                 method = self.push_message
 205                 self._message_tag = gobject.timeout_add(6000, method, None)
 206         return False
 207 
 208     def update_gui(self):
 209         """Update widget sensitivities and states for the current page."""
 210 
 211         page = self.get_current_page()
 212         self._update_actions(page)
 213         self._update_widgets(page)
 214         self._update_revert(page)

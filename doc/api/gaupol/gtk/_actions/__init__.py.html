  24 def _get_actions():
  25     """Get all UIMAction classes."""
  26 
  27     from gaupol import util
  28     from ._action import UIMAction
  29     actions = []
  30     for module in _get_modules():
  31         names = list(x for x in dir(module) if not x.startswith("_"))
  32         while "UIMAction" in names:
  33             names.remove("UIMAction")
  34         values = list(getattr(module, x) for x in names)
  35         values = list(x for x in values if isinstance(x, type))
  36         values = list(x for x in values if issubclass(x, UIMAction))
  37         for value in values:
  38             globals()[value.__name__] = value
  39             actions.append(value.__name__)
  40     return util.get_sorted_unique(actions)
  41

  55 __all__ = ACTIONS = _get_actions()

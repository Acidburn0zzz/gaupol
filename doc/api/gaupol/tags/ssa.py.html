  19 """Sub Station Alpha tag library.
  20 
  21 Module variables:
  22 
  23     _FLAGS: Common regular expression flags
  24 """
  25 
  26 
  27 import re
  28 
  29 from ._taglib import TagLibrary
  30 from .internal import Internal
  31 
  32 
  33 _FLAGS = re.MULTILINE | re.DOTALL
  34 
  35 
  36 class SubStationAlpha(TagLibrary):
  37 
  38     """Sub Station Alpha tag library.
  39 
  40     Class variables:
  41 
  42         _re_redundant: Regular expression for a redundant tag pair
  43     """
  44 
  45     tag = r"\{.*?\}", 0
  46     italic_tag = r"\{\\i[01]\}", re.IGNORECASE
  47 
  48     @classmethod
  49     def pre_decode(cls, text):
  50         """Break combined tags, e.g. {\\b1i1} to {\\b1}{\\i1}."""
  51 
  52         parts = text.split("\\")
  53         for i in range(1, len(parts)):
  54             text_so_far = "\\".join(parts[:i])
  55             if text_so_far.endswith("{"):
  56                 continue
  57             opening_index = text_so_far.rfind("{")
  58             closing_index = text_so_far.rfind("}")
  59             if opening_index > closing_index:
  60                 parts[i - 1] += "}{"
  61         return "\\".join(parts)
  62 
  63     # Decode tags leave </> for reset and a lot of tags unclosed.
  64     decode_tags = [
  65         # Bold opening
  66         (r"\{\\b[1-9]\d*\}", re.IGNORECASE,
  67             r"<b>"),
  68         # Italic opening
  69         (r"\{\\i1\}", re.IGNORECASE,
  70             r"<i>"),
  71         # Bold, Italic closing
  72         (r"\{\\(b|i)0\}", re.IGNORECASE,
  73             r"</\1>"),
  74         # Color
  75         (r"\{\\c&H([a-zA-Z0-9]{6})&\}", re.IGNORECASE,
  76             r'<color="#\1">'),
  77         # Font
  78         (r"\{\\fn(.*?)\}", re.IGNORECASE,
  79             r'<font="\1">'),
  80         # Size
  81         (r"\{\\fs(.*?)\}", re.IGNORECASE,
  82             r'<size="\1">'),
  83         # Reset
  84         (r"\{\\r\}", re.IGNORECASE,
  85             r"</>"),
  86         # Remove all else.
  87         (r"\{.*?\}", re.IGNORECASE,
  88             r"")]
  89 
  90     @classmethod
  91     def post_decode(cls, text):
  92         """Fix or add closing tags."""
  93 
  94         re_opening_tag = re.compile(*Internal.opening_tag)
  95         re_closing_tag = re.compile(*Internal.closing_tag)
  96 
  97         parts = text.split("</>")
  98         for i, part in enumerate(parts):
  99 
 100             suffix = ""
 101             opening_tags = re_opening_tag.findall(part)
 102             closing_tags = re_closing_tag.findall(part)
 103 
 104             # Find out which tags have already been closed.
 105             for j in reversed(range(len(closing_tags))):
 106                 closing_core = closing_tags[j][2:-1]
 107                 for k in range(len(opening_tags)):
 108                     opening_core = opening_tags[k][1:-1].split("=")[0]
 109                     if opening_core == closing_core:
 110                         opening_tags.pop(k)
 111                         break
 112 
 113             # Assemble suffix string to close remaining tags.
 114             for j in reversed(range(len(opening_tags))):
 115                 tag = "</" + opening_tags[j][1:-1].split("=")[0] + ">"
 116                 suffix += tag
 117 
 118             parts[i] = part + suffix
 119 
 120         return "".join(parts)
 121 
 122     @classmethod
 123     def pre_encode(cls, text):
 124         """Remove pointless closing tags at the end of the text."""
 125 
 126         regex = re.compile(*Internal.closing_tag_end)
 127         while regex.search(text) is not None:
 128             text = regex.sub("", text)
 129         return text
 130 
 131     encode_tags = [
 132         # Remove duplicate style tags (e.g. <b>foo</b><b>bar</b>).
 133         (r"</(b|i|u)>(\n?)<\1>", _FLAGS,
 134             r"\2", 3),
 135         # Remove other duplicate tags.
 136         (r"<(.*?)=(.*?)>(.*?)</\1>(\n?)<\1=\2>", _FLAGS,
 137             r"<\1=\2>\3\4", 3),
 138         # Bold and italic, \061 = 1
 139         (r"<(b|i)>", 0,
 140             r"{\\\1\061}"),
 141         # Bold and italic, \060 = 0
 142         (r"</(b|i)>", 0,
 143             r"{\\\1\060}"),
 144         # Color opening
 145         (r'<color="#(.*?)">', 0,
 146             r"{\\c&H\1&}"),
 147         # Font opening
 148         (r'<font="(.*?)">', 0,
 149             r"{\\fn\1}"),
 150         # Size opening
 151         (r'<size="(.*?)">', 0,
 152             r"{\\fs\1}"),
 153         # Color, font or size closing
 154         (r"</[a-z]{3,}>", 0,
 155             r"{\\r}"),
 156         # Remove underline.
 157         (r"</?u>", 0,
 158             r"")]
 159 
 160     # \061 = 1
 161     _re_redundant = re.compile(
 162         r"\{\\([ib])0\}([^\w\n]*?)\{\\\1\061\}", re.UNICODE)
 163 
 164     @classmethod
 165     def italicize(cls, text):
 166         """Italicize text."""
 167 
 168         return u"{\\i1}%s" % text
 169 
 170     @classmethod
 171     def remove_redundant(cls, text):
 172         """Remove redundant tags from text."""
 173 
 174         return cls._re_redundant.sub(r"\2", text)

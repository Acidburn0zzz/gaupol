  34     def _toggle_column(self, col):
  35         """Show or hide column."""
  36 
  37         page = self.get_current_page()
  38         column = page.view.get_column(col)
  39         visible = column.get_visible()
  40         active = self.uim.get_action(col.uim_path).get_active()
  41         if visible is active:
  42             return
  43         util.set_cursor_busy(self.window)
  44         column.set_visible(not visible)
  45         visible_columns = []
  46         for col in cons.COLUMN.members:
  47             if page.view.get_column(col).get_visible():
  48                 visible_columns.append(col)
  49         conf.editor.visible_cols = visible_columns
  50         self.update_gui()
  51         util.set_cursor_normal(self.window)
  52 
  53     def on_framerate_combo_changed(self, *args):
  54         """Change the framerate with which unnative units are calculated."""
  55 
  56         page = self.get_current_page()
  57         index = self.framerate_combo.get_active()
  58         framerate = cons.FRAMERATE.members[index]
  59         if framerate == page.project.framerate:
  60             return
  61         util.set_cursor_busy(self.window)
  62         page.project.change_framerate(framerate)
  63         conf.editor.framerate = framerate
  64         self.uim.get_widget(framerate.uim_path).set_active(True)
  65         if page.edit_mode != page.project.main_file.mode:
  66             rows = range(len(page.project.times))
  67             page.reload_view(rows, [SHOW, HIDE, DURN])
  68         util.set_cursor_normal(self.window)
  69 
  70     def on_output_window_notify_visible(self, *args):

  72 
  73         action = self.uim.get_action("/ui/menubar/view/output_window")
  74         action.set_active(self.output_window.props.visible)

 104     @util.gc_collected
 105     def on_show_times_activate(self, item, active_item):
 106         """Change the units in which postions are shown."""
 107 
 108         page = self.get_current_page()
 109         name = active_item.get_name()
 110         index = cons.MODE.uim_action_names.index(name)
 111         edit_mode = cons.MODE.members[index]
 112         if edit_mode == page.edit_mode:
 113             return
 114         util.set_cursor_busy(self.window)
 115         page.edit_mode = edit_mode
 116         conf.editor.mode = edit_mode
 117         has_focus = page.view.props.has_focus
 118         focus_row, focus_col = page.view.get_focus()
 119         selected_rows = page.view.get_selected_rows()
 120         scroller = page.view.get_parent()
 121         scroller.remove(page.view)
 122         page.view = View(edit_mode)
 123         self.connect_to_view_signals(page.view)
 124         scroller.add(page.view)
 125         scroller.show_all()
 126         page.reload_view_all()
 127         if not None in (focus_row, focus_col):
 128             page.view.set_focus(focus_row, focus_col)
 129         page.view.select_rows(selected_rows)
 130         if focus_row is not None:
 131             page.view.scroll_to_row(focus_row)
 132         page.view.props.has_focus = has_focus
 133         util.set_cursor_normal(self.window)

 163     def on_toggle_output_window_activate(self, *args):
 164         """Show or hide the output window."""
 165 
 166         visible = self.output_window.props.visible
 167         action = self.uim.get_action("/ui/menubar/view/output_window")
 168         active = action.get_active()
 169         if visible is not active:
 170             if self.output_window.props.visible:
 171                 return self.output_window.hide()
 172             self.output_window.show()

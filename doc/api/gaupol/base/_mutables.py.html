  19 """Observable versions of built-in mutable objects."""
  20 
  21 # pylint: disable-msg=C0103
  22 
  23 
  24 import copy
  25 
  26 
  27 def _mutation(function):
  28     """Decorator for sending a notification after mutating object."""
  29 
  30     def wrapper(*args, **kwargs):
  31         value = function(*args, **kwargs)
  32         args[0].master.notify(args[0].name)
  33         return value
  34 
  35     return wrapper
  36

  38 class ObservableDict(dict):
  39 
  40     """Observable version of dict.
  41 
  42     Instance variables:
  43 
  44         master: Master instance with a 'notify' method
  45         name:   Argument given when calling master's 'notify' method
  46     """
  47 
  48     def __copy__(self):
  49         dic = dict(copy.copy(x) for  x in self.items())
  50         return self.__class__(dic, self.master, self.name)
  51 
  52     def __deepcopy__(self, memo):
  53         dic = dict(copy.deepcopy(x) for  x in self.items())
  54         return self.__class__(dic, self.master, self.name)
  55 
  56     @_mutation
  57     def __delitem__(self, *args, **kwargs):
  58         return dict.__delitem__(self, *args, **kwargs)
  59 
  60     def __init__(self, *args, **kwargs):
  61         dict.__init__(self, *args[:-2], **kwargs)
  62         self.master = args[-2]
  63         self.name = args[-1]
  64 
  65     @_mutation
  66     def __setitem__(self, *args, **kwargs):
  67         return dict.__setitem__(self, *args, **kwargs)
  68 
  69     @_mutation
  70     def clear(self, *args, **kwargs):
  71         return dict.clear(self, *args, **kwargs)
  72 
  73     @_mutation
  74     def pop(self, *args, **kwargs):
  75         return dict.pop(self, *args, **kwargs)
  76 
  77     @_mutation
  78     def popitem(self, *args, **kwargs):
  79         return dict.popitem(self, *args, **kwargs)
  80 
  81     @_mutation
  82     def setdefault(self, *args, **kwargs):
  83         return dict.setdefault(self, *args, **kwargs)
  84 
  85     @_mutation
  86     def update(self, *args, **kwargs):
  87         return dict.update(self, *args, **kwargs)

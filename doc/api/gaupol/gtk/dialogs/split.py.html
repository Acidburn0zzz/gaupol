  19 """Dialog for splitting project in two."""
  20 
  21 
  22 import copy
  23 from gettext import gettext as _
  24 
  25 import gtk
  26 
  27 from gaupol.gtk import cons, util
  28 from gaupol.gtk.page import Page
  29 from .glade import GladeDialog
  30 
  31 
  32 class SplitDialog(GladeDialog):
  33 
  34     """Dialog for splitting a project in two.
  35 
  36     Instance variables:
  37 
  38         _dialog:        gtk.Dialog
  39         _subtitle_spin: gtk.SpinButton, first subtitle of new project
  40         application:    Associated application
  41     """
  42 
  43     def __init__(self, application):
  44 
  45         GladeDialog.__init__(self, "split-dialog")
  46         self._dialog        = self._glade_xml.get_widget("dialog")
  47         self._subtitle_spin = self._glade_xml.get_widget("subtitle_spin")
  48         self.application    = application
  49 
  50         self._init_subtitle_spin()
  51         self._init_signal_handlers()
  52         self._dialog.set_transient_for(application.window)
  53         self._dialog.set_default_response(gtk.RESPONSE_OK)
  54 
  55     def _init_signal_handlers(self):
  56         """Initialize signal handlers."""
  57 
  58         util.connect(self, self, "response")
  59 
  60     def _init_subtitle_spin(self):
  61         """Initialize the subtitle spin button."""
  62 
  63         page = self.application.get_current_page()
  64         self._subtitle_spin.set_range(2, len(page.project.times))
  65         self._subtitle_spin.set_value(1)
  66         rows = page.view.get_selected_rows()
  67         subtitle = (rows[0] + 1 if rows else 1)
  68         self._subtitle_spin.set_value(subtitle)
  69 
  70     def _on_response(self, dialog, response):
  71         """Split the current project if OK."""
  72 
  73         if response == gtk.RESPONSE_OK:
  74             self._split_project()
  75 
  76     def _remove_from_source(self, source, row):
  77         """Remove rows from the source page."""
  78 
  79         rows = range(row, len(source.project.times))
  80         source.project.block("action-done")
  81         source.project.remove_subtitles(rows)
  82         source.project.set_action_description(
  83             cons.REGISTER.DO, _("Splitting project"))
  84         source.project.unblock("action-done")
  85 
  86     def _shift_destination(self, source, destination):
  87         """Shift subtitles in the destination page."""
  88 
  89         mode = destination.project.get_mode()
  90         calc = destination.project.calc
  91         if mode == cons.MODE.TIME:
  92             count = source.project.times[-1][1]
  93             count = -1 * calc.time_to_seconds(count)
  94             method = destination.project.shift_seconds
  95         elif mode == cons.MODE.FRAME:
  96             count = -1 * source.project.frames[-1][1]
  97             method = destination.project.shift_frames
  98         method([], count, register=None)
  99 
 100     def _split_project(self):
 101         """Split the current project in two."""
 102 
 103         util.set_cursor_busy(self.application.window)
 104         row = self._subtitle_spin.get_value_as_int() - 1
 105         source = self.application.get_current_page()
 106         self.application.counter += 1
 107         destination = Page(self.application.counter)
 108 
 109         times = copy.deepcopy(source.project.times[row:])
 110         frames = copy.deepcopy(source.project.frames[row:])
 111         main_texts = copy.deepcopy(source.project.main_texts[row:])
 112         tran_texts = copy.deepcopy(source.project.tran_texts[row:])
 113         destination.project.times = times
 114         destination.project.frames = frames
 115         destination.project.main_texts = main_texts
 116         destination.project.tran_texts = tran_texts
 117         destination.reload_view_all()
 118 
 119         self._remove_from_source(source, row)
 120         self._shift_destination(source, destination)
 121         self.application.add_new_page(destination)
 122         amount = len(destination.project.times)
 123         fields = {'amount': amount, 'name': destination.untitle}
 124         self.application.push_message(
 125             _('Split %(amount)d subtitles to project "%(name)s"') % fields)
 126         util.set_cursor_normal(self.application.window)

  33 def _create_backup(path, bak_path):
  34     """Create a temporary backup file.
  35 
  36     Return success (True or False).
  37     """
  38     try:
  39         shutil.copyfile(path, bak_path)
  40         return True
  41     except IOError, (no, message):
  42         print "Failed to create temporary backup file '%s': %s." % (
  43             bak_path, message)
  44         return False
  45 
  46 def _remove_backup(bak_path):
  47     """Remove the temporary backup file.
  48 
  49     Return success (True or False).
  50     """
  51     try:
  52         os.remove(bak_path)
  53         return True
  54     except OSError, (no, message):
  55         print "Failed to remove temporary backup file '%s': %s." % (
  56             bak_path, message)
  57         return False
  58 
  59 def _remove_failed(path):
  60     """Remove the empty file after failing to write it.
  61 
  62     Return success (True or False).
  63     """
  64     try:
  65         os.remove(path)
  66         return True
  67     except OSError, (no, message):
  68         print "Failed to remove file '%s' after failing to write it: %s." % (
  69             path, message)
  70         return False
  71 
  72 def _restore_original(path, bak_path):
  73     """Restore the file from the temporary backup after failing writing.
  74 
  75     Return success (True or False).
  76     """
  77     try:
  78         shutil.move(bak_path, path)
  79         return True
  80     except (IOError, OSError), (no, message):
  81         print "Failed to restore file '%s' from temporary backup file '%s' " \
            "after failing to write it: %s." % (path, bak_path, message)
  82         return False

  85 class SaveAgent(Delegate):
  86 
  87     """Saving subtitle files."""
  88 
  89     # pylint: disable-msg=E0203,W0201
  90 
  91     def _write_file(self, file, texts):
  92         """Write subtitle file.
  93 
  94         Raise IOError if writing fails.
  95         Raise UnicodeError if encoding fails.
  96         """
  97         positions = self.get_positions(file.mode)
  98         shows = list(positions[i][SHOW] for i in range(len(positions)))
  99         hides = list(positions[i][HIDE] for i in range(len(positions)))
 100 
 101         file_existed = os.path.isfile(file.path)
 102         if file_existed:
 103             bak_path = tempfile.mkstemp(suffix=".bak", prefix="gaupol.")[1]
 104             bak_success = _create_backup(file.path, bak_path)
 105         try:
 106             write_success = False
 107             file.write(shows, hides, texts)
 108             write_success = True
 109         finally:
 110             if write_success:
 111                 if file_existed and bak_success:
 112                     _remove_backup(bak_path)
 113             elif not file_existed:
 114                 _remove_failed(file.path)
 115             elif bak_success:
 116                 _restore_original(file.path, bak_path)
 117 
 118     def save(self, doc, props=None, keep_changes=True):
 119         """Save subtitle file.
 120 
 121         props should be a sequence of path, format, encoding, newline.
 122         Raise IOError if writing fails.
 123         Raise UnicodeError if encoding fails.
 124         """
 125         file = self.get_file(doc)
 126         path, format, encoding, newline = props or [None] * 4
 127 
 128         # Convert tags if saving in different format.
 129         updated_rows = []
 130         texts = self.get_texts(doc)[:]
 131         if file is not None and format is not None:
 132             if file.format != format:
 133                 converter = TagConverter(file.format, format)
 134                 for i, text in enumerate(texts):
 135                     new = converter.convert(text)
 136                     if new != text:
 137                         texts[i] = new
 138                         updated_rows.append(i)
 139 
 140         # Create file if needed.
 141         if not None in (path, format, encoding, newline):
 142             new_file = eval(format.class_name)(path, encoding, newline)
 143             if file is not None and file.format == format:
 144                 # Copy header and mode to new file.
 145                 new_file.header = file.header
 146                 if format == cons.FORMAT.MPSUB:
 147                     new_file.mode = file.mode
 148             file = new_file
 149 
 150         self._write_file(file, texts)
 151 
 152         if keep_changes:
 153             if doc == cons.DOCUMENT.MAIN:
 154                 self.main_file = file
 155                 self.main_texts = texts
 156                 self.main_changed = 0
 157                 self.emit("main-texts-changed", updated_rows)
 158             elif doc == cons.DOCUMENT.TRAN:
 159                 self.tran_file = file
 160                 self.tran_texts = texts
 161                 self.tran_changed = 0
 162                 self.tran_active = True
 163                 self.emit("translation-texts-changed", updated_rows)
 164 
 165         signal = ("main-file-saved", "translation-file-saved")[doc]
 166         self.emit(signal, file)

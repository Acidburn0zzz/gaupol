  37     def expand_frames(self, show, hide):
  38         """Expand single subtitle position data to all items.
  39 
  40         Return time, frame.
  41         """
  42         frame = [show, hide, self.calc.get_frame_duration(show, hide)]
  43         time = list(self.calc.frame_to_time(x) for x in (show, hide))
  44         time.append(self.calc.get_time_duration(time[0], time[1]))
  45         return time, frame
  46 
  47     def expand_positions(self, show, hide):
  48         """Expand single subtitle position data to all items.
  49 
  50         Return time, frame.
  51         """
  52         if isinstance(show, basestring):
  53             return self.expand_times(show, hide)
  54         if isinstance(show, int):
  55             return self.expand_frames(show, hide)
  56         if isinstance(show, float):
  57             return self.expand_seconds(show, hide)
  58         raise ValueError
  59 
  60     def expand_seconds(self, show, hide):
  61         """Expand single subtitle position data to all items.
  62 
  63         Return time, frame.
  64         """
  65         time = list(self.calc.seconds_to_time(x) for x in (show, hide))
  66         time.append(self.calc.get_time_duration(time[0], time[1]))
  67         frame = list(self.calc.seconds_to_frame(x) for x in (show, hide))
  68         frame.append(self.calc.get_frame_duration(frame[0], frame[1]))
  69         return time, frame
  70 
  71     def expand_times(self, show, hide):
  72         """Expand single subtitle position data to all items.
  73 
  74         Return time, frame.
  75         """
  76         time = [show, hide, self.calc.get_time_duration(show, hide)]
  77         frame = list(self.calc.time_to_frame(x) for x in (show, hide))
  78         frame.append(self.calc.get_frame_duration(frame[0], frame[1]))
  79         return time, frame

  81     def get_file(self, doc):
  82         """Get the file corresponding to document."""
  83 
  84         return (self.main_file, self.tran_file)[doc]
  85 
  86     def get_format_class_name(self, doc):
  87         """Get the class name of document's file format or None."""
  88 
  89         if doc == cons.DOCUMENT.MAIN:
  90             try:
  91                 return self.main_file.format.class_name
  92             except AttributeError:
  93                 return None
  94         if doc == cons.DOCUMENT.TRAN:
  95             try:
  96                 return self.tran_file.format.class_name
  97             except AttributeError:
  98                 return self.get_format_class_name(cons.DOCUMENT.MAIN)
  99 
 100     def get_line_lengths(self, row, doc):
 101         """Get a list of line lengths in text without tags."""
 102 
 103         text = self.get_texts(doc)[row]
 104         re_tag = self.get_tag_regex(doc)
 105         if re_tag is not None:
 106             text = re_tag.sub("", text)
 107         return list(len(x) for x in text.split("\n"))
 108 
 109     def get_mode(self):
 110         """Get the mode of the main file or default."""
 111 
 112         try:
 113             return self.main_file.mode
 114         except AttributeError:
 115             return cons.MODE.TIME
 116 
 117     def get_positions(self, mode=None):
 118         """Get either times or frames depending the mode."""
 119 
 120         if mode is None:
 121             mode = self.get_mode()
 122         if mode == cons.MODE.TIME:
 123             return self.times
 124         if mode == cons.MODE.FRAME:
 125             return self.frames
 126         raise ValueError
 127 
 128     def get_tag_regex(self, doc):
 129         """Get the regular expression for a tag in document or None."""
 130 
 131         name = self.get_format_class_name(doc)
 132         if name is None:
 133             return None
 134         cls = eval(name)
 135         if cls.tag is None:
 136             return None
 137         return re.compile(*cls.tag)
 138 
 139     def get_text_length(self, row, doc):
 140         """Get the legth of text without tags."""
 141 
 142         text = self.get_texts(doc)[row]
 143         re_tag = self.get_tag_regex(doc)
 144         if re_tag is not None:
 145             text = re_tag.sub("", text)
 146         return len(text)
 147 
 148     def get_texts(self, doc):
 149         """Get the texts corresponding to document."""
 150 
 151         return (self.main_texts, self.tran_texts)[doc]
 152 
 153     @revertable
 154     @notify_frozen
 155     def insert_subtitles(
 156         self, rows, times, frames, main_texts, tran_texts, register=-1):
 157         """Insert subtitles.
 158 
 159         Subtitles are inserted in ascending order by simply inserting items of
 160         data in positions defined by items of rows. This means that the
 161         addition of subtitles must be taken into account beforehand in the
 162         'rows' argument. Data is not resorted, so this method must be called
 163         with ordered positions.
 164         """
 165         for i, row in enumerate(rows):
 166             self.times.insert(row, times[i])
 167             self.frames.insert(row, frames[i])
 168             self.main_texts.insert(row, main_texts[i])
 169             self.tran_texts.insert(row, tran_texts[i])
 170 
 171         self.register_action(
 172             register=register,
 173             docs=[cons.DOCUMENT.MAIN, cons.DOCUMENT.TRAN],
 174             description=_("Inserting subtitles"),
 175             revert_method=self.remove_subtitles,
 176             revert_args=[sorted(rows)],)
 177 
 178         self.emit("subtitles-inserted", rows)
 179 
 180     @revertable
 181     def replace_both_texts(self, rows, new_texts, register=-1):
 182         """Replace texts in both documents' rows with new_texts.
 183 
 184         rows should be a list of main rows, translation rows.
 185         new_texts should be a list of main texts, translation texts.
 186         """
 187         if not rows[0] and not rows[1]:
 188             return
 189         main_args = [rows[0], cons.DOCUMENT.MAIN, new_texts[0]]
 190         tran_args = [rows[1], cons.DOCUMENT.TRAN, new_texts[1]]
 191         kwargs = {"register": register}
 192         if not rows[1]:
 193             return self.replace_texts(*main_args, **kwargs)
 194         if not rows[0]:
 195             return self.replace_texts(*tran_args, **kwargs)
 196         self.replace_texts(*main_args, **kwargs)
 197         self.replace_texts(*tran_args, **kwargs)
 198         self.group_actions(register, 2, _("Replacing texts"))
 199 
 200     @revertable
 201     @notify_frozen
 202     def replace_positions(self, rows, new_times, new_frames, register=-1):
 203         """Replace times and frames in rows with new_times and new_frames."""
 204 
 205         orig_times = []
 206         orig_frames = []
 207         for i, row in enumerate(rows):
 208             orig_times.append(self.times[row])
 209             orig_frames.append(self.frames[row])
 210             self.times[row] = new_times[i]
 211             self.frames[row] = new_frames[i]
 212 
 213         self.register_action(
 214             register=register,
 215             docs=[cons.DOCUMENT.MAIN, cons.DOCUMENT.TRAN],
 216             description=_("Replacing positions"),
 217             revert_method=self.replace_positions,
 218             revert_args=[rows, orig_times, orig_frames],)
 219 
 220         self.emit("positions-changed", rows)
 221 
 222     @revertable
 223     @notify_frozen
 224     def replace_texts(self, rows, doc, new_texts, register=-1):
 225         """Replace texts in document's rows with new_texts."""
 226 
 227         orig_texts = []
 228         texts = self.get_texts(doc)
 229         for i, row in enumerate(rows):
 230             orig_texts.append(texts[row])
 231             texts[row] = new_texts[i]
 232 
 233         self.register_action(
 234             register=register,
 235             docs=[doc],
 236             description=_("Replacing texts"),
 237             revert_method=self.replace_texts,
 238             revert_args=[rows, doc, orig_texts],)
 239 
 240         signal = ("main-texts-changed", "translation-texts-changed")[doc]
 241         self.emit(signal, rows)

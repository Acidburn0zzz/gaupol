  32 class MenuAgent(Delegate):
  33 
  34     """Building and updating menus.
  35 
  36     Instance attributes:
  37 
  38         _projects_id: UI manager merge ID for the projects menu
  39     """
  40 
  41     # pylint: disable-msg=E0203,W0201
  42 
  43     def __init__(self, master):
  44 
  45         Delegate.__init__(self, master)
  46 
  47         self._projects_id = None
  48 
  49     def _get_action_group(self, name):
  50         """Get action group from the UI manager.
  51 
  52         name should be 'main' or 'projects'.
  53         """
  54         for group in self.uim.get_action_groups():
  55             if group.get_name() == name:
  56                 return group
  57         raise ValueError
  58 
  59     def _get_project_actions(self):
  60         """Get a list of UI manager actions for the projects menu."""
  61 
  62         radio_actions = []
  63         for i in range(len(self.pages)):
  64             name = "activate_project_%d" % i
  65             basename  = self.pages[i].get_main_basename()
  66             label = self.pages[i].tab_label.get_text()
  67             label = "%d. %s" % (i + 1, label)
  68             if len(label) > 60:
  69                 label = label[:60] + "..."
  70             label = label.replace("_", "__")
  71             if i < 9:
  72                 label = "_" + label
  73             tip = _('Activate "%s"') % basename
  74             radio_actions.append((name, None, label, None, tip, i))
  75         return radio_actions
  76 
  77     def _get_project_ui_string(self):
  78         """Get a UI manager layout string for the projects menu."""
  79 
  80         ui = ""
  81         for i in range(len(self.pages)):
  82             ui += '<menuitem name="%d" action="activate_project_%d"/>' % (i, i)
  83         ui = """
  84         <ui>
  85           <menubar>
  86             <menu name="projects" action="show_projects_menu">
  87               <placeholder name="open">%s</placeholder>
  88             </menu>
  89           </menubar>
  90         </ui>""" % ui
  91         return ui
  92 
  93     def _get_recent_menu(self, doc):
  94         """Return a new gtk.RecentChooserMenu."""
  95 
  96         menu = gtk.RecentChooserMenu(self.recent_manager)
  97         menu.set_show_not_found(False)
  98         menu.set_show_tips(False)
  99         menu.set_sort_type(gtk.RECENT_SORT_MRU)
 100 
 101         group = ("gaupol-main", "gaupol-translation")[doc]
 102         recent_filter = gtk.RecentFilter()
 103         recent_filter.add_group(group)
 104         menu.add_filter(recent_filter)
 105         menu.set_filter(recent_filter)
 106         menu.set_data("group", group)
 107 
 108         self._update_recent_menu(menu)
 109         return menu
 110 
 111     def _show_revert_button_menu(self, register):
 112         """Show the undo or redo button menu."""
 113 
 114         menu_items = []
 115         menu = gtk.Menu()
 116         page = self.get_current_page()
 117         if register == cons.REGISTER.UNDO:
 118             stack = page.project.undoables
 119             button = self.undo_button
 120             revert_method = self.undo
 121             get_tip = lambda desc: _('Undo "%s"') % desc
 122         elif register == cons.REGISTER.REDO:
 123             stack = page.project.redoables
 124             button = self.redo_button
 125             revert_method = self.redo
 126             get_tip = lambda desc: _('Redo "%s"') % desc
 127 
 128         def on_activate(menu_item, index):
 129             revert_method(index + 1)
 130         def on_enter(menu_item, event, index, tip):
 131             for i in range(index):
 132                 menu_items[i].set_state(gtk.STATE_PRELIGHT)
 133             self.push_message(tip, False)
 134         def on_leave(menu_item, event, index):
 135             for i in range(index):
 136                 menu_items[i].set_state(gtk.STATE_NORMAL)
 137             self.push_message(None)
 138 
 139         for i, action in enumerate(stack):
 140             tip = get_tip(action.description)
 141             menu_item = gtk.MenuItem(action.description, False)
 142             menu_item.connect("activate", on_activate, i)
 143             menu_item.connect("enter-notify-event", on_enter, i, tip)
 144             menu_item.connect("leave-notify-event", on_leave, i)
 145             menu_items.append(menu_item)
 146             menu.append(menu_item)
 147         menu.show_all()
 148         button.set_menu(menu)
 149 
 150     def _update_recent_menu(self, menu):
 151         """Update recent menu item limit."""
 152 
 153         if conf.file.max_recent < 1:
 154             return menu.set_limit(0)
 155         i = -1
 156         matches = 0
 157         group = menu.get_data("group")
 158         menu.set_limit(i)
 159         # pylint: disable-msg=W0631
 160         for i, item in enumerate(menu.get_items()):
 161             if not item.has_group(group):
 162                 continue
 163             path = util.uri_to_path(item.get_uri())
 164             if not os.path.isfile(path):
 165                 continue
 166             matches += 1
 167             if matches >= conf.file.max_recent:
 168                 break
 169         return menu.set_limit(i + 1)

 171     def on_open_button_show_menu(self, *args):
 172         """Build and attach a new recent menu on the open button."""
 173 
 174         menu = self._get_recent_menu(cons.DOCUMENT.MAIN)
 175         method = self.on_recent_main_menu_item_activated
 176         menu.connect("item-activated", method)
 177         self.open_button.set_menu(menu)
 178         while gtk.events_pending():
 179             gtk.main_iteration()
 180 
 181     def on_page_tab_widget_button_press_event(self, button, event):
 182         """Display a tab pop-up menu."""
 183 
 184         if event.button == 3:
 185             menu = self.uim.get_widget("/ui/tab_popup")
 186             menu.popup(None, None, None, event.button, event.time)
 187 
 188     def on_redo_button_show_menu(self, *args):
 189         """Show the redo button menu."""
 190 
 191         self._show_revert_button_menu(cons.REGISTER.REDO)

 193     def on_show_projects_menu_activate(self, *args):
 194         """Add all open projects to the projects menu."""
 195 
 196         action_group = self._get_action_group("projects")
 197         for action in action_group.list_actions():
 198             action_group.remove_action(action)
 199         if self._projects_id is not None:
 200             self.uim.remove_ui(self._projects_id)
 201         page = self.get_current_page()
 202         if page is None:
 203             return
 204 
 205         def switch(item, active_item):
 206             index = int(active_item.get_name().split("_")[-1])
 207             self.notebook.set_current_page(index)
 208         radio_actions = self._get_project_actions()
 209         active = self.notebook.get_current_page()
 210         action_group.add_radio_actions(radio_actions, active, switch)
 211         ui = self._get_project_ui_string()
 212         self._projects_id = self.uim.add_ui_from_string(ui)
 213 
 214         for i in range(len(self.pages)):
 215             path = "/ui/menubar/projects/open/%d" % i
 216             action = self.uim.get_action(path)
 217             widget = self.uim.get_widget(path)
 218             action.connect_proxy(widget)
 219         self.set_menu_notify_events("projects")
 220

 248     def set_menu_notify_events(self, name):
 249         """Set statusbar tooltips for menu items.
 250 
 251         name should be 'main' or 'projects'.
 252         """
 253         def on_enter(menu_item, event, action):
 254             self.push_message(action.props.tooltip, False)
 255         def on_leave(menu_item, event, action):
 256             self.push_message(None)
 257 
 258         for action in self._get_action_group(name).list_actions():
 259             for widget in action.get_proxies():
 260                 widget.connect("enter-notify-event", on_enter, action)
 261                 widget.connect("leave-notify-event", on_leave, action)

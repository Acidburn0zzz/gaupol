  54     GTK user interface controller.
  55 
  56     Instance variables:

  58         _delegations:    Dictionary mapping method names to Agents
  59         clipboard        Application-wide Clipboard
  60         counter:         Integer for naming unsaved projects
  61         framerate_combo: gtk.ComboBox
  62         notebook:        gtk.Notebook
  63         open_button:     gtk.MenuToolButton
  64         output_window:   OutputWindow
  65         pages:           List of Pages
  66         recent_manager:  gtk.RecentManager
  67         redo_button:     gtk.MenuToolButton
  68         static_tooltips: gtk.Tooltips, enabled always
  69         statusbar:       gtk.Statusbar
  70         tooltips:        gtk.Tooltips, enabled if a project is open

  72         undo_button:     gtk.MenuToolButton
  73         video_button:    gtk.Button
  74         video_toolbar:   gtk.Toolbar
  75         window:          gtk.Window
  76         x_clipboard:     gtk.Clipboard (X clipboard)
  77 
  78     Signals:
  79 
  80         page-added (application, page)
  81         page-changed (application, page)
  82         page-closed (application, page)
  83         pages-reordered (application, page, number)

 126     def _init_delegations(self):
 127         """Initialize the delegate mappings."""
 128 
 129         for agent in (eval(x)(self) for x in AGENTS):
 130             for attr_name in (x for x in dir(agent) if not x.startswith("_")):
 131                 attr = getattr(agent, attr_name)
 132                 if type(attr) is types.MethodType:
 133                     if attr_name in self._delegations:
 134                         raise ValueError
 135                     self._delegations[attr_name] = agent

 160     def _init_gui(self):
 161         """Initialize the user interface."""
 162 
 163         vbox = gtk.VBox()
 164         self._init_window()
 165         self._init_uim()
 166         self._init_menubar(vbox)
 167         self._init_main_toolbar(vbox)
 168         self._init_recent_menus()
 169         self._init_notebook(vbox)
 170         self._init_video_toolbar(vbox)
 171         self._init_statusbar(vbox)
 172         self._init_output_window()
 173         self.window.add(vbox)
 174 
 175         vbox.show_all()
 176         if not conf.application_window.show_main_toolbar:
 177             self.uim.get_widget("/ui/main_toolbar").hide()
 178         if not conf.application_window.show_video_toolbar:
 179             self.video_toolbar.hide()
 180         if not conf.application_window.show_statusbar:
 181             self.statusbar.hide()
 182 
 183         self.set_menu_notify_events("main")
 184         self.update_gui()
 185         self.window.show()

 307     def _init_uim_actions(self, action_group):
 308         """Initialize the UI manager actions."""
 309 
 310         def get_method(name):
 311             name = "on_" + name.replace("-", "_") + "_activate"
 312             return getattr(self, name)
 313 
 314         for class_name in ACTIONS:
 315             cls = eval(class_name)
 316             if cls.menu_item is not None:
 317                 item = list(cls.menu_item)
 318                 item.append(get_method(item[0]))
 319                 action_group.add_actions([tuple(item)], None)
 320             if cls.action_item is not None:
 321                 item = list(cls.action_item)
 322                 item.append(get_method(item[0]))
 323                 action_group.add_actions([tuple(item)], None)
 324             if cls.toggle_item is not None:
 325                 item = list(cls.toggle_item)
 326                 item.insert(5, get_method(item[0]))
 327                 action_group.add_toggle_actions([tuple(item)], None)
 328             if cls.radio_items is not None:
 329                 items, value = cls.radio_items
 330                 method = get_method(items[0][0])
 331                 action_group.add_radio_actions(items, value, method)

 410     def _init_window(self):
 411         """Initialize the window."""
 412 
 413         self.window = gtk.Window()
 414         self.window.resize(*conf.application_window.size)
 415         self.window.move(*conf.application_window.position)
 416         if conf.application_window.maximized:
 417             self.window.maximize()
 418         util.connect(self, "window", "delete-event")
 419         util.connect(self, "window", "window-state-event")
 420 
 421         icon_theme = gtk.icon_theme_get_default()
 422         if icon_theme.has_icon("gaupol"):
 423             self.window.set_icon_name("gaupol")
 424             gtk.window_set_default_icon_name("gaupol")
 425         if self.window.get_icon() is None:
 426             gtk.window_set_default_icon_from_file(_ICON_FILE)

 428     def get_current_page(self):
 429         """Get the currently active Page or None."""
 430 
 431         index = self.notebook.get_current_page()
 432         if index >= 0:
 433             return self.pages[index]
 434         return None

  19 """MPL2 tag library.
  20 
  21 Module variables:
  22 
  23     _FLAGS: Common regular expression flags
  24 """
  25 
  26 
  27 import re
  28 
  29 from ._taglib import TagLibrary
  30 from .microdvd import MicroDVD
  31 
  32 
  33 _FLAGS = re.MULTILINE | re.DOTALL
  34 
  35 
  36 class MPL2(TagLibrary):
  37 
  38     """MPL2 tag library.
  39 
  40     Class variables:
  41 
  42         _style_tags: List of tuples of style tags
  43     """
  44 
  45     tag = r"(\{[a-z]:.*?\})|(\\|/|_)", re.IGNORECASE
  46     italic_tag = r"(\{y:i\})|(/)", re.IGNORECASE
  47 
  48     decode_tags = [
  49         # Italic (single line)
  50         (r"/(.*?)$", _FLAGS,
  51             r"<i>\1</i>"),
  52         # Bold (single line)
  53         (r"\\(.*?)$", _FLAGS,
  54             r"<b>\1</b>"),
  55         # Underline (single line)
  56         (r"_(.*?)$", _FLAGS,
  57             r"<u>\1</u>"),
  58         # Remove duplicate style tags (e.g. <b>test</b><b>test</b>).
  59         (r"</(b|i|u)>(\n?)<\1>", _FLAGS,
  60             r"\2", 3)
  61     ] + MicroDVD.decode_tags
  62 
  63     _style_tags = [
  64         ("<i>", "</i>", "/" ),
  65         ("<b>", "</b>", "\\"),
  66         ("<u>", "</u>", "_" ),]
  67 
  68     @classmethod
  69     def pre_encode(cls, text):
  70         """Convert style tags."""
  71 
  72         for opening, closing, replacement in cls._style_tags:
  73             opening_lenght = len(opening)
  74             closing_length = len(closing)
  75             while True:
  76                 if not opening in text:
  77                     break
  78                 a = text.index(opening)
  79                 z = (text.index(closing) if closing in text else len(text))
  80                 before = text[:a]
  81                 middle = text[a + opening_lenght:z]
  82                 after = text[z + closing_length:]
  83                 lines = middle.split("\n")
  84                 for i in range(1, len(lines)):
  85                     if not lines[i].startswith(replacement):
  86                         lines[i] = replacement + lines[i]
  87                 middle = "\n".join(lines)
  88                 text = before + replacement + middle + after
  89         return text
  90 
  91     encode_tags = MicroDVD.encode_tags
  92 
  93     @classmethod
  94     def italicize(cls, text):
  95         """Italicize text."""
  96 
  97         return "/" + text.replace("\n", "\n/")

  19 """MPL2 file."""
  20 
  21 
  22 import codecs
  23 import re
  24 
  25 from gaupol import cons
  26 from gaupol.calculator import Calculator
  27 from ._subfile import SubtitleFile
  28 
  29 
  30 class MPL2(SubtitleFile):
  31 
  32     """MPL2 file."""
  33 
  34     format = cons.FORMAT.MPL2
  35     has_header = False
  36     identifier = r"^\[\d+\]\[\d+\].*?$", 0
  37     mode = cons.MODE.TIME
  38 
  39     def read(self):
  40         """Read file.
  41 
  42         Raise IOError if reading fails.
  43         Raise UnicodeError if decoding fails.
  44         Return show times, hide times, texts.
  45         """
  46         re_line = re.compile(r"^\[(\d+)\]\[(\d+)\](.*?)$")
  47         calc = Calculator()
  48 
  49         shows = []
  50         hides = []
  51         texts = []
  52         for line in self._read_lines():
  53             match = re_line.match(line)
  54             if match is not None:
  55                 shows.append(match.group(1))
  56                 hides.append(match.group(2))
  57                 texts.append(match.group(3))
  58 
  59         for seq in (shows, hides):
  60             for i, value in enumerate(seq):
  61                 seconds = float(value[:-1] + "." + value[-1])
  62                 seq[i] = calc.seconds_to_time(seconds)
  63         texts = list(x.replace("|", "\n") for x in texts)
  64         return shows, hides, texts
  65 
  66     def write(self, shows, hides, texts):
  67         """Write file.
  68 
  69         Raise IOError if writing fails.
  70         Raise UnicodeError if encoding fails.
  71         """
  72         calc = Calculator()
  73         def get_deca(time):
  74             return "%.0f" % (calc.time_to_seconds(time) * 10)
  75         shows = list(get_deca(x) for x in shows)
  76         hides = list(get_deca(x) for x in hides)
  77         texts = list(x.replace("\n", "|") for x in texts)
  78 
  79         fobj = codecs.open(self.path, "w", self.encoding)
  80         try:
  81             for i in range(len(shows)):
  82                 fobj.write("[%s][%s]%s%s" % (
  83                     shows[i], hides[i], texts[i], self.newline.value))
  84         finally:
  85             fobj.close()

  19 """Configuration.
  20 
  21 Module variables:
  22 
  23     _CONFIG: Instance of Config used
  24 
  25 The configuration file is read and validated upon import. All configuration
  26 sections are made available as module variables. See the spec file for what
  27 these sections and options are.
  28 """
  29 
  30 
  31 import os
  32 
  33 from gaupol import util, __version__
  34 from gaupol.gtk import paths
  35 from gaupol.gtk.errors import ConfigParseError
  36 from .wrappers import Config, Container
  37 
  38 
  39 __all__ = ["Config", "Container"]
  40 
  41 _CONFIG = None
  42 
  43 
  44 def _translate_nones(config):
  45     """Translate Nones to a False value of proper type."""
  46 
  47     options = (
  48         ("editor" , "font"          , ""),
  49         ("file"   , "directory"     , ""),
  50         ("file"   , "encoding"      , ""),
  51         ("preview", "custom_command", ""),)
  52     for section, option, value in options:
  53         if config[section][option] is None:
  54             config[section][option] = value
  55             config[section].defaults.append(option)
  56 
  57 def connect(obj, section, option):
  58     """Connect option's signal to object's callback method."""
  59 
  60     signal = "notify::%s" % option
  61     method_name = "_on_conf_%s_%s" % (section, signal.replace("::", "_"))
  62     if not hasattr(obj, method_name):
  63         method_name = method_name[1:]
  64     method = getattr(obj, method_name)
  65     eval(section).connect(signal, method)
  66 
  67 def read():
  68     """Read configurations from file."""
  69 
  70     try:
  71         config = Config(paths.CONFIG_FILE, paths.SPEC_FILE)
  72     except ConfigParseError:
  73         raise SystemExit(1)
  74     version = config["general"]["version"]
  75     if version is not None:
  76         if util.compare_versions(version, "0.7.999") == -1:
  77             print "Ignoring old-style configuration file entirely."
  78             config = Config(None, paths.SPEC_FILE)
  79     config["general"]["version"] = __version__
  80     _translate_nones(config)
  81 
  82     globals()["_CONFIG"] = config
  83     for key, value in config.items():
  84         globals()[key] = Container(value)
  85 
  86 def write():
  87     """Write configurations to file."""
  88 
  89     # pylint: disable-msg=E1101
  90     _CONFIG.filename = paths.CONFIG_FILE
  91     _CONFIG.write_to_file()
  92 
  93 
  94 read()

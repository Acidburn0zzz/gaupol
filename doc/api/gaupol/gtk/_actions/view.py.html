  28 class ShowColumnsMenuAction(UIMAction):
  29 
  30     """Show the columns view menu."""
  31 
  32     menu_item = (
  33         "show_columns_menu",
  34         None,
  35         _("_Columns"),
  36         None,
  37         None,)
  38 
  39     paths = ["/ui/menubar/view/columns"]
  40 
  41     @classmethod
  42     def is_doable(cls, application, page):
  43         """Return True if action can be done."""
  44 
  45         if page is not None:
  46             return (page.project.main_file is not None)
  47         return False

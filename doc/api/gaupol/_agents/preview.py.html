  19 """Previewing subtitles with a video player.
  20 
  21 Module variables:
  22 
  23     _EXTENSIONS: List of video file extensions used for guessing
  24 """
  25 
  26 
  27 import os
  28 import string
  29 import subprocess
  30 import tempfile
  31 
  32 from gaupol import cons, util
  33 from gaupol.base import Delegate
  34 
  35 
  36 _EXTENSIONS = (
  37     ".3ivx", ".asf", ".avi", ".dat", ".divx", ".m2v", ".mkv", ".mov", ".mp4",
  38     ".mpeg", ".mpg", ".ogg", ".ogm", ".qt", ".rm", ".rmvb", ".swf", ".vob",
  39     ".wmv",)
  40 
  41 
  42 class PreviewAgent(Delegate):
  43 
  44     """Previewing subtitles with a video player."""
  45 
  46     # pylint: disable-msg=E0203,W0201
  47 
  48     def __init__(self, master):
  49 
  50         Delegate.__init__(self, master)
  51 
  52         util.connect(self, self, "notify::main_file")
  53 
  54     def _get_subtitle_path(self, doc):
  55         """Save the subtitle data to a temporary file if needed.
  56 
  57         Raise IOError if writing to temporary file fails.
  58         Raise UnicodeError if encoding temporary file fails.
  59         Return subtitle file path, True if file is temporary.
  60         """
  61         if doc == cons.DOCUMENT.MAIN:
  62             if not self.main_changed:
  63                 return self.main_file.path, False
  64         if doc == cons.DOCUMENT.TRAN:
  65             if not self.tran_active or not self.tran_changed:
  66                 return self.tran_file.path, False
  67         return self.get_temp_file_path(doc), True
  68 
  69     def _on_notify_main_file(self, *args):
  70         """Guess the video file path if it is unset."""
  71 
  72         if not self.video_path:
  73             self.guess_video_path()
  74 
  75     def get_temp_file_path(self, doc):
  76         """Save the subtitle data to a temporary file and return path.
  77 
  78         Raise IOError if writing to temporary file fails.
  79         Raise UnicodeError if encoding temporary file fails.
  80         """
  81         file = self.get_file(doc)
  82         extension = file.format.extension
  83         path = tempfile.mkstemp(prefix="gaupol.", suffix=extension)[1]
  84         props = (path, file.format, file.encoding, file.newline)
  85         self.save(doc, props, False)
  86         return path
  87 
  88     def guess_video_path(self, extensions=_EXTENSIONS):
  89         """Guess the video file path based on main file's path.
  90 
  91         Video file is searched for in the same directory as the subtitle file.
  92         Subtitle file's filename without extension is assumed to start with or
  93         match video file's filename without extension.
  94         Return the video file path or None.
  95         """
  96         if self.main_file is None:
  97             return None
  98         subroot = os.path.splitext(self.main_file.path)[0]
  99         dirname = os.path.dirname(self.main_file.path)
 100         for filename in os.listdir(dirname):
 101             filepath = os.path.join(dirname, filename)
 102             fileroot, extension = os.path.splitext(filepath)
 103             if (extension in extensions) and subroot.startswith(fileroot):
 104                 self.video_path = filepath
 105                 return filepath
 106         return None
 107 
 108     def preview(self, time, doc, command, offset, temp_path=None):
 109         """Preview subtitles with a video player.
 110 
 111         command should have variables $SECONDS, $SUBFILE and $VIDEOFILE.
 112         offset is the amount of seconds before time to start.
 113         temp_path should be a temporary subtitle file path if already written.
 114 
 115         Raise IOError if writing to temporary file fails.
 116         Raise UnicodeError if encoding temporary file fails.
 117         Return PID, command, path to output file, path to temporary file.
 118         """
 119         sub_path = temp_path
 120         is_temp = temp_path is not None
 121         if temp_path is None:
 122             sub_path, is_temp = self._get_subtitle_path(doc)
 123         temp_path = (sub_path if is_temp else None)
 124 
 125         output_fd, output_path = tempfile.mkstemp(
 126             prefix="gaupol.", suffix=".output")
 127         seconds = self.calc.time_to_seconds(time)
 128         seconds = "%.3f" % max(0.0, seconds - float(offset))
 129         command = string.Template(command).safe_substitute(
 130             SECONDS=seconds,
 131             SUBFILE=util.shell_quote_path(sub_path),
 132             VIDEOFILE=util.shell_quote_path(self.video_path))
 133 
 134         process = util.start_process(
 135             command, stderr=subprocess.STDOUT, stdout=output_fd)
 136         return process.pid, command, output_path, temp_path

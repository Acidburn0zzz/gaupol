  37     @revertable
  38     def adjust_durations(self, rows=None, optimal=None, lengthen=False,
  39         shorten=False, maximum=None, minimum=None, gap=None, register=-1):
  40         """Adjust durations.
  41 
  42         rows can be None to process all rows.
  43         optimal and gap should be in seconds.
  44         Return the adjusted rows.
  45         """
  46         new_rows = []
  47         new_times = []
  48         new_frames = []
  49         rows = rows or range(len(self.times))
  50         for row in rows:
  51             show = self.calc.time_to_seconds(self.times[row][SHOW])
  52             hide = self.calc.time_to_seconds(self.times[row][HIDE])
  53             hide_max = 360000
  54             if row < len(self.times) - 1:
  55                 hide_max = self.calc.time_to_seconds(self.times[row + 1][SHOW])
  56             if optimal is not None:
  57                 length = self.get_text_length(row, cons.DOCUMENT.MAIN)
  58                 optimal_durn = optimal * length
  59                 if hide - show < optimal_durn and lengthen:
  60                     hide = show + optimal_durn
  61                 if hide - show > optimal_durn and shorten:
  62                     hide = show + optimal_durn
  63             if minimum is not None and hide - show < minimum:
  64                 hide = show + minimum
  65             if maximum is not None and hide - show > maximum:
  66                 hide = show + maximum
  67             if gap is not None and (hide_max - hide) < gap:
  68                 hide = max(show, hide_max - gap)
  69             if self.calc.seconds_to_time(hide) == self.times[row][HIDE]:
  70                 continue
  71             time, frame = self.expand_seconds(show, hide)
  72             new_rows.append(row)
  73             new_times.append(time)
  74             new_frames.append(frame)
  75 
  76         if new_rows:
  77             self.replace_positions(
  78                 new_rows, new_times, new_frames, register=register)
  79             self.set_action_description(register, _("Adjusting durations"))
  80         return new_rows
  81 
  82     @revertable
  83     def adjust_frames(self, rows, point_1, point_2, register=-1):
  84         """Adjust times and frames in rows.
  85 
  86         rows can be None to process all rows.
  87         point_N should be a tuple of row and new show frame.
  88         """
  89         # Think of this as a linear transformation where current times are
  90         # located on the x-axis and correct times on the y-axis.
  91         x_1 = self.frames[point_1[0]][SHOW]
  92         x_2 = self.frames[point_2[0]][SHOW]
  93         y_1 = point_1[1]
  94         y_2 = point_2[1]
  95         coefficient = (y_2 - y_1) / (x_2 - x_1)
  96         constant = - coefficient * x_1 + y_1
  97         def get_correct(current):
  98             return max(0, int(round(coefficient * current + constant, 0)))
  99 
 100         new_times = []
 101         new_frames = []
 102         rows = rows or range(len(self.times))
 103         for row in rows:
 104             show = get_correct(self.frames[row][SHOW])
 105             hide = get_correct(self.frames[row][HIDE])
 106             time, frame = self.expand_frames(show, hide)
 107             new_times.append(time)
 108             new_frames.append(frame)
 109 
 110         self.replace_positions(rows, new_times, new_frames, register=register)
 111         self.set_action_description(register, _("Adjusting frames"))
 112 
 113     @revertable
 114     def adjust_times(self, rows, point_1, point_2, register=-1):
 115         """Adjust times and frames in rows.
 116 
 117         rows can be None to process all rows.
 118         point_N should be a tuple of row and new show time.
 119         """
 120         # Think of this as a linear transformation where current times are
 121         # located on the x-axis and correct times on the y-axis.
 122         x_1 = self.calc.time_to_seconds(self.times[point_1[0]][SHOW])
 123         x_2 = self.calc.time_to_seconds(self.times[point_2[0]][SHOW])
 124         y_1 = self.calc.time_to_seconds(point_1[1])
 125         y_2 = self.calc.time_to_seconds(point_2[1])
 126         coefficient = (y_2 - y_1) / (x_2 - x_1)
 127         constant = - coefficient * x_1 + y_1
 128         def get_correct(current):
 129             seconds = self.calc.time_to_seconds(current)
 130             seconds = max(0, coefficient * seconds + constant)
 131             return self.calc.seconds_to_time(seconds)
 132 
 133         new_times = []
 134         new_frames = []
 135         rows = rows or range(len(self.times))
 136         for row in rows:
 137             show = get_correct(self.times[row][SHOW])
 138             hide = get_correct(self.times[row][HIDE])
 139             time, frame = self.expand_times(show, hide)
 140             new_times.append(time)
 141             new_frames.append(frame)
 142 
 143         self.replace_positions(rows, new_times, new_frames, register=register)
 144         self.set_action_description(register, _("Adjusting times"))

 146     @notify_frozen
 147     def change_framerate(self, framerate):
 148         """Change the framerate and update positions.
 149 
 150         Only change is to what is assumed to be the video framerate and thus
 151         effect only how non-native position data is calculated. Native
 152         positions will remain unchanged.
 153         """
 154         self.set_framerate(framerate, register=None)
 155         if self.main_file.mode == cons.MODE.TIME:
 156             convert = self.calc.time_to_frame
 157             for i in range(len(self.times)):
 158                 self.frames[i][SHOW] = convert(self.times[i][SHOW])
 159                 self.frames[i][HIDE] = convert(self.times[i][HIDE])
 160                 self.frames[i][DURN] = self.calc.get_frame_duration(
 161                     self.frames[i][SHOW], self.frames[i][HIDE])
 162         elif self.main_file.mode == cons.MODE.FRAME:
 163             convert = self.calc.frame_to_time
 164             for i in range(len(self.times)):
 165                 self.times[i][SHOW] = convert(self.frames[i][SHOW])
 166                 self.times[i][HIDE] = convert(self.frames[i][HIDE])
 167                 self.times[i][DURN] = self.calc.get_time_duration(
 168                     self.times[i][SHOW], self.times[i][HIDE])
 169         self.emit("positions-changed", range(len(self.times)))

 171     @revertable
 172     def convert_framerate(self, rows, current, correct, register=-1):
 173         """Convert and set the framerate.
 174 
 175         rows can be None to process all rows.
 176         """
 177         self.set_framerate(current, register=None)
 178         self.set_framerate(correct)
 179 
 180         new_times = []
 181         new_frames = []
 182         rows = rows or range(len(self.times))
 183         coefficient = correct.value / current.value
 184         for row in rows:
 185             if self.main_file.mode == cons.MODE.TIME:
 186                 show = self.calc.time_to_seconds(self.times[row][SHOW])
 187                 hide = self.calc.time_to_seconds(self.times[row][HIDE])
 188                 time, frame = self.expand_seconds(
 189                     show / coefficient,
 190                     hide / coefficient)
 191             elif self.main_file.mode == cons.MODE.FRAME:
 192                 time, frame = self.expand_frames(
 193                     int(round(coefficient * self.frames[row][SHOW], 0)),
 194                     int(round(coefficient * self.frames[row][HIDE], 0)))
 195             new_times.append(time)
 196             new_frames.append(frame)
 197 
 198         self.replace_positions(rows, new_times, new_frames, register=register)
 199         self.group_actions(register, 2, _("Converting framerate"))
 200 
 201     @revertable
 202     def set_framerate(self, framerate, register=-1):
 203         """Set the framerate."""
 204 
 205         orig_framerate = self.framerate
 206         self.framerate = framerate
 207         self.calc.set_framerate(framerate)
 208 
 209         self.register_action(
 210             register=register,
 211             docs=[cons.DOCUMENT.MAIN, cons.DOCUMENT.TRAN],
 212             description=_("Setting framerate"),
 213             revert_method=self.set_framerate,
 214             revert_args=[orig_framerate],)
 215 
 216     @revertable
 217     def shift_frames(self, rows, count, register=-1):
 218         """Shift times and frames.
 219 
 220         rows can be None to process all rows.
 221         """
 222         new_times = []
 223         new_frames = []
 224         rows = rows or range(len(self.times))
 225         for row in rows:
 226             show = max(0, self.frames[row][SHOW] + count)
 227             hide = max(0, self.frames[row][HIDE] + count)
 228             time, frame = self.expand_frames(show, hide)
 229             new_times.append(time)
 230             new_frames.append(frame)
 231 
 232         self.replace_positions(rows, new_times, new_frames, register=register)
 233         self.set_action_description(register, _("Shifting frames"))
 234 
 235     @revertable
 236     def shift_seconds(self, rows, count, register=-1):
 237         """Shift times and frames.
 238 
 239         rows can be None to process all rows.
 240         """
 241         new_times = []
 242         new_frames = []
 243         rows = rows or range(len(self.times))
 244         for row in rows:
 245             show = self.calc.add_seconds_to_time(self.times[row][SHOW], count)
 246             hide = self.calc.add_seconds_to_time(self.times[row][HIDE], count)
 247             time, frame = self.expand_times(show, hide)
 248             new_times.append(time)
 249             new_frames.append(frame)
 250 
 251         self.replace_positions(rows, new_times, new_frames, register=register)
 252         self.set_action_description(register, _("Shifting times"))

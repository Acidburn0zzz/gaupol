  41     def _get_main_props(self, page):
  42         """Get path, format, encoding, newline of page's main file."""
  43 
  44         if page.project.main_file is not None:
  45             return [
  46                 page.project.main_file.path,
  47                 page.project.main_file.format,
  48                 page.project.main_file.encoding,
  49                 page.project.main_file.newline,]
  50         return [None, None, None, None]
  51 
  52     def _get_translation_props(self, page):
  53         """Get path, format, encoding, newline of page's translation file."""
  54 
  55         if page.project.tran_file is not None:
  56             return [
  57                 page.project.tran_file.path,
  58                 page.project.tran_file.format,
  59                 page.project.tran_file.encoding,
  60                 page.project.tran_file.newline,]
  61         if page.project.main_file is not None:
  62             return [
  63                 None,
  64                 page.project.main_file.format,
  65                 page.project.main_file.encoding,
  66                 page.project.main_file.newline,]
  67         return [None, None, None, None]
  68 
  69     def _save_document(self, page, doc, props):
  70         """Save document.
  71 
  72         Raise Default if something goes wrong.
  73         """
  74         util.set_cursor_busy(self.window)
  75         try:
  76             page.project.save(doc, props)
  77         except IOError, (no, message):
  78             util.set_cursor_normal(self.window)
  79             basename = os.path.basename(props[0])
  80             self._show_io_error_dialog(basename, message)
  81             raise Default
  82         except UnicodeError:
  83             util.set_cursor_normal(self.window)
  84             basename = os.path.basename(props[0])
  85             self._show_encoding_error_dialog(basename, props[2])
  86             raise Default
  87         finally:
  88             util.set_cursor_normal(self.window)
  89 
  90     @util.gc_collected
  91     def _select_file(self, title, props):
  92         """Select a file and return path, format, encoding, newline."""
  93 
  94         path, format, encoding, newline = props
  95         props = [None, None, None, None]
  96         util.set_cursor_busy(self.window)
  97         dialog = SaveDialog(title, self.window)
  98         dialog.set_name(path)
  99         dialog.set_format(format)
 100         dialog.set_encoding(encoding)
 101         dialog.set_newline(newline)
 102         util.set_cursor_normal(self.window)
 103         response = self.run_dialog(dialog)
 104         if response == gtk.RESPONSE_OK:
 105             props[0] = dialog.get_filename()
 106             props[1] = dialog.get_format()
 107             props[2] = dialog.get_encoding()
 108             props[3] = dialog.get_newline()
 109         dialog.destroy()
 110         while gtk.events_pending():
 111             gtk.main_iteration()
 112         return props
 113 
 114     def _show_encoding_error_dialog(self, basename, codec):
 115         """Show an error dialog after failing to decode file."""
 116 
 117         codec = enclib.get_display_name(codec)
 118         fields = {'filename': basename, 'codec': codec}
 119         title = _('Failed to encode file "%(filename)s" '
 120             'with codec "%(codec)s"') % fields
 121         message = _("Please try to save the file with "
 122             "a different character encoding.")
 123         dialog = ErrorDialog(self.window, title, message)
 124         dialog.add_button(gtk.STOCK_OK, gtk.RESPONSE_OK)
 125         self.flash_dialog(dialog)
 126 
 127     def _show_io_error_dialog(self, basename, message):
 128         """Show an error dialog after failing to write file."""
 129 
 130         title = _('Failed to save file "%s"') % basename
 131         message = _("%s.") % message
 132         dialog = ErrorDialog(self.window, title, message)
 133         dialog.add_button(gtk.STOCK_OK, gtk.RESPONSE_OK)
 134         self.flash_dialog(dialog)
 135 
 136     def on_save_all_documents_activate(self, *args):
 137         """Save all open documents."""
 138 
 139         save_main = util.ignore_exceptions(Default)(self.save_main)
 140         save_tran = util.ignore_exceptions(Default)(self.save_translation)
 141         for page in self.pages:
 142             save_main(page)
 143             if page.project.tran_active:
 144                 save_tran(page)
 145         self.push_message(_("Saved all open documents"))
 146         self.update_gui()
 147

 180     def save_main(self, page):
 181         """Save the main document of page.
 182 
 183         Raise Default if something goes wrong.
 184         """
 185         props = self._get_main_props(page)
 186         if None in props:
 187             return self.save_main_as(page)
 188         self._save_document(page, cons.DOCUMENT.MAIN, props)
 189         self.push_message(_("Saved main document"))
 190 
 191     def save_main_as(self, page):
 192         """Save the main document of page to a selected file.
 193 
 194         Raise Default if something goes wrong.
 195         """
 196         props = self._get_main_props(page)
 197         if props[0] is None:
 198             props[0] = page.untitle
 199         orig_format = props[1]
 200         props = self._select_file(_("Save As"), props)
 201         if None in props:
 202             raise Default
 203         self._save_document(page, cons.DOCUMENT.MAIN, props)
 204         if orig_format is not None:
 205             if props[1] != orig_format:
 206                 page.reload_view_all()
 207         self.add_to_recent_files(props[0], cons.DOCUMENT.MAIN)
 208         basename = os.path.basename(props[0])
 209         self.push_message(_('Saved main document as "%s"') % basename)
 210 
 211     def save_translation(self, page):
 212         """Save the translation document of page.
 213 
 214         Raise Default if something goes wrong.
 215         """
 216         props = self._get_translation_props(page)
 217         if None in props:
 218             return self.save_translation_as(page)
 219         self._save_document(page, cons.DOCUMENT.TRAN, props)
 220         self.push_message(_("Saved translation document"))
 221 
 222     def save_translation_as(self, page):
 223         """Save the translation document of page to a selected file.
 224 
 225         Raise Default if something goes wrong.
 226         """
 227         props = self._get_translation_props(page)
 228         if props[0] is None:
 229             props[0] = page.get_translation_corename()
 230         orig_format = props[1]
 231         props = self._select_file(_("Save Translation As"), props)
 232         if None in props:
 233             raise Default
 234         self._save_document(page, cons.DOCUMENT.TRAN, props)
 235         if orig_format is not None:
 236             if props[1] != orig_format:
 237                 rows = range(len(page.project.times))
 238                 page.reload_view(rows, [TTXT])
 239         self.add_to_recent_files(props[0], cons.DOCUMENT.TRAN)
 240         basename = os.path.basename(props[0])
 241         self.push_message(_('Saved translation document as "%s"') % basename)

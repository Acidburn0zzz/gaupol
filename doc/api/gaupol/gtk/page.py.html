"""<b>Path:</b> $path

  49 class Page(Observable):
  50 
  51     """User interface for a single project.
  52 
  53     Instance variables:
  54 
  55         edit_mode:  MODE constant, currently used in the view
  56         project:    The associated Project
  57         tab_label:  gtk.Label used in the tab_widget
  58         tab_widget: Widget to use in a notebook tab
  59         tooltips:   Markup-enabled gtk.Tooltips
  60         untitle:    Title used if the main document is unsaved
  61         view:       The associated View
  62 
  63     Signals:
  64 
  65         close-request (page)
  66     """
  67 
  68     _signals = ["close-request"]
  69 
  70     def __init__(self, counter=0):
  71 
  72         Observable.__init__(self)
  73 
  74         limit = conf.editor.limit_undo
  75         levels = conf.editor.undo_levels
  76         undo_levels = (levels if limit else None)
  77 
  78         self.edit_mode  = conf.editor.mode
  79         self.project    = Project(conf.editor.framerate, undo_levels)
  80         self.tab_label  = None
  81         self.tab_widget = None
  82         self.tooltips   = None
  83         self.untitle    = _("Untitled %d") % counter
  84         self.view       = View(conf.editor.mode)
  85 
  86         self._init_tooltips()
  87         self._init_widgets()
  88         self._init_signal_handlers()
  89         self.update_tab_label()
  90 
  91     def _get_positions(self):
  92         """Return times or frames depending on edit mode."""
  93 
  94         if self.edit_mode == cons.MODE.TIME:
  95             return self.project.times
  96         if self.edit_mode == cons.MODE.FRAME:
  97             return self.project.frames
  98         raise ValueError
  99 
 100     def _init_signal_handlers(self):
 101         """Initialize signal handlers."""
 102 
 103         util.connect(self, "project", "main-file-opened")
 104         util.connect(self, "project", "main-texts-changed")
 105         util.connect(self, "project", "positions-changed")
 106         util.connect(self, "project", "subtitles-changed")
 107         util.connect(self, "project", "subtitles-inserted")
 108         util.connect(self, "project", "subtitles-removed")
 109         util.connect(self, "project", "translation-file-opened")
 110         util.connect(self, "project", "translation-texts-changed")
 111 
 112         def update_label(*args):
 113             self.update_tab_label()
 114         self.project.connect("main-file-opened", update_label)
 115         self.project.connect("main-file-saved", update_label)
 116         self.project.connect("notify::main_changed", update_label)
 117         self.project.connect("notify::tran_active", update_label)
 118         self.project.connect("notify::tran_changed", update_label)
 119 
 120         def update_levels(*args):
 121             limit = conf.editor.limit_undo
 122             limit = (conf.editor.undo_levels if limit else None)
 123             self.project.undo_levels = limit
 124         conf.editor.connect("notify::limit_undo", update_levels)
 125         conf.editor.connect("notify::undo_levels", update_levels)
 126 
 127     def _init_tooltips(self):
 128         """Initialize the markup-enabled tooltips."""
 129 
 130         def revert(label, *args):
 131             label.set_use_markup(True)
 132         self.tooltips = gtk.Tooltips()
 133         self.tooltips.force_window()
 134         label = self.tooltips.tip_label
 135         if label is not None:
 136             # pylint: disable-msg=E1101
 137             label.set_use_markup(True)
 138             label.connect("notify::use-markup", revert)
 139 
 140     def _init_widgets(self):
 141         """Initialize the widgets to use in a notebook tab."""
 142 
 143         self.tab_label = gtk.Label()
 144         self.tab_label.props.xalign = 0
 145         self.tab_label.set_ellipsize(pango.ELLIPSIZE_END)
 146         self.tab_label.set_max_width_chars(24)
 147         event_box = gtk.EventBox()
 148         event_box.add(self.tab_label)
 149         self.tooltips.set_tip(event_box, self.untitle)
 150         method = self._on_tab_event_box_enter_notify_event
 151         event_box.connect("enter-notify-event", method)
 152 
 153         def request_close(*args):
 154             self.emit("close-request")
 155         image = gtk.image_new_from_stock(gtk.STOCK_CLOSE, gtk.ICON_SIZE_MENU)
 156         width, height = image.size_request()
 157         button = gtk.Button()
 158         button.add(image)
 159         button.set_relief(gtk.RELIEF_NONE)
 160         button.set_focus_on_click(False)
 161         button.set_name("gaupol-tab-close-button")
 162         button.set_size_request(width + 2, height + 2)
 163         button.connect("clicked", request_close)
 164         self.tooltips.set_tip(button, _("Close project"))
 165 
 166         self.tab_widget = gtk.HBox(False, 4)
 167         self.tab_widget.pack_start(event_box, True, True, 0)
 168         self.tab_widget.pack_start(button, False, False, 0)
 169         self.tab_widget.set_data("button", button)
 170         self.tab_widget.set_data("event_box", event_box)
 171         self.tab_widget.show_all()
 172 
 173     def _on_project_main_file_opened(self, project, main_file):
 174         """Reload the entire view."""
 175 
 176         self.reload_view_all()
 177 
 178     def _on_project_main_texts_changed(self, project, rows):
 179         """Reload main texts in rows."""
 180 
 181         self.reload_view(rows, [MTXT])
 182         self.view.select_rows(rows)
 183 
 184     def _on_project_positions_changed(self, project, rows):
 185         """Reload positions in rows."""
 186 
 187         self.reload_view(rows, [SHOW, HIDE, DURN])
 188         self.view.select_rows(rows)
 189 
 190     def _on_project_subtitles_changed(self, project, rows):
 191         """Reload subtitles in rows."""
 192 
 193         self.reload_view(rows, [SHOW, HIDE, DURN, MTXT, TTXT])
 194         self.view.select_rows(rows)
 195 
 196     def _on_project_subtitles_inserted(self, project, rows):
 197         """Insert rows to the list store."""
 198 
 199         store = self.view.get_model()
 200         positions = self._get_positions()
 201         mains = self.project.main_texts
 202         trans = self.project.tran_texts
 203         for row in sorted(rows):
 204             item = [row + 1] + positions[row] + [mains[row], trans[row]]
 205             store.insert(row, item)
 206         self.view.set_focus(rows[0])
 207         self.view.select_rows(rows)
 208 
 209     def _on_project_subtitles_removed(self, project, rows):
 210         """Remove rows from the list store."""
 211 
 212         store = self.view.get_model()
 213         for row in reversed(sorted(rows)):
 214             store.remove(store.get_iter(row))
 215         if self.project.times:
 216             row = min(rows[0], len(self.project.times) - 1)
 217             col = self.view.get_focus()[1]
 218             self.view.set_focus(row, col)
 219 
 220     def _on_project_translation_file_opened(self, project, tran_file):
 221         """Reload the entire view."""
 222 
 223         self.reload_view_all()
 224 
 225     def _on_project_translation_texts_changed(self, project, rows):
 226         """Reload translation texts in rows."""
 227 
 228         self.reload_view(rows, [TTXT])
 229         self.view.select_rows(rows)
 230 
 231     def _on_tab_event_box_enter_notify_event(self, *args):
 232         """Update the text in the tooltip."""
 233 
 234         main_file = self.project.main_file
 235         if main_file is not None:
 236             event_box = self.tab_widget.get_data("event_box")
 237             tooltip = string.Template(_TOOLTIP).substitute(
 238                 path=main_file.path,
 239                 format=main_file.format.display_name,
 240                 encoding=enclib.get_long_name(main_file.encoding),
 241                 newline=main_file.newline.display_name,)
 242             self.tooltips.set_tip(event_box, tooltip)
 243 
 244     def assert_store(self):
 245         """Assert that store's data matches project's."""
 246 
 247         store = self.view.get_model()
 248         positions = self._get_positions()
 249         mains = self.project.main_texts
 250         trans = self.project.tran_texts
 251         for i in range(len(store)):
 252             assert store[i][1] == positions[i][0]
 253             assert store[i][2] == positions[i][1]
 254             assert store[i][3] == positions[i][2]
 255             assert store[i][4] == mains[i]
 256             assert store[i][5] == trans[i]
 257 
 258     def document_to_text_column(self, document):
 259         """Translate document index to text column index."""
 260 
 261         if document == cons.DOCUMENT.MAIN:
 262             return MTXT
 263         if document == cons.DOCUMENT.TRAN:
 264             return TTXT
 265         raise ValueError

 267     def get_main_basename(self):
 268         """Get the basename of the main document."""
 269 
 270         if self.project.main_file is not None:
 271             # pylint: disable-msg=E1101
 272             return os.path.basename(self.project.main_file.path)
 273         return self.untitle
 274 
 275     def get_main_corename(self):
 276         """Get the basename of the main document without extension."""
 277 
 278         # pylint: disable-msg=E1101
 279         if self.project.main_file is None:
 280             return self.untitle
 281         basename = os.path.basename(self.project.main_file.path)
 282         extension = self.project.main_file.format.extension
 283         if basename.endswith(extension):
 284             basename = basename[0:-len(extension)]
 285         return basename
 286 
 287     def get_translation_basename(self):
 288         """Get the basename of the translation document."""
 289 
 290         if self.project.tran_file is not None:
 291             # pylint: disable-msg=E1101
 292             return os.path.basename(self.project.tran_file.path)
 293         basename = self.get_main_corename()
 294         return _("%s translation") % basename
 295 
 296     def get_translation_corename(self):
 297         """Get basename of translation document without extension."""
 298 
 299         # pylint: disable-msg=E1101
 300         if self.project.tran_file is None:
 301             return self.get_translation_basename()
 302         basename = os.path.basename(self.project.tran_file.path)
 303         extension = self.project.tran_file.format.extension
 304         if basename.endswith(extension):
 305             basename = basename[0:-len(extension)]
 306         return basename

 308     def reload_view(self, rows, cols):
 309         """Reload the view in columns and rows."""
 310 
 311         store = self.view.get_model()
 312         for col in cols:
 313             if col in (SHOW, HIDE, DURN):
 314                 positions = self._get_positions()
 315                 get_data = lambda row: positions[row][col - 1]
 316             elif col == MTXT:
 317                 texts = self.project.main_texts
 318                 get_data = lambda row: texts[row]
 319             elif col == TTXT:
 320                 texts = self.project.tran_texts
 321                 get_data = lambda row: texts[row]
 322             for row in rows:
 323                 store[row][col] = get_data(row)
 324         # FIX: REMOVE WHEN THIS WORKS.
 325         self.assert_store()
 326 
 327     def reload_view_all(self):
 328         """Clear and repopulate the entire view."""
 329 
 330         store = self.view.get_model()
 331         self.view.set_model(None)
 332         store.clear()
 333         positions = self._get_positions()
 334         mains = self.project.main_texts
 335         trans = self.project.tran_texts
 336         for row in range(len(self.project.times)):
 337             item = [row + 1] + positions[row] + [mains[row], trans[row]]
 338             store.append(item)
 339         self.view.set_model(store)
 340         # FIX: REMOVE WHEN THIS WORKS.
 341         self.assert_store()
 342 
 343     def text_column_to_document(self, col):
 344         """Translate text column constant to document constant."""
 345 
 346         if col == MTXT:
 347             return cons.DOCUMENT.MAIN
 348         if col == TTXT:
 349             return cons.DOCUMENT.TRAN
 350         raise ValueError
 351 
 352     def update_tab_label(self):
 353         """Update the notebook tab label and return the title."""
 354 
 355         title = self.get_main_basename()
 356         if self.project.main_changed:
 357             title = "*%s" % title
 358         elif self.project.tran_active and self.project.tran_changed:
 359             title = "*%s" % title
 360         self.tab_label.set_text(title)
 361         return title

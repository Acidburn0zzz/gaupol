  19 """Internal clipboard."""
  20 
  21 
  22 class Clipboard(object):
  23 
  24     """Internal clipboard.
  25 
  26     Instance variables:
  27 
  28         data: List of strings or Nones
  29 
  30     Nones in data express that those items are to be skipped.
  31     """
  32 
  33     def __init__(self):
  34 
  35         self.data = []
  36 
  37     def get_data_as_string(self):
  38         """Get the data as a string."""
  39 
  40         return "\n\n".join(list(x or "" for x in self.data))
